### Step 2: Bug 11-12 тАФ Fix `_detect_content_hierarchy()` (lines 897-922)

**Current:** Threshold `max_size * 0.8` тАФ marks anything within 80% of max as heading. Uniform sizes all become headings.

**Fix:** Replace with median-based approach:
```python
def _detect_content_hierarchy(texts: list[TextBlock]) -> list[TextBlock]:
    if not texts:
        return texts
    sizes = [t.font_size for t in texts if t.font_size is not None]
    if not sizes or len(set(sizes)) == 1:
        return texts  # Bug 12: uniform sizes тЖТ no headings
    median_size = statistics.median(sizes)
    threshold = median_size * 1.3  # Bug 11: 1.3x median
    return [
        dataclasses.replace(t, is_heading=(t.font_size is not None and t.font_size >= threshold))
        if t.font_size is not None and t.font_size >= threshold
        else t
        for t in texts
    ]
```

Note: `TextBlock` is frozen тАФ use `dataclasses.replace()` (also fixes Bug 17 pattern).

def _has_large_image_child(node: DesignNode, *, max_depth: int = 2) -> bool:
    if max_depth <= 0:
        return False
    for child in node.children:
        if child.type == DesignNodeType.IMAGE and child.width and child.width >= 200:
            return True
        if child.type in _FRAME_TYPES and _has_large_image_child(child, max_depth=max_depth - 1):

### Step 6: Bug 16 тАФ Fix `_detect_position_columns()` column grouping (lines 712-741)

**Current:** Dict-based Y-grouping with tolerance тАФ order-dependent because first-match wins.

**Fix:** Sort children by Y first, then use greedy non-overlapping bands. Sort each row by X:
```python
frame_children.sort(key=lambda c: (c.y if c.y is not None else 0.0, c.x if c.x is not None else 0.0))
y_groups: list[list[DesignNode]] = []
for child in frame_children:
    if child.y is None:
        continue
    placed = False
    for group in y_groups:
        ref_y = group[0].y
        if ref_y is not None and abs(child.y - ref_y) <= _Y_TOLERANCE:
            group.append(child)
            placed = True
            break
    if not placed:
        y_groups.append([child])

# Find largest row
max_group = max(y_groups, key=len) if y_groups else []
if len(max_group) < 2:
    return []
max_group.sort(key=lambda c: c.x if c.x is not None else 0.0)
return _build_column_groups(max_group)
```

def _extract_texts(node: DesignNode, *, exclude_node_ids: set[str] | None = None) -> list[TextBlock]:
    results: list[TextBlock] = []
    _walk_for_texts(node, results, exclude_node_ids=exclude_node_ids)
    return results

def _walk_for_texts(node: DesignNode, results: list[TextBlock], *, exclude_node_ids: set[str] | None = None) -> None:
    if exclude_node_ids and node.id in exclude_node_ids:
        return  # Skip button subtrees
    if node.type == DesignNodeType.TEXT and node.text_content:
        results.append(TextBlock(...))
    for child in node.children:
        _walk_for_texts(child, results, exclude_node_ids=exclude_node_ids)

def _collect_button_node_ids(node: DesignNode, *, extra_hints: list[str] | None = None) -> set[str]:
