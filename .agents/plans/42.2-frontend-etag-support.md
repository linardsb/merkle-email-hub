# Plan: 42.2 Frontend ETag Support in auth-fetch

## Context

The backend (42.1) now sends `ETag` + `Cache-Control: no-cache, must-revalidate` on polling endpoints. In browsers, fetch() handles ETags automatically via the HTTP cache — 304 responses never surface to JavaScript. However, Next.js SSR uses Node.js `fetch()` which lacks a backing HTTP cache, so a 304 response surfaces with an empty body, causing `res.json()` to throw a `SyntaxError`. The SWR fetcher must handle this edge case gracefully.

**SSR detection:** `auth-fetch.ts:47` uses `typeof window === "undefined"` for server-side auth. This is where 304 can surface.

## Implementation Steps

### Step 1: Update swr-fetcher.ts

Replace `swr-fetcher.ts` content — add 304 handling between the `authFetch` call and the `res.ok` check:

```typescript
import { authFetch } from "./auth-fetch";
import { ApiError } from "./api-error";

/**
 * SWR fetcher that automatically injects JWT auth headers.
 * Usage: useSWR("/api/v1/items", fetcher)
 *
 * ## 304 Not Modified handling
 *
 * When the backend sends ETag headers, browsers handle 304 responses
 * transparently via the HTTP cache — the fetcher never sees them.
 * In SSR (Node.js fetch), 304 surfaces with an empty body. Returning
 * undefined tells SWR to keep its cached data, which is the correct
 * semantic for "data unchanged".
 */
export async function fetcher<T>(url: string): Promise<T> {
  const res = await authFetch(url);

  // 304 Not Modified — data unchanged. In SSR, this surfaces as an
  // empty-body response. Return undefined so SWR keeps cached data.
  if (res.status === 304) {
    return undefined as unknown as T;
  }

  if (!res.ok) {
    let message = "Request failed";
    let code: string | undefined;

    try {
      const body = await res.json();
      if (body.error) message = body.error;
      if (body.type) code = body.type;
    } catch {
      message = res.statusText || message;
    }

    throw new ApiError(res.status, message, code);
  }

  return res.json();
}
```

### Step 2: Create fetcher unit tests

`cms/apps/web/src/lib/__tests__/swr-fetcher.test.ts`:

```typescript
import { describe, it, expect, vi, beforeEach } from "vitest";

// Mock authFetch at module level
const mockAuthFetch = vi.fn();
vi.mock("@/lib/auth-fetch", () => ({
  authFetch: (...args: unknown[]) => mockAuthFetch(...args),
}));

// Import after mock
const { fetcher } = await import("../swr-fetcher");

/** Helper to create a minimal Response-like object */
function makeResponse(status: number, body?: unknown): Response {
  return {
    status,
    ok: status >= 200 && status < 300,
    statusText: status === 304 ? "Not Modified" : "OK",
    headers: new Headers(),
    json: body !== undefined
      ? vi.fn().mockResolvedValue(body)
      : vi.fn().mockRejectedValue(new SyntaxError("Unexpected end of JSON input")),
  } as unknown as Response;
}

beforeEach(() => {
  mockAuthFetch.mockReset();
});

describe("fetcher", () => {
  it("returns parsed JSON on 200", async () => {
    const data = { id: 1, name: "test" };
    mockAuthFetch.mockResolvedValue(makeResponse(200, data));

    const result = await fetcher("/api/v1/items");
    expect(result).toEqual(data);
    expect(mockAuthFetch).toHaveBeenCalledWith("/api/v1/items");
  });

  it("returns undefined on 304 Not Modified (SSR path)", async () => {
    mockAuthFetch.mockResolvedValue(makeResponse(304));

    const result = await fetcher("/api/v1/items");
    expect(result).toBeUndefined();
  });

  it("does not call res.json() on 304", async () => {
    const res = makeResponse(304);
    mockAuthFetch.mockResolvedValue(res);

    await fetcher("/api/v1/items");
    expect(res.json).not.toHaveBeenCalled();
  });

  it("throws ApiError on non-ok response", async () => {
    const errorBody = { error: "Not found", type: "NOT_FOUND" };
    const res = makeResponse(404, errorBody);
    mockAuthFetch.mockResolvedValue(res);

    await expect(fetcher("/api/v1/items")).rejects.toThrow();
    await expect(fetcher("/api/v1/items")).rejects.toMatchObject({
      status: 404,
      message: "Not found",
      code: "NOT_FOUND",
    });
  });

  it("falls back to statusText when error body is unparseable", async () => {
    const res = {
      status: 500,
      ok: false,
      statusText: "Internal Server Error",
      headers: new Headers(),
      json: vi.fn().mockRejectedValue(new SyntaxError("bad json")),
    } as unknown as Response;
    mockAuthFetch.mockResolvedValue(res);

    await expect(fetcher("/api/v1/items")).rejects.toMatchObject({
      status: 500,
      message: "Internal Server Error",
    });
  });
});
```
