// src/notes.rs
//! Lightweight cross-agent note storage via append-only JSONL files.
//!
//! Agents store project-scoped observations that other agents can recall.
//! Zero dependencies beyond std + serde. Storage: `.dcg/agent_notes.jsonl`.

use chrono::{DateTime, Utc};
use serde::{Deserialize, Serialize};
use std::fs::{self, File, OpenOptions};
use std::io::{self, BufRead, BufReader, Write};
use std::path::{Path, PathBuf};

/// Maximum value length in characters.
const MAX_VALUE_LEN: usize = 1024;
/// Maximum key length in characters.
const MAX_KEY_LEN: usize = 128;
/// Maximum notes per project file.
const MAX_NOTES_PER_PROJECT: usize = 500;
/// Maximum results returned by recall.
const MAX_RECALL_LIMIT: usize = 50;

/// A single cross-agent observation.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct AgentNote {
    pub timestamp: DateTime<Utc>,
    pub agent: String,
    pub key: String,
    pub value: String,
    pub project: String,
}

/// Errors from note operations.
#[derive(Debug, thiserror::Error)]
pub enum NoteError {
    #[error("key too long (max {MAX_KEY_LEN} chars)")]
    KeyTooLong,
    #[error("value too long (max {MAX_VALUE_LEN} chars)")]
    ValueTooLong,
    #[error("key must not be empty")]
    EmptyKey,
    #[error("value must not be empty")]
    EmptyValue,
    #[error("project limit reached ({MAX_NOTES_PER_PROJECT} notes)")]
    ProjectLimitReached,
    #[error("IO error: {0}")]
    Io(#[from] io::Error),
    #[error("JSON error: {0}")]
    Json(#[from] serde_json::Error),
}

/// Resolve the JSONL file path for a project.
///
/// Default: `.dcg/agent_notes.jsonl` in the project root.
/// Override via `DCG_NOTES_DIR` env var.
fn notes_file_path(project: &str) -> PathBuf {
    let base = std::env::var("DCG_NOTES_DIR")
        .map(PathBuf::from)
        .unwrap_or_else(|_| {
            // Use working directory / .dcg/
            PathBuf::from(".dcg")
        });
    // Sanitize project name for filesystem safety
    let safe_project = project
        .chars()
        .map(|c| if c.is_alphanumeric() || c == '-' || c == '_' { c } else { '_' })
        .collect::<String>();
    base.join(format!("{safe_project}_notes.jsonl"))
}

/// Read all notes from a project's JSONL file.
fn read_notes(path: &Path) -> Result<Vec<AgentNote>, NoteError> {
    if !path.exists() {
        return Ok(Vec::new());
    }
    let file = File::open(path)?;
    let reader = BufReader::new(file);
    let mut notes = Vec::new();
    for line in reader.lines() {
        let line = line?;
        let trimmed = line.trim();
        if trimmed.is_empty() {
            continue;
        }
        match serde_json::from_str::<AgentNote>(trimmed) {
            Ok(note) => notes.push(note),
            Err(_) => continue, // Skip malformed lines gracefully
        }
    }
    Ok(notes)
}

/// Store a note. Appends to JSONL file with POSIX atomic write (< PIPE_BUF).
///
/// Auto-detects agent identity from the calling environment.
pub fn store_note(
    agent: &str,
    key: &str,
    value: &str,
    project: &str,
) -> Result<AgentNote, NoteError> {
    // Validate inputs
    if key.is_empty() {
        return Err(NoteError::EmptyKey);
    }
    if value.is_empty() {
        return Err(NoteError::EmptyValue);
    }
    if key.len() > MAX_KEY_LEN {
        return Err(NoteError::KeyTooLong);
    }
    if value.len() > MAX_VALUE_LEN {
        return Err(NoteError::ValueTooLong);
    }

    let path = notes_file_path(project);

    // Check project limit
    let existing = read_notes(&path)?;
    if existing.len() >= MAX_NOTES_PER_PROJECT {
        return Err(NoteError::ProjectLimitReached);
    }

    // Create parent directory if needed
    if let Some(parent) = path.parent() {
        fs::create_dir_all(parent)?;
    }

    let note = AgentNote {
        timestamp: Utc::now(),
        agent: agent.to_string(),
        key: key.to_string(),
        value: value.to_string(),
        project: project.to_string(),
    };

    // Atomic append — serialize to single line + newline
    let mut line = serde_json::to_string(&note)?;
    line.push('\n');

    // POSIX guarantees atomic writes for < PIPE_BUF (4096 bytes).
    // Our max note JSON is well under this limit.
    let mut file = OpenOptions::new()
        .create(true)
        .append(true)
        .open(&path)?;
    file.write_all(line.as_bytes())?;

    Ok(note)
}

/// Filter criteria for recalling notes.
#[derive(Debug, Default)]
pub struct RecallFilter {
    pub key: Option<String>,
    pub key_prefix: Option<String>,
    pub agent: Option<String>,
    pub limit: Option<usize>,
}

/// Recall notes matching the given filters, sorted by recency (newest first).
pub fn recall_notes(
    project: &str,
    filter: &RecallFilter,
) -> Result<Vec<AgentNote>, NoteError> {
    let path = notes_file_path(project);
    let mut notes = read_notes(&path)?;

    // Apply filters
    if let Some(ref key) = filter.key {
        notes.retain(|n| n.key == *key);
    }
    if let Some(ref prefix) = filter.key_prefix {
        notes.retain(|n| n.key.starts_with(prefix.as_str()));
    }
    if let Some(ref agent) = filter.agent {
        notes.retain(|n| n.agent == *agent);
    }

    // Sort by recency (newest first)
    notes.sort_by(|a, b| b.timestamp.cmp(&a.timestamp));

    // Apply limit
    let limit = filter.limit.unwrap_or(MAX_RECALL_LIMIT).min(MAX_RECALL_LIMIT);
    notes.truncate(limit);

    Ok(notes)
}

/// Count notes in a project file.
pub fn count_notes(project: &str) -> Result<usize, NoteError> {
    let path = notes_file_path(project);
    let notes = read_notes(&path)?;
    Ok(notes.len())
}

## Important Implementation Notes

1. **Read `mcp.rs` first** — The exact patterns for `CallToolResult`, `TextContent`, tool registration, and param extraction MUST match the existing code. The pseudocode above is structural — adapt to actual types.

2. **Agent Display impl** — Check how `Agent` enum implements `Display` or `ToString`. Use that for the agent name in stored notes. If it uses snake_case (e.g., `claude_code`), keep that convention.

3. **Params extraction** — The existing `handle_call_tool_request` likely destructures `request.params.arguments` into a `serde_json::Map`. Follow the exact same pattern.

4. **JSONL atomicity** — The note JSON + newline must be < 4096 bytes (POSIX PIPE_BUF) for atomic writes. With max 1024-char value + 128-char key + metadata, worst case is ~1300 bytes — well within the limit.

5. **No async needed** — File I/O for JSONL is fast enough synchronously. The MCP handler can call `notes::store_note()` and `notes::recall_notes()` directly without spawning async tasks.

6. **`.dcg/` directory** — Must be gitignored. Add `.dcg/` to `.gitignore` in projects that use dcg if not already present.
