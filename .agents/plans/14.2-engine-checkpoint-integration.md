# Plan: 14.2 Engine Checkpoint Integration

## Context
Task 14.1 created the checkpoint storage layer (`CheckpointStore` protocol, `PostgresCheckpointStore`, `serialize_run`/`restore_run`). Task 14.2 wires it into `BlueprintEngine.run()` so that a checkpoint is saved after each successful node execution. This is fire-and-forget (errors logged, never crash the pipeline) and opt-in via `settings.blueprint.checkpoints_enabled`.

- `app/ai/blueprints/engine.py` — Accept `CheckpointStore` in `__init__`, save checkpoint after each node completes
- `app/ai/blueprints/service.py` — Create `PostgresCheckpointStore` when enabled, pass to engine
- `app/ai/blueprints/tests/test_checkpoint.py` — Add engine integration tests

### Step 2: Add `_save_checkpoint` helper method to `BlueprintEngine`

Add a private fire-and-forget method that wraps `serialize_run` + `store.save()`:

```python
async def _save_checkpoint(self, run: BlueprintRun, node_name: str, node_index: int) -> None:
    """Fire-and-forget checkpoint save after a successful node."""
    if self._checkpoint_store is None:
        return
    try:
        from app.ai.blueprints.checkpoint import serialize_run  # noqa: F811

        data = serialize_run(
            run,
            node_name=node_name,
            node_index=node_index,
            blueprint_name=self._definition.name,
        )
        await self._checkpoint_store.save(data)
    except Exception:
        logger.warning(
            "blueprint.checkpoint_save_failed",
            node=node_name,
            run_id=run.run_id,
            exc_info=True,
        )
```

### Step 3: Call `_save_checkpoint` in the main loop

In `BlueprintEngine.run()`, after the `logger.info("blueprint.node_completed", ...)` call at line ~378 and before `current_node_name = self._resolve_next_node(...)`, add the checkpoint save call. The `node_index` is `len(run.progress) - 1` since progress was just appended.

Insert after line 385 (the `blueprint.node_completed` log):

```python
            # Fire-and-forget checkpoint after successful node
            await self._save_checkpoint(run, current_node_name, len(run.progress) - 1)
```

This saves checkpoints for ALL completed nodes (success, failed, skipped — but skipped nodes `continue` before reaching this point, which is fine since skipped nodes have no meaningful state change). The checkpoint captures the run state **after** the node's results have been applied.

### Step 5: Add engine integration tests

Add to `app/ai/blueprints/tests/test_checkpoint.py`:

```python
from unittest.mock import AsyncMock, patch


# ── Engine integration tests ──


class TestEngineCheckpointIntegration:
    """Verify engine calls checkpoint store after each node."""

    @pytest.fixture
    def mock_store(self) -> AsyncMock:
        store = AsyncMock()
        store.save = AsyncMock()
        return store

    @pytest.fixture
    def simple_definition(self) -> "BlueprintDefinition":
        """Two-node blueprint: entry → terminal."""
        from app.ai.blueprints.engine import BlueprintDefinition, Edge
        from app.ai.blueprints.protocols import BlueprintNode, NodeContext, NodeResult

        class StubNode:
            name = "stub"
            node_type = "deterministic"

            async def execute(self, context: NodeContext) -> NodeResult:
                return NodeResult(status="success", html="<p>done</p>")

        class StubNode2:
            name = "stub2"
            node_type = "deterministic"

            async def execute(self, context: NodeContext) -> NodeResult:
                return NodeResult(status="success", html="<p>final</p>")

        return BlueprintDefinition(
            name="test_bp",
            nodes={"entry": StubNode(), "finish": StubNode2()},
            edges=[Edge(from_node="entry", to_node="finish", condition="success")],
            entry_node="entry",
        )

    @pytest.mark.asyncio
    async def test_checkpoint_saved_after_each_node(
        self, mock_store: AsyncMock, simple_definition: "BlueprintDefinition"
    ) -> None:
        engine = BlueprintEngine(simple_definition, checkpoint_store=mock_store)
        run = await engine.run(brief="test")

        assert run.status == "completed"
        # Two nodes executed → two checkpoint saves
        assert mock_store.save.call_count == 2

    @pytest.mark.asyncio
    async def test_checkpoint_contains_correct_node_info(
        self, mock_store: AsyncMock, simple_definition: "BlueprintDefinition"
    ) -> None:
        engine = BlueprintEngine(simple_definition, checkpoint_store=mock_store)
        await engine.run(brief="test")

        # First checkpoint: entry node at index 0
        first_call = mock_store.save.call_args_list[0]
        data = first_call[0][0]  # positional arg
        assert data.node_name == "entry"
        assert data.node_index == 0
        assert data.blueprint_name == "test_bp"

        # Second checkpoint: finish node at index 1
        second_call = mock_store.save.call_args_list[1]
        data2 = second_call[0][0]
        assert data2.node_name == "finish"
        assert data2.node_index == 1

    @pytest.mark.asyncio
    async def test_checkpoint_failure_does_not_crash_pipeline(
        self, simple_definition: "BlueprintDefinition"
    ) -> None:
        """Store.save() raising should not abort the run."""
        failing_store = AsyncMock()
        failing_store.save = AsyncMock(side_effect=RuntimeError("DB gone"))

        engine = BlueprintEngine(simple_definition, checkpoint_store=failing_store)
        run = await engine.run(brief="test")

        # Pipeline completes despite checkpoint failures
        assert run.status == "completed"
        assert len(run.progress) == 2

    @pytest.mark.asyncio
    async def test_no_checkpoint_when_store_is_none(
        self, simple_definition: "BlueprintDefinition"
    ) -> None:
        """No errors when checkpoint_store is not provided."""
        engine = BlueprintEngine(simple_definition, checkpoint_store=None)
        run = await engine.run(brief="test")
        assert run.status == "completed"
```
