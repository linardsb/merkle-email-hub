# Plan: 36.5 AI-Powered HTML Reverse Engineering Adapter

## Context

Enterprises have thousands of legacy HTML email templates that need importing into the platform as editable `EmailDesignDocument` objects. The import annotator agent (Phase 32.3) detects section boundaries and builder patterns but does NOT extract structured data. This adapter bridges the gap: arbitrary email HTML → structured `EmailDesignDocument` with sections, content, and design tokens.

### Step 2: DOM Parser (`dom_parser.py`)

Section extraction from HTML DOM tree.

```python
# app/design_sync/html_import/dom_parser.py
from __future__ import annotations
from dataclasses import dataclass
from lxml.html import HtmlElement, fromstring as parse_html
from app.design_sync.email_design_document import (
    DocumentSection, DocumentText, DocumentImage, DocumentButton,
    DocumentColumn, DocumentPadding,
)

@dataclass(frozen=True)
class ParsedEmail:
    """Result of DOM parsing."""
    sections: list[DocumentSection]
    container_width: int
    has_mso_conditionals: bool
    has_dark_mode: bool
    style_blocks: list[str]        # raw <style> contents
    meta_charset: str | None
    preheader_text: str | None

def parse_email_dom(html: str) -> ParsedEmail:
    """Main entry: HTML string → ParsedEmail with extracted sections.

    Pipeline:
    1. Parse with lxml.html.fromstring()
    2. Detect container width from outermost table (width attr or max-width style)
    3. Find content table (skip MSO wrappers via comment detection)
    4. Walk top-level <tr> rows → candidate sections
    5. For each section: extract texts, images, buttons, columns, styling
    6. Check for data-section-id/data-section-type (import annotator annotations)
    """

def _find_content_table(root: HtmlElement) -> HtmlElement | None:
    """Find the outermost content table, skipping MSO conditional wrappers.
    Strategy: find <table> with width="600" or max-width:600px or role="presentation"
    with the most nested content.
    """

def _extract_section_rows(table: HtmlElement) -> list[HtmlElement]:
    """Get top-level <tr> elements from content table.
    If import annotator annotations exist (data-section-id), use those boundaries instead.
    """

def _extract_texts(element: HtmlElement) -> list[DocumentText]:
    """Recursively extract text content from td/p/h1-h6/span/a elements.
    Parses inline style for: font-family, font-size, font-weight, color, line-height.
    Detects headings: <h1>-<h6> tags or font-size > 20px.
    Skips elements with display:none or visibility:hidden.
    """

def _extract_images(element: HtmlElement) -> list[DocumentImage]:
    """Find all <img> tags → DocumentImage.
    src: validate http/https only (security).
    width/height: from attributes first, then inline style.
    alt: from alt attribute.
    """

def _extract_buttons(element: HtmlElement) -> list[DocumentButton]:
    """Detect buttons via multiple patterns:
    1. <a> with background-color in style
    2. <table> with single <a> child (bulletproof button pattern)
    3. <v:roundrect> (VML Outlook button) — extract href from inner <a>
    4. role="button" attribute
    Returns DocumentButton with text, url, bg_color, text_color, border_radius.
    """

def _detect_columns(row_element: HtmlElement) -> tuple[str, list[DocumentColumn]]:
    """Detect column layout from <td> children.
    1 td → "single", 2 → "two_col", 3 → "three_col", 4+ → "multi_col".
    Also detect display:inline-block column pattern (fluid hybrid).
    Returns (layout_type, list of DocumentColumn with per-column content).
    """

def _extract_background(element: HtmlElement) -> str | None:
    """Extract background color from bgcolor attribute or background-color in style."""

def _extract_padding(element: HtmlElement) -> DocumentPadding | None:
    """Parse padding from inline style on section-level <td>.
    Handles shorthand (4-value, 2-value, 1-value) padding notation.
    """

def _detect_preheader(root: HtmlElement) -> str | None:
    """Find preheader text: hidden div/span with display:none or max-height:0 at top of body."""
```

from __future__ import annotations
import re
from app.design_sync.email_design_document import DocumentSection

def classify_sections(
    sections: list[DocumentSection],
    *, container_position_map: dict[str, float],  # node_id → relative y (0.0-1.0)
) -> list[DocumentSection]:
    """Apply heuristic rules to classify UNKNOWN sections.
    Returns new list with updated section_type fields.

    Rules (applied in order, first match wins):
    1. Has data-section-type attribute (from import annotator) → use directly
    2. First section with logo image (small img, < 200px height) → HEADER
    3. Section with large heading (> 24px) + optional hero image + optional button → HERO
    4. Section with only button(s), no body text → CTA
    5. Section with social media URLs (facebook|twitter|linkedin|instagram|youtube) → SOCIAL
    6. Section with <hr> or 1px-height element → DIVIDER
    7. Section with no content, height-only → SPACER
    8. Last section with "unsubscribe" or "©" or small text (< 12px) → FOOTER
    9. Section in bottom 20% with small text → FOOTER
    10. Remaining sections with text + optional images → CONTENT
    """

_SOCIAL_URL_PATTERN: re.Pattern[str]  # facebook|twitter|x\.com|linkedin|instagram|youtube

async def classify_with_ai_fallback(
    sections: list[DocumentSection],
    *, container_position_map: dict[str, float],
    ai_enabled: bool = True,
) -> list[DocumentSection]:
    """Classify sections: heuristics first, then AI for remaining UNKNOWN.

    1. Run classify_sections() (deterministic)
    2. Collect still-UNKNOWN sections
    3. If ai_enabled and UNKNOWN sections exist:
       a. Bridge DocumentSection → EmailSection (for classify_sections_batch compatibility)
       b. Call classify_sections_batch() from ai_layout_classifier

    """

def _merge_classification(

from __future__ import annotations
import html
from app.core.config import get_settings
from app.core.logging import get_logger
from app.design_sync.email_design_document import (
    EmailDesignDocument, DocumentSource, DocumentLayout, DocumentTokens,
)

logger = get_logger(__name__)

_MAX_HTML_SIZE = 2 * 1024 * 1024  # 2MB

class HtmlImportAdapter:
    """Reverse-engineer arbitrary email HTML into an EmailDesignDocument."""

    async def parse(
        self,
        raw_html: str,
        *,
        use_ai: bool | None = None,  # None = use config default
        source_name: str | None = None,
    ) -> EmailDesignDocument:
        """Main entry point: HTML → EmailDesignDocument.

        Pipeline:
        1. Validate input size (≤ 2MB)
        2. DOM parse → ParsedEmail (sections, container_width, style_blocks)
        3. Classify sections (heuristic + optional AI)
        4. Extract tokens from CSS + inline styles
        5. Assemble EmailDesignDocument
        6. Validate against JSON Schema
        7. Return

        Raises HtmlImportError on validation/parsing failures.
        Logs: design_sync.html_import.parse_completed with section_count, token_count, ai_used.
        """

    def _resolve_ai_enabled(self, use_ai: bool | None) -> bool:
        """Check use_ai param → config fallback (DESIGN_SYNC__HTML_IMPORT_AI_ENABLED)."""

    def _build_document(
        self,
        parsed: ParsedEmail,
        classified_sections: list[DocumentSection],
        tokens: DocumentTokens,
        source_name: str | None,
    ) -> EmailDesignDocument:
        """Assemble final EmailDesignDocument from parsed components."""

**`app/design_sync/routes.py`** — add endpoint:
```python
@router.post(
    "/import/html",
    response_model=HtmlImportResponse,
    dependencies=[Depends(require_role("developer")), Depends(RateLimiter(times=10, seconds=60))],
    summary="Import email HTML as EmailDesignDocument",
)
async def import_html(
    request: HtmlImportRequest,
    current_user: User = Depends(require_role("developer")),
) -> HtmlImportResponse:
    """Reverse-engineer email HTML into structured EmailDesignDocument."""
    adapter = HtmlImportAdapter()
    doc = await adapter.parse(
        request.html,
        use_ai=request.use_ai,
        source_name=request.source_name,
    )
    doc_json = doc.to_json()
    warnings: list[str] = []
    ai_count = sum(1 for s in doc.sections if s.classification_confidence is not None)
    return HtmlImportResponse(
        document=doc_json,
        section_count=len(doc.sections),
        ai_sections_classified=ai_count,
        warnings=warnings,
    )
```
