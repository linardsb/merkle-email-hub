# Plan: 33.10 — Image Asset Import for Design Sync Pipeline

## Context

`node_to_email_html()` renders IMAGE nodes as `<img src="" data-node-id="X">` — empty `src`. The design sync pipeline already has all the pieces to fill these: `DesignAssetService.download_and_store()` downloads from Figma's export API, `_fill_image_urls()` rewrites `src` attributes via regex, and the asset serving endpoint delivers stored files. But the converter output → asset pipeline wiring has gaps. Phase 33.10 closes them and adds dimension preservation + responsive image styles.

### What's Actually Missing

1. **`max-width:{w}px`** on IMAGE nodes — current style has `width:100%;height:auto;` but no `max-width` cap
2. **`ConversionResult.images`** — no field to carry image metadata through the pipeline
3. **Image metadata in `structure_json`** — not persisted in import record
4. **`ImportedImageResponse` schema** — no API response type for images
5. **Tests** — no tests for the image rendering styles or `ConversionResult.images` field

### Critical: `ConversionResult` is `frozen=True`

`@dataclass(frozen=True)` at `converter_service.py:66` — plan's `convert_with_images()` tries `result.html = ...` and `result.images = [...]` which raises `FrozenInstanceError`. **Must use `dataclasses.replace()`** to create new instances.

## Implementation Steps

### Step 1: Update IMAGE node rendering in converter.py

In `node_to_email_html()` (~line 529), update the IMAGE branch:

```python
if node.type == DesignNodeType.IMAGE:
    w_val = int(node.width) if node.width else None
    h_val = int(node.height) if node.height else None
    w = f' width="{w_val}"' if w_val else ""
    h = f' height="{h_val}"' if h_val else ""
    alt = html.escape(node.name or "")
    node_id_attr = f' data-node-id="{html.escape(node.id)}"'
    max_w = f"max-width:{w_val}px;" if w_val else ""
    return (
        f'{pad}<img src="" alt="{alt}"{node_id_attr}{w}{h}'
        f' style="display:block;border:0;outline:none;text-decoration:none;'
        f'-ms-interpolation-mode:bicubic;width:100%;{max_w}height:auto;" />'
    )
```

Key changes: add `max-width:{w}px` for responsive sizing. Rest is already correct.

### Step 2: Add `images` field to `ConversionResult`

```python
"""Tests for image asset pipeline in design sync converter."""
from __future__ import annotations

import pytest
from unittest.mock import AsyncMock, MagicMock, patch

from app.design_sync.converter import node_to_email_html
from app.design_sync.converter_service import ConversionResult, DesignConverterService
from app.design_sync.import_service import DesignImportService
from app.design_sync.protocol import DesignNode, DesignNodeType


class TestImageNodeRendering:
    """IMAGE node emits correct HTML attributes and styles."""

    def test_image_node_has_responsive_styles(self) -> None:
        node = DesignNode(id="img1", name="Hero", type=DesignNodeType.IMAGE, width=600, height=400)
        html = node_to_email_html(node)
        assert 'src=""' in html
        assert 'data-node-id="img1"' in html
        assert 'width="600"' in html
        assert 'height="400"' in html
        assert "max-width:600px" in html
        assert "display:block" in html
        assert "height:auto" in html

    def test_image_node_without_dimensions(self) -> None:
        node = DesignNode(id="img2", name="Icon", type=DesignNodeType.IMAGE)
        html = node_to_email_html(node)
        assert 'src=""' in html
        assert 'width=' not in html or 'width=""' not in html
        assert "max-width:" not in html

    def test_image_alt_text_escaped(self) -> None:
        node = DesignNode(id="img3", name='Hero "banner" <1>', type=DesignNodeType.IMAGE, width=300, height=200)
        html = node_to_email_html(node)
        assert "&quot;" in html or "&#" in html
        assert "<1>" not in html  # escaped


class TestFillImageUrls:
    """_fill_image_urls() rewrites src attributes correctly."""

    def test_fills_matching_node_ids(self) -> None:
        html = '<img src="" data-node-id="1:2" width="600" /><img src="" data-node-id="3:4" />'
        urls = {"1:2": "/api/v1/design-sync/assets/42/1_2.png", "3:4": "/api/v1/design-sync/assets/42/3_4.png"}
        result = DesignImportService._fill_image_urls(html, urls)
        assert 'src="/api/v1/design-sync/assets/42/1_2.png"' in result
        assert 'src="/api/v1/design-sync/assets/42/3_4.png"' in result
        assert 'src=""' not in result

    def test_leaves_unmatched_nodes_empty(self) -> None:
        html = '<img src="" data-node-id="1:2" /><img src="" data-node-id="5:6" />'
        urls = {"1:2": "/api/v1/design-sync/assets/42/1_2.png"}
        result = DesignImportService._fill_image_urls(html, urls)
        assert 'src="/api/v1/design-sync/assets/42/1_2.png"' in result
        assert 'src=""' in result  # 5:6 not filled

    def test_empty_mapping_returns_unchanged(self) -> None:
        html = '<img src="" data-node-id="1:2" />'
        result = DesignImportService._fill_image_urls(html, {})
        assert result == html

    def test_no_img_tags_returns_unchanged(self) -> None:
        html = "<table><tr><td>No images</td></tr></table>"
        result = DesignImportService._fill_image_urls(html, {"1:2": "/url"})
        assert result == html


class TestConversionResultImages:
    """ConversionResult carries image metadata."""

    def test_default_empty_images(self) -> None:
        result = ConversionResult(html="<table></table>", sections_count=1)
        assert result.images == []

    def test_images_populated(self) -> None:
        result = ConversionResult(
            html="<table></table>",
            sections_count=1,
            images=[{"node_id": "1:2", "filename": "1_2.png", "hub_url": "/assets/1/1_2.png"}],
        )
        assert len(result.images) == 1
        assert result.images[0]["node_id"] == "1:2"
```
