# Plan: 10.12 Intelligence Dashboard Enhancements

## Context
The `/intelligence` page currently shows only QA-centric data: score overview cards, check performance chart, score trend bars, recent results table, and a rendering summary card. It needs to become a comprehensive operations dashboard by adding: graph health status, blueprint success rates, agent performance charts, top failure patterns summary, and component coverage metrics. All data sources already exist via backend APIs and existing hooks.

## Files to Create/Modify

### New Components
- `cms/apps/web/src/components/intelligence/graph-health-card.tsx` — Cognee graph health status card (ping `/health` + `/api/v1/knowledge/graph/search` with a trivial query)
- `cms/apps/web/src/components/intelligence/blueprint-success-card.tsx` — Blueprint run success rates computed from `useBlueprintRuns` data
- `cms/apps/web/src/components/intelligence/agent-performance-chart.tsx` — Horizontal bar chart of per-agent success/failure from blueprint run handoff history
- `cms/apps/web/src/components/intelligence/top-failure-patterns-card.tsx` — Top 5 failure patterns summary with link to `/failure-patterns`
- `cms/apps/web/src/components/intelligence/component-coverage-card.tsx` — Component compatibility coverage breakdown (full/partial/issues/untested)

### New Hooks
- `cms/apps/web/src/hooks/use-intelligence-stats.ts` — Aggregation hook that combines blueprint runs, failure pattern stats, and component data into dashboard-ready metrics

### Modified Files

- `cms/apps/web/messages/en.json` — Add i18n keys under `"intelligence"` namespace

```typescript
"use client";

import useSWR from "swr";
import { fetcher } from "@/lib/swr-fetcher";
import type { FailurePatternStats } from "@/types/failure-patterns";
import type { PaginatedResponseComponentResponse } from "@email-hub/sdk";

/** Aggregated stats for the enhanced intelligence dashboard. */
export interface IntelligenceStats {
  failurePatternStats: FailurePatternStats | null;
  components: {
    total: number;
    full: number;
    partial: number;
    issues: number;
    untested: number;
  };
  blueprintRuns: {
    total: number;
    completed: number;
    failed: number;
    qaPassed: number;
    avgDurationMs: number;
    avgTokens: number;
  };
  graphHealthy: boolean | null; // null = loading/unknown
}

export function useFailurePatternStatsForDashboard() {
  return useSWR<FailurePatternStats>(
    "/api/v1/blueprints/failure-patterns/stats",
    fetcher,
    { revalidateOnFocus: false }
  );
}

export function useComponentCoverage() {
  const { data, ...rest } = useSWR<PaginatedResponseComponentResponse>(
    "/api/v1/components/?page=1&page_size=100",
    fetcher,
    { revalidateOnFocus: false }
  );

  const components = data?.items ?? [];
  const coverage = {
    total: components.length,
    full: components.filter((c) => c.compatibility_badge === "full").length,
    partial: components.filter((c) => c.compatibility_badge === "partial").length,
    issues: components.filter((c) => c.compatibility_badge === "issues").length,
    untested: components.filter(
      (c) => !c.compatibility_badge || c.compatibility_badge === "untested"
    ).length,
  };

  return { coverage, isLoading: rest.isLoading, error: rest.error };
}

export function useGraphHealth() {
  // Simple health check — try to hit the graph search endpoint with minimal query
  // A 200 response means Cognee is reachable; any error means it's down/disabled
  return useSWR<{ healthy: boolean }>(
    "graph-health-check",
    async () => {
      try {
        const { authFetch } = await import("@/lib/auth-fetch");
        const res = await authFetch("/api/v1/knowledge/graph/search", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ query: "health", top_k: 1 }),
        });
        return { healthy: res.ok };
      } catch {
        return { healthy: false };
      }
    },
    { revalidateOnFocus: false, revalidateOnReconnect: false }
  );
}
```

### Step 4: Create `agent-performance-chart.tsx`

Create `cms/apps/web/src/components/intelligence/agent-performance-chart.tsx`:

```tsx
"use client";

import { useMemo } from "react";
import { useTranslations } from "next-intl";
import { Bot } from "lucide-react";
import { Skeleton } from "@email-hub/ui/components/ui/skeleton";
import { useBlueprintRuns } from "@/hooks/use-blueprint-runs";

const AGENT_NAMES = [
  "scaffolder", "dark_mode", "content", "outlook_fixer",
  "accessibility", "personalisation", "code_reviewer",
  "knowledge", "innovation",
];

function scoreColorClass(rate: number): string {
  if (rate >= 0.8) return "bg-status-success";
  if (rate >= 0.5) return "bg-status-warning";
  return "bg-status-danger";
}

export function AgentPerformanceChart() {
  const t = useTranslations("intelligence");
  const { data, isLoading } = useBlueprintRuns(null);

  const agentStats = useMemo(() => {
    const runs = data?.items ?? [];
    // Count per-agent appearances and successes from handoff history in run_data
    const stats: Record<string, { total: number; success: number }> = {};
    for (const name of AGENT_NAMES) {
      stats[name] = { total: 0, success: 0 };
    }

    for (const run of runs) {
      if (!run.run_data?.handoff_history) continue;
      for (const handoff of run.run_data.handoff_history) {
        const agent = handoff.from_agent ?? handoff.to_agent;
        if (agent && stats[agent]) {
          stats[agent]!.total++;
          // A "successful" handoff means the agent's confidence was above threshold
          if (handoff.confidence !== undefined && handoff.confidence !== null && handoff.confidence >= 0.5) {
            stats[agent]!.success++;
          }
        }
      }
    }

    // If no handoff data exists, derive from blueprint_name/status
    const hasHandoffData = Object.values(stats).some((s) => s.total > 0);
    if (!hasHandoffData) {
      for (const run of runs) {
        const name = run.blueprint_name;
        if (stats[name]) {
          stats[name]!.total++;
          if (run.status === "completed") stats[name]!.success++;
        }
      }
    }

    return AGENT_NAMES
      .map((name) => ({
        name,
        total: stats[name]!.total,
        successRate: stats[name]!.total > 0
          ? stats[name]!.success / stats[name]!.total
          : 0,
      }))
      .filter((a) => a.total > 0)
      .sort((a, b) => b.successRate - a.successRate);
  }, [data]);

  if (isLoading) {
    return <Skeleton className="h-64 rounded-lg border border-card-border" />;
  }

  if (agentStats.length === 0) {
    return (
      <div className="rounded-lg border border-card-border bg-card-bg p-6">
        <div className="flex items-center gap-2">
          <Bot className="h-5 w-5 text-foreground-muted" />
          <h2 className="text-lg font-semibold text-foreground">
            {t("agentPerformance")}
          </h2>
        </div>
        <p className="mt-2 text-sm text-foreground-muted">
          {t("agentPerformanceNoData")}
        </p>
      </div>
    );
  }

  return (
    <div className="rounded-lg border border-card-border bg-card-bg p-6">
      <div className="flex items-center gap-2">
        <Bot className="h-5 w-5 text-foreground-muted" />
        <h2 className="text-lg font-semibold text-foreground">
          {t("agentPerformance")}
        </h2>
      </div>
      <p className="mt-1 text-sm text-foreground-muted">
        {t("agentPerformanceDescription")}
      </p>
      <div className="mt-4 space-y-3">
        {agentStats.map((agent) => {
          const pct = Math.round(agent.successRate * 100);
          return (
            <div key={agent.name}>
              <div className="flex items-center justify-between text-sm">
                <span className="text-foreground">
                  {agent.name.replace(/_/g, " ")}
                </span>
                <div className="flex items-center gap-3">
                  <span className="text-xs text-foreground-muted">
                    {agent.total} {t("agentRuns")}
                  </span>
                  <span className="w-10 text-right font-medium text-foreground">
                    {pct}%
                  </span>
                </div>
              </div>
              <div className="mt-1 h-2 w-full rounded-full bg-surface-muted">
                <div
                  className={`h-2 rounded-full transition-all ${scoreColorClass(agent.successRate)}`}
                  style={{ width: `${pct}%` }}
                />
              </div>
            </div>
          );
        })}
      </div>
    </div>
  );
}
```

### Step 6: Create `component-coverage-card.tsx`

Create `cms/apps/web/src/components/intelligence/component-coverage-card.tsx`:

```tsx
"use client";

import Link from "next/link";
import { useTranslations } from "next-intl";
import { Puzzle } from "lucide-react";
import { Skeleton } from "@email-hub/ui/components/ui/skeleton";
import { useComponentCoverage } from "@/hooks/use-intelligence-stats";

const BADGE_COLORS: Record<string, string> = {
  full: "bg-status-success",
  partial: "bg-status-warning",
  issues: "bg-status-danger",
  untested: "bg-surface-muted",
};

export function ComponentCoverageCard() {
  const t = useTranslations("intelligence");
  const { coverage, isLoading } = useComponentCoverage();

  if (isLoading) {
    return <Skeleton className="h-40 rounded-lg border border-card-border" />;
  }

  if (coverage.total === 0) {
    return (
      <div className="rounded-lg border border-card-border bg-card-bg p-6">
        <div className="flex items-center gap-2">
          <Puzzle className="h-5 w-5 text-foreground-muted" />
          <h2 className="text-lg font-semibold text-foreground">
            {t("componentCoverage")}
          </h2>
        </div>
        <p className="mt-2 text-sm text-foreground-muted">
          {t("noComponents")}
        </p>
      </div>
    );
  }

  const segments = [
    { key: "full", count: coverage.full, label: t("coverageFull") },
    { key: "partial", count: coverage.partial, label: t("coveragePartial") },
    { key: "issues", count: coverage.issues, label: t("coverageIssues") },
    { key: "untested", count: coverage.untested, label: t("coverageUntested") },
  ];

  return (
    <div className="rounded-lg border border-card-border bg-card-bg p-6">
      <div className="flex items-center justify-between">
        <div className="flex items-center gap-2">
          <Puzzle className="h-5 w-5 text-foreground-muted" />
          <h2 className="text-lg font-semibold text-foreground">
            {t("componentCoverage")}
          </h2>
        </div>
        <Link
          href="/components"
          className="text-sm text-foreground-accent hover:underline"
        >
          {t("viewComponents")}
        </Link>
      </div>
      <p className="mt-1 text-sm text-foreground-muted">
        {t("componentCoverageDescription", { total: coverage.total })}
      </p>

      {/* Stacked bar */}
      <div className="mt-4 flex h-4 overflow-hidden rounded-full">
        {segments.map((seg) =>
          seg.count > 0 ? (
            <div
              key={seg.key}
              className={`${BADGE_COLORS[seg.key]} transition-all`}
              style={{ width: `${(seg.count / coverage.total) * 100}%` }}
              title={`${seg.label}: ${seg.count}`}
            />
          ) : null
        )}
      </div>

      {/* Legend */}
      <div className="mt-3 flex flex-wrap gap-4">
        {segments.map((seg) => (
          <div key={seg.key} className="flex items-center gap-1.5">
            <div className={`h-2.5 w-2.5 rounded-full ${BADGE_COLORS[seg.key]}`} />
            <span className="text-xs text-foreground-muted">
              {seg.label} ({seg.count})
            </span>
          </div>
        ))}
      </div>
    </div>
  );
}
```
