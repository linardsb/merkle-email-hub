# Plan: 25.4 Tolgee Frontend & Per-Locale Maizzle Builds

## Context
Translation workflows fail when translators can't see email context. This phase adds frontend UI for the existing Tolgee backend (Phase 25.3): translation key viewer/editor, locale build dashboard with per-locale QA, side-by-side locale preview, and in-context translation overlay. The backend already provides 5 endpoints at `/api/v1/connectors/tolgee` (connect, sync-keys, pull, build-locales, get-languages).

## Backend API Reference (already implemented)

| Endpoint | Method | Path | Auth | Response |
|----------|--------|------|------|----------|
| Connect | POST | `/api/v1/connectors/tolgee/connect` | developer | `TolgeeConnectionResponse` |
| Sync Keys | POST | `/api/v1/connectors/tolgee/sync-keys` | developer | `TranslationSyncResponse` |
| Pull Translations | POST | `/api/v1/connectors/tolgee/pull` | developer | `TranslationPullResponse[]` |
| Build Locales | POST | `/api/v1/connectors/tolgee/build-locales` | developer | `LocaleBuildResponse` |
| Get Languages | GET | `/api/v1/connectors/tolgee/connections/{id}/languages` | viewer | `TolgeeLanguage[]` |

Key response shapes:
- `TolgeeConnectionResponse`: `{ id, name, status, credentials_hint, tolgee_project_id, project_id, last_synced_at, created_at }`
- `TranslationSyncResponse`: `{ keys_extracted, push_result: { created, updated, skipped }, template_id }`
- `TranslationPullResponse`: `{ locale, translations_count, translations: Record<string, string> }`
- `LocaleBuildResponse`: `{ template_id, results: [{ locale, html, build_time_ms, gmail_clipping_warning, text_direction }], total_build_time_ms }`
- `TolgeeLanguage`: `{ id, tag, name, original_name, flag_emoji, base }`

### Step 2: SWR Hooks (`cms/apps/web/src/hooks/use-tolgee.ts`)

Follow the project's SWR pattern from `use-projects.ts` and `use-esp-sync.ts`:

```typescript
"use client";

import useSWR from "swr";
import useSWRMutation from "swr/mutation";
import { fetcher } from "@/lib/swr-fetcher";
import { mutationFetcher, longMutationFetcher } from "@/lib/mutation-fetcher";
import type {
  TolgeeConnectionResponse,
  TolgeeLanguage,
  TranslationSyncResponse,
  TranslationPullResponse,
  LocaleBuildResponse,
} from "@/types/tolgee";

const BASE = "/api/v1/connectors/tolgee";

// ── Connection ───────────────────────────────────────────────────────

export function useTolgeeConnection(connectionId: number | null) {
  // Reuse the ESP sync connection list endpoint filtered by type
  // or create a dedicated GET endpoint. For now, use connection list.
  return useSWR<TolgeeConnectionResponse | null>(
    connectionId ? `${BASE}/connections/${connectionId}` : null,
    fetcher
  );
}

export function useCreateTolgeeConnection() {
  return useSWRMutation<
    TolgeeConnectionResponse,
    Error,
    string,
    {
      name: string;
      project_id: number;
      tolgee_project_id: number;
      base_url?: string;
      pat: string;
    }
  >(`${BASE}/connect`, mutationFetcher);
}

// ── Languages ────────────────────────────────────────────────────────

export function useTolgeeLanguages(connectionId: number | null) {
  return useSWR<TolgeeLanguage[]>(
    connectionId ? `${BASE}/connections/${connectionId}/languages` : null,
    fetcher
  );
}

// ── Key Sync ─────────────────────────────────────────────────────────

export function useSyncKeys() {
  return useSWRMutation<
    TranslationSyncResponse,
    Error,
    string,
    { connection_id: number; template_id: number; namespace?: string }
  >(`${BASE}/sync-keys`, mutationFetcher);
}

// ── Pull Translations ────────────────────────────────────────────────

export function usePullTranslations() {
  return useSWRMutation<
    TranslationPullResponse[],
    Error,
    string,
    {
      connection_id: number;
      tolgee_project_id: number;
      locales: string[];
      namespace?: string;
    }
  >(`${BASE}/pull`, mutationFetcher);
}

// ── Locale Build ─────────────────────────────────────────────────────

export function useLocaleBuild() {
  return useSWRMutation<
    LocaleBuildResponse,
    Error,
    string,
    {
      connection_id: number;
      template_id: number;
      tolgee_project_id: number;
      locales: string[];
      namespace?: string;
      is_production?: boolean;
    }
  >(`${BASE}/build-locales`, longMutationFetcher);
}
```

### Step 7: In-Context Overlay (`cms/apps/web/src/components/tolgee/InContextOverlay.tsx`)

Translation overlay that wraps an email preview iframe:

```typescript
"use client";

import { useState, useCallback, useRef, useEffect } from "react";
import type { TranslationKeyRow, TolgeeLanguage } from "@/types/tolgee";

interface Props {
  html: string;
  translationKeys: TranslationKeyRow[];
  selectedLocale: string;
  isEditable: boolean;  // false for viewer role
  onTranslationEdit?: (key: string, locale: string, value: string) => void;
  enabled: boolean;
  onToggle: () => void;
}
```

**Implementation notes:**

1. **Overlay mechanism:**
   - Wrap the preview iframe in a relative container
   - When enabled, overlay a transparent div over the iframe
   - Parse HTML to identify text nodes that match translation key source texts
   - Build a map of `{ key, boundingRect }` for each translatable text region
   - Render highlight rectangles (absolute positioned) over matching regions

2. **Text matching strategy:**
   - On the HTML string (not DOM — iframe is sandboxed), find positions of source text strings
   - Create a visual map by injecting `data-tolgee-key` attributes into a shadow copy of the HTML
   - Render the annotated HTML in a hidden measurement iframe to get bounding rects
   - Overlay highlight boxes at computed positions

3. **Interaction:**
   - Hover → show tooltip with: key name, current translation, status badge
   - Click (if `isEditable`) → show inline edit popover:
     - Text area with current translation
     - "Save" + "Cancel" buttons
     - On save → `onTranslationEdit(key, locale, newValue)`
   - Untranslated strings → yellow background highlight (`bg-status-warning/20`)
   - Translated strings → subtle border on hover (`border-border-accent`)

4. **Toggle button:**
   - Toolbar button with globe icon (lucide `Globe`)
   - Active state: `bg-interactive text-foreground-inverse`
   - Inactive state: `bg-surface-elevated text-foreground`

5. **Simplification for V1:**
   - Instead of complex bounding rect computation, use a simpler approach:
   - Inject the overlay HTML directly: replace each translatable text span with a `<span data-tolgee-key="..." style="outline: 2px dashed orange; cursor: pointer;">` wrapper
   - Render this annotated HTML in the preview iframe (with `sandbox="allow-same-origin"` to enable postMessage)
   - Use `postMessage` from iframe to parent for click/hover events
   - **Security:** The iframe sandbox still prevents script execution. The `allow-same-origin` is needed only for postMessage communication. All translation values are HTML-escaped before injection.
