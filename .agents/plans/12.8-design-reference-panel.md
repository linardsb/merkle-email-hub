# Plan: 12.8 Design Reference Panel in Workspace

## Context
When editing a template that was converted from a Figma design (via 12.5 pipeline), developers need to see the original design alongside the editor. This panel shows the exported frame image, component screenshots, and design tokens (colors, typography, spacing) with rich token interactions. It appears as a right sidebar panel (like QA results) for side-by-side comparison with the preview.

**Key decisions:**
- Right sidebar (not bottom panel tab) — better for visual comparison
- Auto-detect design import from active template's `result_template_id`, fallback to manual connection picker
- Shows frame images + component screenshots from the design import assets
- 10 token interactions (see Token Interactions section below)
- Editor ref must be exposed for programmatic insertion, find, and drag-and-drop

## Token Interactions (all 10)

| # | Interaction | Trigger | What it does |
|---|-------------|---------|-------------|
| 1 | **Insert CSS at cursor** | Click token value | Inserts `color: #E4002B;` or `font-family: 'Proxima Nova';` at editor cursor position |
| 2 | **Find & highlight in code** | Click search icon on token | Highlights all occurrences of that hex/font in the editor (like Ctrl+F but visual) |
| 3 | **Replace color in selection** | Select code + click swatch | Replaces all hex values in the editor selection with the clicked color |
| 4 | **Copy as different formats** | Right-click / long-press token | Context menu: hex, rgb, hsl, CSS custom property. Fonts: CSS shorthand |
| 5 | **Token usage heatmap** | Always visible | Badge on each swatch showing occurrence count in current HTML. "0" = unused indicator |
| 6 | **Drag-and-drop to editor** | Drag swatch/font row | Drops CSS declaration at the drop position in the editor |
| 7 | **Token diff overlay** | Always visible | Warning dot on off-brand colors in code with closest token match. Click → auto-correct |
| 8 | **Generate CSS variables block** | Button at top of tokens | Generates `:root { --brand-primary: #E4002B; ... }` block, inserts into `<style>` section |
| 9 | **Token spotlight** | Hover over swatch | Live highlight of matching hex values in editor (disappears on mouse leave) |
| 10 | **Copy as Maizzle/Tailwind utility** | In copy format menu | Copies as `text-[#E4002B]`, `bg-[#E4002B]`, `font-['Proxima_Nova']` |

```python
async def get_by_template_id(
    self, template_id: int, project_id: int
) -> DesignImport | None:
    """Get the design import that produced a given template."""
    stmt = (
        select(DesignImport)
        .options(selectinload(DesignImport.assets))
        .where(
            DesignImport.result_template_id == template_id,
            DesignImport.project_id == project_id,
            DesignImport.status == "completed",
        )
        .order_by(DesignImport.created_at.desc())
        .limit(1)
    )
    result = await self.db.execute(stmt)
    return result.scalar_one_or_none()
```

### Step 2: Color utility library

**`cms/apps/web/src/lib/color-utils.ts`**

Pure utility functions for color conversion, counting, and distance calculation. No React dependencies.

```typescript
/** Convert hex (#RRGGBB) to {r, g, b} 0-255. */
export function hexToRgb(hex: string): { r: number; g: number; b: number } | null {
  const match = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
  if (!match) return null;
  return {
    r: parseInt(match[1], 16),
    g: parseInt(match[2], 16),
    b: parseInt(match[3], 16),
  };
}

/** Convert hex to rgb() string. */
export function hexToRgbString(hex: string): string {
  const rgb = hexToRgb(hex);
  if (!rgb) return hex;
  return `rgb(${rgb.r}, ${rgb.g}, ${rgb.b})`;
}

/** Convert hex to hsl() string. */
export function hexToHslString(hex: string): string {
  const rgb = hexToRgb(hex);
  if (!rgb) return hex;
  const r = rgb.r / 255, g = rgb.g / 255, b = rgb.b / 255;
  const max = Math.max(r, g, b), min = Math.min(r, g, b);
  const l = (max + min) / 2;
  if (max === min) return `hsl(0, 0%, ${Math.round(l * 100)}%)`;
  const d = max - min;
  const s = l > 0.5 ? d / (2 - max - min) : d / (max + min);
  let h = 0;
  if (max === r) h = ((g - b) / d + (g < b ? 6 : 0)) / 6;
  else if (max === g) h = ((b - r) / d + 2) / 6;
  else h = ((r - g) / d + 4) / 6;
  return `hsl(${Math.round(h * 360)}, ${Math.round(s * 100)}%, ${Math.round(l * 100)}%)`;
}

/** Euclidean RGB distance between two hex colors. */
export function colorDistance(hex1: string, hex2: string): number {
  const c1 = hexToRgb(hex1), c2 = hexToRgb(hex2);
  if (!c1 || !c2) return Infinity;
  return Math.sqrt((c1.r - c2.r) ** 2 + (c1.g - c2.g) ** 2 + (c1.b - c2.b) ** 2);
}

/** Count occurrences of a hex value in HTML string (case-insensitive). */
export function countHexOccurrences(html: string, hex: string): number {
  const normalized = hex.toLowerCase().replace(/^#/, "");
  const pattern = new RegExp(`#${normalized}`, "gi");
  return (html.match(pattern) || []).length;
}

/**
 * Find all hex colors in HTML that don't match any design token.
 * Returns array of {hex, line, closestToken, distance}.
 */
export function findOffBrandColors(
  html: string,
  tokenHexes: string[]
): Array<{ hex: string; closestHex: string; distance: number }> {
  const allHexes = html.match(/#[a-f\d]{6}/gi) || [];
  const unique = [...new Set(allHexes.map((h) => h.toLowerCase()))];
  const tokenSet = new Set(tokenHexes.map((h) => h.toLowerCase()));

  return unique
    .filter((h) => !tokenSet.has(h))
    .map((hex) => {
      let closestHex = tokenHexes[0] ?? hex;
      let minDist = Infinity;
      for (const t of tokenHexes) {
        const d = colorDistance(hex, t);
        if (d < minDist) { minDist = d; closestHex = t; }
      }
      return { hex, closestHex, distance: minDist };
    });
}

/** Generate CSS custom properties block from token colors. */
export function generateCssVariablesBlock(
  colors: Array<{ name: string; hex: string }>,
  typography: Array<{ name: string; family: string }>,
): string {
  const lines: string[] = [":root {"];
  for (const c of colors) {
    const varName = c.name.toLowerCase().replace(/[^a-z0-9]+/g, "-");
    lines.push(`  --${varName}: ${c.hex};`);
  }
  for (const t of typography) {
    const varName = t.name.toLowerCase().replace(/[^a-z0-9]+/g, "-");
    lines.push(`  --font-${varName}: '${t.family}';`);
  }
  lines.push("}");
  return lines.join("\n");
}
```

#### Step 3c: Create `useEditorBridge` hook

**`cms/apps/web/src/hooks/use-editor-bridge.ts`**

Wraps the CodeMirror `EditorView` with high-level operations for token interactions.

```typescript
"use client";

import { useCallback, useRef, type RefObject } from "react";
import type { EditorView } from "@codemirror/view";
import { Decoration, type DecorationSet } from "@codemirror/view";
import { StateEffect, StateField } from "@codemirror/state";
import type { CodeEditorHandle } from "@/components/workspace/editor/code-editor";

/** Effect to set/clear highlights. */
const setHighlights = StateEffect.define<DecorationSet>();

/** StateField to store highlight decorations (added to CodeMirror extensions). */
export const highlightField = StateField.define<DecorationSet>({
  create: () => Decoration.none,
  update(value, tr) {
    for (const e of tr.effects) {
      if (e.is(setHighlights)) return e.value;
    }
    return value.map(tr.changes);
  },
  provide: (f) => EditorView.decorations.from(f),
});

// CSS class for highlights — add to global CSS:
// .cm-token-highlight { background-color: oklch(0.85 0.15 85 / 0.4); outline: 1px solid oklch(0.7 0.15 85); }
// .cm-token-spotlight { background-color: oklch(0.85 0.15 250 / 0.3); }

const highlightMark = Decoration.mark({ class: "cm-token-highlight" });
const spotlightMark = Decoration.mark({ class: "cm-token-spotlight" });

export interface EditorBridge {
  /** #1 — Insert CSS declaration at current cursor position. */
  insertAtCursor: (cssText: string) => void;
  /** #2 — Find and highlight all occurrences of a string in the editor. */
  findAndHighlight: (searchText: string) => void;
  /** Clear all highlights set by findAndHighlight or spotlight. */
  clearHighlights: () => void;
  /** #3 — Replace all occurrences of a hex color in the current selection. */
  replaceInSelection: (oldHex: string, newHex: string) => void;
  /** #6 — Get the character offset from a drop event for drag-and-drop. */
  getDropOffset: (event: DragEvent) => number | null;
  /** #6 — Insert text at a specific offset (for drag-and-drop). */
  insertAtOffset: (offset: number, text: string) => void;
  /** #7 — Replace all occurrences of a hex in the entire document. */
  replaceAll: (oldHex: string, newHex: string) => void;
  /** #8 — Insert a CSS variables block into the <style> section. */
  insertCssVariablesBlock: (block: string) => void;
  /** #9 — Spotlight (hover highlight) all occurrences — non-destructive. */
  spotlight: (searchText: string) => void;
  /** Ref to attach to the EditorPanel. */
  editorRef: RefObject<CodeEditorHandle | null>;
}

export function useEditorBridge(): EditorBridge {
  const editorRef = useRef<CodeEditorHandle | null>(null);

  const getView = useCallback((): EditorView | null => {
    return editorRef.current?.getView() ?? null;
  }, []);

  // Helper: find all matches of a string in the document
  const findMatches = useCallback(
    (view: EditorView, text: string): Array<{ from: number; to: number }> => {
      const doc = view.state.doc.toString();
      const lower = text.toLowerCase();
      const matches: Array<{ from: number; to: number }> = [];
      let idx = 0;
      while (true) {
        idx = doc.toLowerCase().indexOf(lower, idx);
        if (idx === -1) break;
        matches.push({ from: idx, to: idx + text.length });
        idx += text.length;
      }
      return matches;
    },
    []
  );

  // #1 — Insert CSS at cursor
  const insertAtCursor = useCallback(
    (cssText: string) => {
      const view = getView();
      if (!view) return;
      const pos = view.state.selection.main.head;
      view.dispatch({ changes: { from: pos, to: pos, insert: cssText } });
      // Move cursor to end of inserted text
      view.dispatch({
        selection: { anchor: pos + cssText.length },
      });
      view.focus();
    },
    [getView]
  );

  // #2 — Find and highlight
  const findAndHighlight = useCallback(
    (searchText: string) => {
      const view = getView();
      if (!view) return;
      const matches = findMatches(view, searchText);
      if (matches.length === 0) return;
      const decorations = Decoration.set(
        matches.map((m) => highlightMark.range(m.from, m.to))
      );
      view.dispatch({ effects: setHighlights.of(decorations) });
      // Scroll to first match
      view.dispatch({
        selection: { anchor: matches[0].from },
        scrollIntoView: true,
      });
    },
    [getView, findMatches]
  );

  // Clear highlights
  const clearHighlights = useCallback(() => {
    const view = getView();
    if (!view) return;
    view.dispatch({ effects: setHighlights.of(Decoration.none) });
  }, [getView]);

  // #3 — Replace in selection
  const replaceInSelection = useCallback(
    (oldHex: string, newHex: string) => {
      const view = getView();
      if (!view) return;
      const { from, to } = view.state.selection.main;
      if (from === to) return; // No selection
      const selectedText = view.state.sliceDoc(from, to);
      const replaced = selectedText.replace(
        new RegExp(oldHex.replace("#", "#?"), "gi"),
        newHex
      );
      if (replaced !== selectedText) {
        view.dispatch({ changes: { from, to, insert: replaced } });
      }
    },
    [getView]
  );

  // #6 — Get drop offset from DragEvent
  const getDropOffset = useCallback(
    (event: DragEvent): number | null => {
      const view = getView();
      if (!view) return null;
      const pos = view.posAtCoords({ x: event.clientX, y: event.clientY });
      return pos;
    },
    [getView]
  );

  // #6 — Insert at specific offset
  const insertAtOffset = useCallback(
    (offset: number, text: string) => {
      const view = getView();
      if (!view) return;
      view.dispatch({ changes: { from: offset, to: offset, insert: text } });
      view.focus();
    },
    [getView]
  );

  // #7 — Replace all in document
  const replaceAll = useCallback(
    (oldHex: string, newHex: string) => {
      const view = getView();
      if (!view) return;
      const matches = findMatches(view, oldHex);
      if (matches.length === 0) return;
      // Apply replacements in reverse order to preserve offsets
      const changes = [...matches].reverse().map((m) => ({
        from: m.from,
        to: m.to,
        insert: newHex,
      }));
      view.dispatch({ changes });
    },
    [getView, findMatches]
  );

  // #8 — Insert CSS variables block into <style> section
  const insertCssVariablesBlock = useCallback(
    (block: string) => {
      const view = getView();
      if (!view) return;
      const doc = view.state.doc.toString();
      // Try to find existing <style> tag and insert after opening tag
      const styleMatch = doc.match(/<style[^>]*>/i);
      if (styleMatch && styleMatch.index !== undefined) {
        const insertPos = styleMatch.index + styleMatch[0].length;
        view.dispatch({
          changes: { from: insertPos, to: insertPos, insert: `\n${block}\n` },
        });
      } else {
        // No <style> tag — insert at beginning of <head> or start of doc
        const headMatch = doc.match(/<head[^>]*>/i);
        if (headMatch && headMatch.index !== undefined) {
          const insertPos = headMatch.index + headMatch[0].length;
          view.dispatch({
            changes: {
              from: insertPos,
              to: insertPos,
              insert: `\n<style>\n${block}\n</style>\n`,
            },
          });
        } else {
          // Fallback: insert at cursor
          const pos = view.state.selection.main.head;
          view.dispatch({
            changes: {
              from: pos,
              to: pos,
              insert: `<style>\n${block}\n</style>\n`,
            },
          });
        }
      }
      view.focus();
    },
    [getView]
  );

  // #9 — Spotlight (hover highlight, non-destructive)
  const spotlight = useCallback(
    (searchText: string) => {
      const view = getView();
      if (!view) return;
      const matches = findMatches(view, searchText);
      if (matches.length === 0) return;
      const decorations = Decoration.set(
        matches.map((m) => spotlightMark.range(m.from, m.to))
      );
      view.dispatch({ effects: setHighlights.of(decorations) });
    },
    [getView, findMatches]
  );

  return {
    insertAtCursor,
    findAndHighlight,
    clearHighlights,
    replaceInSelection,
    getDropOffset,
    insertAtOffset,
    replaceAll,
    insertCssVariablesBlock,
    spotlight,
    editorRef,
  };
}
```

```typescript
/** Single color swatch with all interactions. */
function ColorSwatch({
  color,
  count,
  editor,
  t,
}: {
  color: DesignColor;
  count: number;
  editor: EditorBridge;
  t: (key: string) => string;
}) {
  // #1 — Click to insert at cursor
  const handleClick = () => {

/** Single font row with interactions. */
function FontRow({

export function TokenDisplay({ tokens, editor, editorContent }: TokenDisplayProps) {
  const t = useTranslations("workspace.designReference");

  // #5 — Usage heatmap: count occurrences of each color in editor content
  const colorCounts = useMemo(() => {
    const counts = new Map<string, number>();
    for (const color of tokens.colors) {
      counts.set(color.hex, countHexOccurrences(editorContent, color.hex));
    }
    return counts;
  }, [tokens.colors, editorContent]);

  // #7 — Token diff: find off-brand colors in editor content
  const offBrandColors = useMemo(() => {
    const tokenHexes = tokens.colors.map((c) => c.hex);
    return findOffBrandColors(editorContent, tokenHexes);
  }, [tokens.colors, editorContent]);

  return (
    <div className="space-y-4">
      {/* #8 — Generate CSS Variables button */}
      {(tokens.colors.length > 0 || tokens.typography.length > 0) && (
        <CssVariablesGenerator tokens={tokens} editor={editor} />
      )}

      {/* Colors */}
      {tokens.colors.length > 0 && (
        <section>
          <h4 className="mb-2 text-xs font-semibold uppercase tracking-wide text-foreground-muted">
            {t("colors")}
          </h4>
          <div className="grid grid-cols-4 gap-3">
            {tokens.colors.map((color) => (
              <ColorSwatch
                key={color.name}
                color={color}
                count={colorCounts.get(color.hex) ?? 0}
                editor={editor}
                t={t}
              />
            ))}
          </div>
        </section>
      )}

      {/* #7 — Off-brand color warnings */}
      {offBrandColors.length > 0 && (
        <section>
          <h4 className="mb-2 flex items-center gap-1 text-xs font-semibold uppercase tracking-wide text-warning">
            <AlertTriangle className="h-3 w-3" />
            {t("offBrandColors")}
          </h4>
          <div className="space-y-1">
            {offBrandColors.map((obc) => (

                  <TooltipTrigger asChild>
                    <button
                      type="button"
                      className="flex w-full items-center gap-2 border border-border px-2 py-1 text-xs hover:bg-surface-elevated transition-colors"
                      onClick={() => {
                        editor.replaceAll(obc.hex, obc.closestHex);
                        toast.success(

                      </span>
                    </button>
                  </TooltipTrigger>
                  <TooltipContent>{t("clickToCorrect")}</TooltipContent>
                </Tooltip>
              </TooltipProvider>
            ))}
          </div>
        </section>
      )}

      {/* Typography */}
      {tokens.typography.length > 0 && (

}
```
