# Plan: 41.2 Adjacent-Section Background Propagation

## Context

Designs frequently use colored backgrounds that flow from an image into a text area (e.g., Mammut blue layering section, orange "Grab a Duvet Day" section). The converter currently ignores this and defaults every content block to `#ffffff`. This phase propagates the facing-edge color of `full-width-image` sections onto adjacent content tables.

## Research Summary

### Assembly Pipeline Flow (`converter_service.py:578-699`)

2. `ComponentRenderer.render_section(match)` → `RenderedSection` per section

**Insertion point:** After rendering loop (line 667), before `sections_html = "\n".join(section_parts)` (line 668). Iterate adjacent pairs and inject bgcolor on content section HTML.

### Step 2: Create `app/design_sync/bgcolor_propagator.py`

```python
"""Adjacent-section background color propagation for visual continuity."""

from __future__ import annotations

import re
from pathlib import Path

from app.core.config import get_settings
from app.core.logging import get_logger
from app.design_sync.assets import _sanitize_node_id
from app.design_sync.image_sampler import sample_edge_color

logger = get_logger(__name__)

# Component slugs that receive propagated bgcolor
_TEXT_LIKE_SLUGS: frozenset[str] = frozenset({
    "text-block",
    "heading-block",
    "hero-text",
    "hero-block",
    "cta-button",
    "article-card",
    "button-block",
    "text-link",
    "product-card",
    "category-nav",
})

# Regex to inject bgcolor on the outermost <table> of a rendered section
_OUTER_TABLE_RE = re.compile(
    r"(<table\b)([^>]*)(>)",
    re.IGNORECASE,
)


def propagate_adjacent_bgcolor(
    matches: list[object],  # list[ComponentMatch] — use object to avoid circular import
    rendered_parts: list[str],
    *,
    connection_id: str | None = None,
    image_dir: Path | None = None,
) -> list[str]:
    """Scan adjacent (image, text) section pairs and propagate edge color.

    For each full-width-image section adjacent to a text-like section,
    samples the facing edge of the image. If a solid color is detected,
    injects ``bgcolor="{hex}"`` on the outermost content tables of the
    text section.

    Args:
        matches: Ordered list of ComponentMatch objects.
        rendered_parts: Parallel list of rendered HTML strings (may include spacers).
        connection_id: Design connection ID for resolving asset paths.
        image_dir: Override directory for image assets (for testing).

    Returns:
        Updated rendered_parts list with bgcolor injected where applicable.
    """
    from app.design_sync.component_matcher import ComponentMatch

    # Build index: section_idx → position in rendered_parts
    # rendered_parts may have spacer HTML interleaved, so we track
    # the actual ComponentMatch objects and their positions
    if not matches or len(matches) < 2:
        return rendered_parts

    # Resolve asset directory
    asset_dir = _resolve_asset_dir(connection_id, image_dir)
    if asset_dir is None:
        return rendered_parts

    result = list(rendered_parts)

    # Walk adjacent pairs in the matches list
    # rendered_parts is built as: [section_0_html, spacer?, section_1_html, spacer?, ...]
    # We need to map match index → rendered_parts index
    part_indices = _build_part_index(matches, result)

    for i in range(len(matches) - 1):
        m_curr = matches[i]
        m_next = matches[i + 1]

        assert isinstance(m_curr, ComponentMatch)
        assert isinstance(m_next, ComponentMatch)

        # Check: image section followed by text-like section
        if m_curr.component_slug == "full-width-image" and m_next.component_slug in _TEXT_LIKE_SLUGS:
            color = _sample_image_bottom(m_curr, asset_dir)
            if color:
                idx = part_indices.get(i + 1)
                if idx is not None:
                    result[idx] = _inject_bgcolor(result[idx], color)
                    logger.info(
                        "design_sync.bgcolor_propagated",
                        source_section=m_curr.section_idx,
                        target_section=m_next.section_idx,
                        color=color,
                        direction="down",
                    )

        # Check: text-like section followed by image section (propagate upward)
        if m_next.component_slug == "full-width-image" and m_curr.component_slug in _TEXT_LIKE_SLUGS:
            color = _sample_image_top(m_next, asset_dir)
            if color:
                idx = part_indices.get(i)
                if idx is not None:
                    result[idx] = _inject_bgcolor(result[idx], color)
                    logger.info(
                        "design_sync.bgcolor_propagated",
                        source_section=m_next.section_idx,
                        target_section=m_curr.section_idx,
                        color=color,
                        direction="up",
                    )

    return result


def _resolve_asset_dir(
    connection_id: str | None,
    image_dir: Path | None,
) -> Path | None:
    """Resolve the directory containing downloaded image assets."""
    if image_dir is not None:
        return image_dir
    if connection_id is None:
        return None
    settings = get_settings()
    path = Path(settings.design_sync.asset_storage_path) / str(connection_id)
    if path.is_dir():
        return path
    return None


def _get_image_path(match: object, asset_dir: Path) -> Path | None:
    """Get local path for the first image in a section."""
    from app.design_sync.component_matcher import ComponentMatch

    assert isinstance(match, ComponentMatch)
    section = match.section
    if not section.images:
        return None
    node_id = section.images[0].node_id
    safe_id = _sanitize_node_id(node_id)
    # Try common formats
    for ext in ("png", "jpg", "jpeg"):
        path = asset_dir / f"{safe_id}.{ext}"
        if path.is_file():
            return path
    return None


def _sample_image_bottom(match: object, asset_dir: Path) -> str | None:
    """Sample the bottom edge of an image section."""
    path = _get_image_path(match, asset_dir)
    if path is None:
        return None
    return sample_edge_color(path, "bottom")


def _sample_image_top(match: object, asset_dir: Path) -> str | None:
    """Sample the top edge of an image section."""
    path = _get_image_path(match, asset_dir)
    if path is None:
        return None
    return sample_edge_color(path, "top")


def _build_part_index(
    matches: list[object],
    parts: list[str],
) -> dict[int, int]:
    """Map match index → rendered_parts index.

    rendered_parts interleaves section HTML with optional MSO spacer HTML.
    Spacers contain '<!--[if mso]>' and no data-slot attributes.
    We count non-spacer entries to align with match indices.
    """
    index: dict[int, int] = {}
    match_idx = 0
    for part_idx, part in enumerate(parts):
        if "<!--[if mso]>" in part and "data-slot" not in part and "<table" not in part[:50]:
            continue  # spacer
        if match_idx < len(matches):
            index[match_idx] = part_idx
            match_idx += 1
    return index


def _inject_bgcolor(html: str, color: str) -> str:
    """Inject bgcolor attribute on all top-level <table> elements in section HTML.

    Only targets tables that don't already have a bgcolor attribute.
    """
    def _replace(m: re.Match[str]) -> str:
        tag_start = m.group(1)  # "<table"
        attrs = m.group(2)       # existing attributes
        tag_end = m.group(3)     # ">"
        if "bgcolor" in attrs.lower():
            return m.group(0)  # already has bgcolor
        return f'{tag_start} bgcolor="{color}"{attrs}{tag_end}'

    return _OUTER_TABLE_RE.sub(_replace, html, count=1)
```

## Spacer Interleaving Detail

The rendering loop in `converter_service.py:624-666` appends to `section_parts`:
- Line 634: `section_parts.append(rendered.html)` — section HTML
- Line 655-666: `section_parts.append(...)` — MSO spacer (only if `spacing_after > 0`)

So `section_parts` is: `[section_0, spacer_0?, section_1, spacer_1?, ...]`. The `_build_part_index()` function skips spacer entries to align match indices with section HTML positions.

## Edge Cases

| Case | Behavior |
|------|----------|
| Image with gradient/photo edge | `sample_edge_color` returns `None` → no propagation |
| Image section has no images list | `_get_image_path` returns `None` → skip |
| Asset file not downloaded yet | `Path.is_file()` returns `False` → skip |
| Section already has `bgcolor` attr | `_inject_bgcolor` preserves existing value |
| Two consecutive image sections | Neither is text-like → no propagation |
| `connection_id` is `None` (snapshot tests) | Use `image_dir` override or skip |

## Preflight Warnings

- `_build_part_index` heuristic for spacer detection relies on `<!--[if mso]>` — if spacer HTML format changes, this breaks. Add a test that exercises spacer interleaving.
- `_sanitize_node_id` is imported from `assets.py` (private function). If it gets renamed, the import breaks. Consider inlining or making it non-private.
