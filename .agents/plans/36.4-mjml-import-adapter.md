# Plan: 36.4 MJML Import Adapter

## Context

Create `app/design_sync/mjml_import/adapter.py` — a parser that reads MJML markup and produces an `EmailDesignDocument`. This completes the MJML round-trip (import ↔ export) alongside Phase 35.3 (MJML generation). Enterprise email teams have hundreds of MJML templates; importing them unlocks AI agent enhancement, visual editing, multi-client rendering, and QA checks.

Rules (applied in order, only to sections with `type="unknown"`):
1. Explicit types from MJML tags already set (hero, social, nav, spacer, divider) — skip
2. First section: if has image(s) and no text → `"header"`
3. Last section: if has small text (font_size < 14) or has `content_roles` with legal/unsubscribe → `"footer"`
4. Section with only button(s) and optional text → `"cta"`
5. Section with large background image + heading text → `"hero"`
6. Everything else → `"content"`

### Step 5: Main Adapter (`mjml_import/adapter.py`)

```python
from __future__ import annotations
from lxml import etree
from app.design_sync.email_design_document import (
    EmailDesignDocument, DocumentSource, DocumentLayout,
)
from app.design_sync.mjml_import.token_extractor import extract_tokens
from app.design_sync.mjml_import.section_parser import parse_sections
from app.design_sync.mjml_import.type_inferrer import infer_section_types
from app.design_sync.exceptions import MjmlImportError

_MAX_MJML_BYTES = 2 * 1024 * 1024  # 2 MB

class MjmlImportAdapter:
    """Parse MJML markup into EmailDesignDocument."""

    def parse(self, mjml_source: str) -> EmailDesignDocument:
        if len(mjml_source.encode("utf-8")) > _MAX_MJML_BYTES:
            raise MjmlImportError("MJML source exceeds 2 MB limit")

        root = self._parse_xml(mjml_source)
        self._validate_root(root)

        head = root.find("mj-head")
        body = root.find("mj-body")
        if body is None:
            raise MjmlImportError("Missing <mj-body> element")

        tokens = extract_tokens(head)
        sections = parse_sections(body)
        sections = infer_section_types(sections)

        container_width = self._parse_container_width(body)

        return EmailDesignDocument(
            version="1.0",
            source=DocumentSource(provider="mjml"),
            tokens=tokens,
            sections=sections,
            layout=DocumentLayout(
                container_width=container_width,
                naming_convention="mjml",
            ),
        )

    def _parse_xml(self, source: str) -> etree._Element:
        """Parse MJML as XML with XXE prevention."""
        parser = etree.XMLParser(
            resolve_entities=False,
            no_network=True,
            huge_tree=False,
            recover=False,
        )
        try:
            tree = etree.fromstring(source.encode("utf-8"), parser=parser)
        except etree.XMLSyntaxError as exc:
            raise MjmlImportError(f"Invalid MJML XML: {exc}") from exc
        return tree

    def _validate_root(self, root: etree._Element) -> None:
        tag = etree.QName(root.tag).localname if isinstance(root.tag, str) else str(root.tag)
        if tag != "mjml":
            raise MjmlImportError(f"Root element must be <mjml>, got <{tag}>")

    def _parse_container_width(self, body: etree._Element) -> int:
        width_attr = body.get("width", "600px")
        try:
            return int(width_attr.replace("px", "").strip())
        except ValueError:
            return 600
```

### Step 6: API Endpoint

**`app/design_sync/routes.py`** — add:

```python
class ImportMjmlRequest(BaseModel):
    mjml_source: str = Field(..., max_length=2_100_000)  # ~2MB UTF-8

class ImportMjmlResponse(BaseModel):
    document: dict[str, Any]
    sections_count: int
    warnings: list[str]

@router.post("/import/mjml", response_model=ImportMjmlResponse)
@require_role("developer")
@limiter.limit("10/minute")
async def import_mjml(
    request: Request,
    body: ImportMjmlRequest,
    user: User = Depends(get_current_user),
) -> ImportMjmlResponse:
    settings = get_settings()
    if not settings.design_sync.mjml_import_enabled:
        raise HTTPException(status_code=503, detail="MJML import is disabled")

    adapter = MjmlImportAdapter()
    try:
        document = adapter.parse(body.mjml_source)
    except MjmlImportError as exc:
        raise HTTPException(status_code=422, detail=str(exc)) from exc

    doc_json = document.to_json()
    errors = EmailDesignDocument.validate(doc_json)
    warnings = [str(e) for e in errors] if errors else []

    return ImportMjmlResponse(
        document=doc_json,
        sections_count=len(document.sections),
        warnings=warnings,
    )
```

## Preflight Warnings

- `lxml.etree.fromstring` returns `_Element` which pyright may flag — use `# type: ignore[arg-type]` only if needed, prefer `cast()`.
- `etree._Element.find()` returns `_Element | None` — always check for `None` before accessing.
- Existing pyright baseline is 232 errors — new code must not increase this.
- MJML has no XML namespace by default (no `xmlns`), so use `find("mj-head")` not `find("{ns}mj-head")`.
- `<mj-text>` inner content is HTML (mixed content in XML) — use `etree.tostring(el, method="html")` or iterate children to extract inner HTML.
