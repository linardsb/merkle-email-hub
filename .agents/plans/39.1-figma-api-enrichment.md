    """Parse Figma characterStyleOverrides + styleOverrideTable → StyleRun tuple."""
    overrides = node_data.get("characterStyleOverrides", [])

    if not overrides or not table:

    characters = node_data.get("characters", "")

    # ... group contiguous same-override indices, map table entries to StyleRun fields
    return tuple(runs)

New helper:
```python
_HYPERLINK_SCHEMES = frozenset({"http", "https", "mailto"})

def _validate_hyperlink(raw: Any) -> str | None:
    """Extract and validate hyperlink URL. Reject javascript: etc."""
    if isinstance(raw, dict):
        url = raw.get("url", "")
    elif isinstance(raw, str):
        url = raw
    else:
        return None
    if not url:
        return None
    from urllib.parse import urlparse
    parsed = urlparse(url)
    if parsed.scheme.lower() not in _HYPERLINK_SCHEMES:
        return None
    return url
```

## Security Checklist

- **Hyperlink URL validation:** Allowlist `http`, `https`, `mailto` schemes only. Reject `javascript:`, `data:`, `vbscript:`, blank schemes. Use `urllib.parse.urlparse()`.
- **Rich text content:** All text from `characters` field passed through `html.escape()` before wrapping in HTML tags. `StyleRun.link_url` validated with same scheme allowlist.
- **No new endpoints** — this is internal pipeline enrichment, no API surface change.
- **No user input injection** — all data comes from Figma API responses (trusted external API, not user-controlled).
- **XSS via Figma content:** Text content could contain `<script>` if a designer typed it — `html.escape()` handles this.
