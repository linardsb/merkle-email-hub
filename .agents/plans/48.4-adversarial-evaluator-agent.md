```python
from pydantic import BaseModel, Field
from typing import Literal

class EvalIssue(BaseModel):
    severity: Literal["critical", "major", "minor"]
    category: str
    description: str
    location: str | None = None

class EvalVerdict(BaseModel):
    verdict: Literal["accept", "revise", "reject"]
    score: float = Field(ge=0.0, le=1.0)
    issues: list[EvalIssue] = Field(default_factory=list)
    feedback: str = ""
    suggested_corrections: list[str] = Field(default_factory=list)

class EvaluatorRequest(BaseModel):
    original_brief: str
    agent_name: str
    agent_output: str
    quality_criteria: list[str] = Field(default_factory=list)
    iteration: int = 0
    previous_feedback: str | None = None

class EvaluatorResponse(BaseModel):
    verdict: EvalVerdict
    model: str
    confidence: float | None = None
    skills_loaded: list[str] = Field(default_factory=list)
```

### Step 6: Blueprint Node (`app/ai/blueprints/nodes/evaluator_node.py`)

```python
class EvaluatorNode:
    @property
    def name(self) -> str:
        return "evaluator"

    @property
    def node_type(self) -> NodeType:
        return "agentic"

    async def execute(self, context: NodeContext) -> NodeResult:
        if not settings.ai.evaluator.enabled:
            return NodeResult(status="success", html=context.html, details="Evaluator disabled")

        service = get_evaluator_service()
        # Extract brief + upstream agent name from context.metadata
        brief = context.metadata.get("brief", "")
        upstream_agent = context.metadata.get("upstream_agent", "unknown")
        criteria = _load_criteria_names(upstream_agent)

        response = await service.evaluate(
            original_brief=brief,
            agent_name=upstream_agent,
            agent_output=context.html,
            quality_criteria=criteria,
        )

        verdict = response.verdict
        if verdict.verdict == "accept":
            return NodeResult(status="success", html=context.html, details="Evaluator accepted")
        elif verdict.verdict == "reject":
            return NodeResult(
                status="failed", html=context.html,
                details=f"Evaluator rejected: {verdict.feedback}",
                error=verdict.feedback,
            )
        else:  # "revise"
            # Store feedback in context for upstream agent re-run
            return NodeResult(
                status="failed", html=context.html,
                details=f"Evaluator requests revision: {verdict.feedback}",
                handoff=AgentHandoff(
                    agent_name="evaluator",
                    artifact=context.html,
                    status="WARNING",
                    decisions=(verdict.feedback,),
                    warnings=tuple(i.description for i in verdict.issues),
                    component_refs=(),
                    confidence=verdict.score,
                ),
            )
```
