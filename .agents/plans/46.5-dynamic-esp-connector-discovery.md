# Plan: 46.5 Dynamic ESP Connector Discovery via Plugin System

## Context

Adding a new ESP currently requires code changes in two hardcoded registries (`SUPPORTED_CONNECTORS` in `service.py`, `PROVIDER_REGISTRY` in `sync_service.py`). The plugin system (`app/plugins/`) already handles manifest-based directory scanning for QA checks. Extending it to connectors makes ESP integrations pluggable.

## Implementation Steps

### Step 1: `PluginConnectorAPI` in `app/plugins/api.py`

Add connector registration sub-API, following `PluginQAAPI` pattern:

```python
from app.connectors.protocol import ConnectorProvider
from app.connectors.sync_protocol import ESPSyncProvider

class PluginConnectorAPI:
    """Connector registration API for plugins."""

    def __init__(self, manifest: PluginManifest) -> None:
        self._manifest = manifest
        self._providers: dict[str, type[ConnectorProvider]] = {}
        self._sync_providers: dict[str, type[ESPSyncProvider]] = {}

    def register_provider(self, name: str, provider_cls: type[ConnectorProvider]) -> None:
        """Register an export connector provider."""
        if not isinstance(provider_cls, type) or not issubclass_protocol(provider_cls, ConnectorProvider):
            msg = f"Provider '{name}' does not implement ConnectorProvider protocol"
            raise TypeError(msg)
        self._providers[name] = provider_cls
        logger.info("plugins.connector_registered", plugin=self._manifest.name, connector=name)

    def register_sync_provider(self, name: str, provider_cls: type[ESPSyncProvider]) -> None:
        """Register a sync connector provider."""
        self._sync_providers[name] = provider_cls
        logger.info("plugins.sync_provider_registered", plugin=self._manifest.name, connector=name)

    @property
    def registered_providers(self) -> dict[str, type[ConnectorProvider]]:
        return dict(self._providers)

    @property
    def registered_sync_providers(self) -> dict[str, type[ESPSyncProvider]]:
        return dict(self._sync_providers)
```

Update `HubPluginAPI.__init__()` to add `self.connectors = PluginConnectorAPI(manifest)`.

### Step 2: Connector Plugin Bridge in `app/connectors/plugin_bridge.py`

Runtime registration functions that add plugin-provided classes to the module-level dicts:

```python
from app.connectors.protocol import ConnectorProvider
from app.connectors.sync_protocol import ESPSyncProvider
from app.connectors.service import SUPPORTED_CONNECTORS
from app.connectors.sync_service import PROVIDER_REGISTRY
from app.core.logging import get_logger

logger = get_logger(__name__)

def register_plugin_connector(name: str, provider_cls: type[ConnectorProvider]) -> None:
    """Register a plugin-provided export connector at runtime."""
    if name in SUPPORTED_CONNECTORS:
        logger.warning("connectors.plugin_duplicate_export", name=name)
        return  # Built-in connectors take priority
    SUPPORTED_CONNECTORS[name] = provider_cls
    logger.info("connectors.plugin_export_registered", name=name)

def register_plugin_sync_provider(name: str, provider_cls: type[ESPSyncProvider]) -> None:
    """Register a plugin-provided sync provider at runtime."""
    if name in PROVIDER_REGISTRY:
        logger.warning("connectors.plugin_duplicate_sync", name=name)
        return
    PROVIDER_REGISTRY[name] = provider_cls
    logger.info("connectors.plugin_sync_registered", name=name)

def unregister_plugin_connector(name: str) -> None:
    """Remove a plugin connector from both registries."""
    SUPPORTED_CONNECTORS.pop(name, None)
    PROVIDER_REGISTRY.pop(name, None)
```

### Step 3: Wire in `PluginRegistry` (`app/plugins/registry.py`)

Add `_connector_names: dict[str, str]` (connector_name → plugin_name) to `__init__`.

Add `_wire_connector()`:

```python
def _wire_connector(self, manifest: PluginManifest, api: HubPluginAPI) -> None:
    """Register plugin connectors, detecting conflicts."""
    from app.connectors.plugin_bridge import (
        register_plugin_connector,
        register_plugin_sync_provider,
    )

    for name, cls in api.connectors.registered_providers.items():
        if name in self._connector_names:
            existing = self._connector_names[name]
            raise PluginConflictError(
                f"Connector '{name}' already registered by plugin '{existing}'"
            )
        register_plugin_connector(name, cls)
        self._connector_names[name] = manifest.name

    for name, cls in api.connectors.registered_sync_providers.items():
        register_plugin_sync_provider(name, cls)
```

Update `_wire_plugin()`:

```python
def _wire_plugin(self, manifest: PluginManifest, api: HubPluginAPI) -> None:
    if manifest.plugin_type == PluginType.qa_check:
        self._wire_qa_checks(manifest, api)
    elif manifest.plugin_type == PluginType.export_connector:
        self._wire_connector(manifest, api)
```

Update `unregister()` and `disable()` to clean up `_connector_names` and call `unregister_plugin_connector()`.
