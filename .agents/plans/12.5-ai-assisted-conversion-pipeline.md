# Plan: 12.5 AI-Assisted Conversion Pipeline

## Context

Wire the Figma design import flow into the Scaffolder agent to produce Template + TemplateVersion records. The user provides a brief (from ESP/CMS) and a design connection reference. The pipeline runs in the background: fetches Figma structure → exports images → analyzes layout → calls Scaffolder with design context → creates Template + TemplateVersion → updates import status. Two-phase UX: create import with brief → optionally edit brief → trigger conversion.

## Decisions (from discussion)

1. **Two-phase flow**: `POST /imports` creates the record (status `pending`), user can `PATCH` the brief, then `POST /imports/{id}/convert` kicks off the background pipeline.
2. **New `design_context` field** on `ScaffolderRequest` — carries Figma-specific context (image URLs, layout summary). Separate from `brand_config` (brand guidelines).
3. **Only 4 new endpoints** — reuse existing file-structure/components/export-images endpoints. The import service calls them internally via service layer.
4. **Template name derived from brief** — first meaningful line or a truncated summary.
5. **Background pipeline** via `asyncio.create_task` — same pattern as 12.6 component extraction.

### Step 5: Create `DesignImportService` orchestrator

New file: `app/design_sync/import_service.py`

```python
class DesignImportService:
    """Orchestrates the Figma → Scaffolder → Template conversion pipeline.

    Runs as a background task. Updates DesignImport status as it progresses:
    pending → converting → completed/failed
    """

    def __init__(
        self,
        db: AsyncSession,
        design_service: DesignSyncService,
        design_repo: DesignSyncRepository,
    ) -> None:
        self._db = db
        self._design_service = design_service
        self._repo = design_repo

    async def run_conversion(
        self,
        import_id: int,
        user: User,
        *,
        run_qa: bool = True,
        output_mode: str = "structured",
    ) -> None:
        """Background pipeline: layout analysis → asset download → Scaffolder → Template creation."""
        design_import = await self._repo.get_import_with_assets(import_id)
        if design_import is None:
            logger.error("design_sync.import_not_found", import_id=import_id)
            return

        try:
            await self._repo.update_import_status(design_import, "converting")

            # 1. Get connection details
            conn = await self._repo.get_connection(design_import.connection_id)
            # (error handling if conn is None)

            # 2. Analyze layout (reuse existing service method internally)
            layout_response = await self._design_service.analyze_layout(
                design_import.connection_id,
                user,
                selected_node_ids=design_import.selected_node_ids or None,
            )

            # 3. Download assets for all image nodes
            image_node_ids = self._collect_image_node_ids(layout_response)
            asset_response = None
            if image_node_ids:
                asset_response = await self._design_service.download_assets(
                    design_import.connection_id,
                    user,
                    image_node_ids,
                )

            # 4. Get design tokens (if available)
            tokens = None
            try:
                tokens = await self._design_service.get_tokens(
                    design_import.connection_id, user
                )
            except Exception:
                logger.warning("design_sync.tokens_skipped", import_id=import_id, exc_info=True)

            # 5. Build DesignContext
            design_context = self._build_design_context(
                layout_response, asset_response, tokens, conn
            )

            # 6. Call Scaffolder
            scaffolder_response = await self._call_scaffolder(
                brief=design_import.generated_brief,  # brief stored on import record
                design_context=design_context,
                run_qa=run_qa,
                output_mode=output_mode,
            )

            # 7. Create Template + TemplateVersion
            template_id = await self._create_template(
                project_id=design_import.project_id,
                brief=design_import.generated_brief,
                html=scaffolder_response.html,
                user_id=design_import.created_by_id,
                template_name=design_import.structure_json.get("template_name") if design_import.structure_json else None,
            )

            # 8. Mark complete
            await self._repo.update_import_status(
                design_import,
                "completed",
                result_template_id=template_id,
                structure_json={
                    **(design_import.structure_json or {}),
                    "scaffolder_model": scaffolder_response.model,
                    "qa_passed": scaffolder_response.qa_passed,
                    "confidence": scaffolder_response.confidence,
                },
            )

            logger.info(
                "design_sync.conversion_completed",
                import_id=import_id,
                template_id=template_id,
            )

        except Exception:
            logger.exception("design_sync.conversion_failed", import_id=import_id)
            design_import = await self._repo.get_import(import_id)
            if design_import is not None:
                await self._repo.update_import_status(
                    design_import,
                    "failed",
                    error_message="Conversion pipeline failed",
                )
```

Key helper methods:

- `_collect_image_node_ids(layout: LayoutAnalysisResponse) -> list[str]`: Extract node IDs from `layout.sections[].images[].node_id`
- `_build_design_context(layout, assets, tokens, conn) -> dict[str, object]`: Build the context dict with image URL mappings (`/api/v1/design-sync/assets/{conn_id}/{filename}`), layout summary, token data
- `_call_scaffolder(brief, design_context, run_qa, output_mode) -> ScaffolderResponse`: Create `ScaffolderRequest` with brief + design_context, call `ScaffolderService().generate()`
- `_create_template(project_id, brief, html, user_id, template_name) -> int`: Use `TemplateRepository` to create Template + v1 TemplateVersion, derive name from brief if not provided (first line up to 200 chars, or "Imported from Figma" fallback)

```python
async def create_design_import(
    self,
    data: StartImportRequest,
    user: User,
) -> ImportResponse:
    """Create a new import record with brief. Status = pending."""
    conn = await self._repo.get_connection(data.connection_id)
    if conn is None:
        raise ConnectionNotFoundError(f"Connection {data.connection_id} not found")
    if conn.project_id is None:
        raise DomainValidationError("Connection must be linked to a project")
    if conn.project_id is not None:
        await self._verify_access(conn.project_id, user)

    design_import = await self._repo.create_import(
        connection_id=data.connection_id,
        project_id=conn.project_id,
        selected_node_ids=data.selected_node_ids,
        created_by_id=user.id,
    )
    # Store brief and optional template name on the import record
    await self._repo.update_import_status(
        design_import,
        "pending",
        generated_brief=data.brief,
        structure_json={"template_name": data.template_name} if data.template_name else None,
    )

    return self._import_to_response(design_import)


async def get_design_import(
    self, import_id: int, user: User
) -> ImportResponse:
    """Get import status with BOLA check."""
    design_import = await self._repo.get_import_with_assets(import_id)
    if design_import is None:
        raise ImportNotFoundError(f"Import {import_id} not found")
    await self._verify_access(design_import.project_id, user)
    return self._import_to_response(design_import)


async def update_import_brief(
    self, import_id: int, brief: str, user: User
) -> ImportResponse:
    """Update the brief on a pending import."""
    design_import = await self._repo.get_import(import_id)
    if design_import is None:
        raise ImportNotFoundError(f"Import {import_id} not found")
    await self._verify_access(design_import.project_id, user)
    if design_import.status != "pending":
        raise ImportStateError(
            f"Cannot edit brief: import is '{design_import.status}', expected 'pending'"
        )
    await self._repo.update_import_status(
        design_import, "pending", generated_brief=brief
    )
    return self._import_to_response(design_import)


async def start_conversion(
    self,
    import_id: int,
    user: User,
    *,
    run_qa: bool = True,
    output_mode: str = "structured",
) -> ImportResponse:
    """Kick off the background conversion pipeline."""
    design_import = await self._repo.get_import(import_id)
    if design_import is None:
        raise ImportNotFoundError(f"Import {import_id} not found")
    await self._verify_access(design_import.project_id, user)
    if design_import.status not in ("pending", "failed"):
        raise ImportStateError(
            f"Cannot convert: import is '{design_import.status}', expected 'pending' or 'failed'"
        )
    if not design_import.generated_brief:
        raise DomainValidationError("Import has no brief — set one before converting")

    # Launch background pipeline
    from app.design_sync.import_service import DesignImportService

    import_service = DesignImportService(
        db=self.db,
        design_service=self,
        design_repo=self._repo,
    )
    _task = asyncio.create_task(  # noqa: RUF006
        import_service.run_conversion(
            import_id=import_id,
            user=user,
            run_qa=run_qa,
            output_mode=output_mode,
        )
    )

    # Immediately return with "converting" status
    await self._repo.update_import_status(design_import, "converting")
    return self._import_to_response(design_import)


def _import_to_response(self, design_import: DesignImport) -> ImportResponse:
    """Convert DesignImport model to response schema."""
    return ImportResponse.model_validate(design_import, from_attributes=True)
```

```python
def build_design_context_section(design_context: dict[str, object]) -> str:
    """Build a prompt section from Figma design context."""
    parts: list[str] = ["## Design Reference (Figma Import)\n"]

    layout = design_context.get("layout_summary")
    if layout:
        parts.append(f"**Detected layout:** {layout}\n")

    image_urls = design_context.get("image_urls", {})
    if image_urls:
        parts.append("**Available images (use as img src):**")
        for node_id, url in image_urls.items():
            parts.append(f"- `{node_id}`: `{url}`")
        parts.append("")

    tokens = design_context.get("design_tokens")
    if tokens and isinstance(tokens, dict):
        colors = tokens.get("colors", [])
        if colors:
            color_list = ", ".join(
                f"{c.get('name', '?')}: {c.get('hex', '?')}" for c in colors if isinstance(c, dict)
            )
            parts.append(f"**Brand colors:** {color_list}")
        typography = tokens.get("typography", [])
        if typography:
            font_list = ", ".join(
                f"{t.get('name', '?')}: {t.get('family', '?')} {t.get('size', '?')}px"
                for t in typography if isinstance(t, dict)
            )
            parts.append(f"**Typography:** {font_list}")

    source = design_context.get("source_file")
    if source:
        parts.append(f"\n_Source: {source}_")

    return "\n".join(parts)
```

```python
def _build_user_message(self, request: Any) -> str:
    req: ScaffolderRequest = request
    message = req.brief
    if req.design_context:
        from app.ai.agents.scaffolder.prompt import build_design_context_section
        message += "\n\n" + build_design_context_section(req.design_context)
    return message
```
