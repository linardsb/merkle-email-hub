## Implementation Steps

### Step 1: Add bgcolor extraction helper

Add to `test_snapshot_regression.py` after `_html_diff()`:

```python
_SECTION_TABLE_RE = re.compile(
    r'<table[^>]*class="resptab"[^>]*bgcolor="([^"]*)"[^>]*>',
    re.IGNORECASE,
)

def _extract_section_bgcolors(html: str) -> list[str | None]:
    """Extract bgcolor from each resptab section table in order.

    Returns list of hex colors (uppercase) or None for sections without bgcolor.
    Skips MSO conditional tables and spacer tables.
    """
    bgcolors: list[str | None] = []
    for line in html.split("\n"):
        # Skip MSO conditional blocks
        if "<!--[if" in line or "mso-hide" in line:
            continue
        match = _SECTION_TABLE_RE.search(line)
        if match:
            bgcolors.append(match.group(1).upper())
    return bgcolors
```

The `resptab` class is on all section-level `<table>` elements (confirmed by all 3 reference HTMLs). This distinguishes section tables from inner structural tables.

### Step 3: Add TestBackgroundContinuity class

Add after `TestSnapshotSectionCount`:

```python
@pytest.mark.snapshot
class TestBackgroundContinuity:
    """Verify bgcolor propagation on converter output.

    After Phase 41.2, content sections adjacent to full-width images
    with solid-color edges must inherit the image's edge bgcolor.
    The reference HTMLs document expected values per section.
    """

    @pytest.mark.parametrize("case_id", _get_active_case_ids())
    def test_background_continuity(self, case_id: str) -> None:
        expectations = _BGCOLOR_EXPECTATIONS.get(case_id, {})
        if not expectations:
            pytest.skip(f"Case {case_id} has no bgcolor continuity expectations")

        case_dir = _DEBUG_DIR / case_id
        result = _run_conversion(case_dir)
        assert result.html, f"Converter produced empty HTML for case {case_id}"

        bgcolors = _extract_section_bgcolors(result.html)

        mismatches: list[str] = []
        for section_idx, expected_color in expectations.items():
            if section_idx >= len(bgcolors):
                mismatches.append(
                    f"  Section {section_idx}: expected {expected_color}, "
                    f"but only {len(bgcolors)} sections in output"
                )
                continue

            actual = bgcolors[section_idx]
            if actual is None:
                mismatches.append(
                    f"  Section {section_idx}: expected {expected_color}, got no bgcolor"
                )
            elif actual.upper() != expected_color.upper():
                mismatches.append(
                    f"  Section {section_idx}: expected {expected_color}, got {actual}"
                )

        if mismatches:
            # Save actual bgcolors for debugging
            debug_info = "\n".join(
                f"  [{i}] {c}" for i, c in enumerate(bgcolors)
            )
            pytest.fail(
                f"Background continuity mismatch for case {case_id}:\n"
                + "\n".join(mismatches)
                + f"\n\nActual bgcolors:\n{debug_info}"
            )
```

### Step 4: Add reference HTML cross-check test

This validates that the reference HTMLs themselves have the expected bgcolors (catches stale expectations):

```python
_REFERENCE_MAP: dict[str, str] = {
    "10": "mammut-duvet-day.html",
    "6": "starbucks-pumpkin-spice.html",
    "5": "maap-kask.html",
}
_REFERENCE_DIR = Path(__file__).resolve().parents[3] / "email-templates" / "training_HTML" / "for_converter_engine"

class TestReferenceBgcolorSanity:
    """Validate that reference HTMLs match our expectation maps.

    If a reference HTML is updated, this catches stale _BGCOLOR_EXPECTATIONS.
    """

    @pytest.mark.parametrize(
        "case_id",
        [k for k in _BGCOLOR_EXPECTATIONS if _BGCOLOR_EXPECTATIONS[k]],
    )
    def test_reference_bgcolors(self, case_id: str) -> None:
        ref_file = _REFERENCE_DIR / _REFERENCE_MAP[case_id]
        if not ref_file.exists():
            pytest.skip(f"Reference HTML not found: {ref_file}")

        html = ref_file.read_text()
        bgcolors = _extract_section_bgcolors(html)
        expectations = _BGCOLOR_EXPECTATIONS[case_id]

        for section_idx, expected_color in expectations.items():
            assert section_idx < len(bgcolors), (
                f"Reference {_REFERENCE_MAP[case_id]} has {len(bgcolors)} sections, "
                f"but expectation references index {section_idx}"
            )
            actual = bgcolors[section_idx]
            assert actual is not None and actual.upper() == expected_color.upper(), (
                f"Reference {_REFERENCE_MAP[case_id]} section {section_idx}: "
                f"expected {expected_color}, got {actual}"
            )
```

```python
_DARK_BGCOLOR_CASES: dict[str, list[int]] = {
    "10": [1, 2, 3, 4, 6, 7, 8],  # Orange + blue sections
}

@pytest.mark.parametrize("case_id", [k for k in _DARK_BGCOLOR_CASES])
def test_text_inversion_on_dark_sections(self, case_id: str) -> None:
    """Dark bgcolor sections must have white text (color:#ffffff)."""
    case_dir = _DEBUG_DIR / case_id
    result = _run_conversion(case_dir)

    # Split into section-level chunks by resptab tables
    sections = re.split(r'(?=<table[^>]*class="resptab")', result.html)
    dark_indices = _DARK_BGCOLOR_CASES[case_id]

    for idx in dark_indices:
        # +1 because split produces a preamble chunk
        if idx + 1 >= len(sections):
            continue
        section_html = sections[idx + 1]
        # Check for dark text colors that should have been inverted
        dark_colors = re.findall(
            r'(?<!background-)color\s*:\s*(#(?:000|111|222|333|1[aA]1[aA]1[aA]|0{6})\b)',
            section_html,
        )
        assert not dark_colors, (
            f"Case {case_id} section {idx}: dark text colors {dark_colors} "
            f"found on dark bgcolor section — should be inverted to #ffffff"
        )
```
