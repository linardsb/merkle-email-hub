# Plan: 23.2 Adapter Multimodal Serialization

## Context

Both LLM adapters (`OpenAICompatProvider`, `AnthropicProvider`) currently serialize messages as simple `{"role": ..., "content": str}` dicts. Phase 23.1 introduces `ContentBlock` union types (`TextBlock`, `ImageBlock`, `AudioBlock`, `StructuredOutputBlock`) and extends `Message.content` to accept `str | list[ContentBlock]`. This task makes adapters serialize those blocks into provider-specific API formats — keeping agents provider-agnostic.

**Dependency:** Phase 23.1 must be complete (ContentBlock types defined in `app/ai/protocols.py`).

### Step 2: OpenAI Adapter — `_serialize_content()` (`app/ai/adapters/openai_compat.py`)

Add a static method to convert `ContentBlock` list to OpenAI multipart content format:

```python
@staticmethod
def _serialize_content(blocks: list[ContentBlock]) -> list[dict[str, Any]]:
    """Convert ContentBlock list to OpenAI multipart content format."""
    from app.ai.protocols import AudioBlock, ImageBlock, StructuredOutputBlock, TextBlock

    parts: list[dict[str, Any]] = []
    for block in blocks:
        if isinstance(block, TextBlock):
            parts.append({"type": "text", "text": block.text})
        elif isinstance(block, ImageBlock):
            parts.append({
                "type": "image_url",
                "image_url": {
                    "url": f"data:{block.media_type};base64,{block.data}",
                    "detail": "auto",
                },
            })
        elif isinstance(block, AudioBlock):
            # Audio: use transcript if available, otherwise add as text placeholder
            transcript = block.transcript or "[Audio content — transcription not available]"
            parts.append({"type": "text", "text": f"[Audio transcript]: {transcript}"})
        elif isinstance(block, StructuredOutputBlock):
            # StructuredOutputBlock is handled at the request level (response_format),
            # not in message content. Skip here — handled in complete()/stream().
            pass
        else:
            # Unknown block type — convert to text representation
            parts.append({"type": "text", "text": str(block)})
    return parts or [{"type": "text", "text": ""}]
```

#### Vision capability check

Before sending images, check the capability registry (if enabled) to verify the model supports vision. If the model lacks vision capability, replace `ImageBlock`s with text descriptions:

```python
def _check_vision_capability(self, model: str) -> bool:
    """Check if the model supports vision via capability registry."""
    try:
        from app.ai.capability_registry import ModelCapability, get_capability_registry
        registry = get_capability_registry()
        spec = registry.get(model)
        if spec is None:
            return True  # Unknown model — assume capable (fail at API level)
        return ModelCapability.VISION in spec.capabilities
    except Exception:
        return True  # Registry unavailable — don't block

def _build_messages_payload(self, messages: list[Message], model: str | None = None) -> list[dict[str, Any]]:
    """Build messages payload with vision capability check."""
    from app.ai.protocols import ImageBlock

    has_vision = self._check_vision_capability(model or self._model) if model else True
    result: list[dict[str, Any]] = []
    for m in messages:
        if isinstance(m.content, str):
            result.append({"role": m.role, "content": m.content})
        else:
            blocks = m.content
            if not has_vision:
                # Replace images with text descriptions
                blocks = [
                    TextBlock(text=f"[Image: {b.media_type}, {len(b.data)} bytes base64]")
                    if isinstance(b, ImageBlock) else b
                    for b in blocks
                ]
            result.append({"role": m.role, "content": self._serialize_content(blocks)})
    return result
```

### Step 3: Anthropic Adapter — `_serialize_content()` (`app/ai/adapters/anthropic.py`)

Same pattern, different format:

```python
@staticmethod
def _serialize_content(blocks: list[ContentBlock]) -> list[dict[str, Any]]:
    """Convert ContentBlock list to Anthropic content block format."""
    from app.ai.protocols import AudioBlock, ImageBlock, StructuredOutputBlock, TextBlock

    parts: list[dict[str, Any]] = []
    for block in blocks:
        part: dict[str, Any] = {}
        if isinstance(block, TextBlock):
            part = {"type": "text", "text": block.text}
        elif isinstance(block, ImageBlock):
            part = {
                "type": "image",
                "source": {
                    "type": "base64",
                    "media_type": block.media_type,
                    "data": block.data,
                },
            }
        elif isinstance(block, AudioBlock):
            transcript = block.transcript or "[Audio content — transcription not available]"
            part = {"type": "text", "text": f"[Audio transcript]: {transcript}"}
        elif isinstance(block, StructuredOutputBlock):
            # Handled at request level via tool_use pattern — skip in content
            continue
        else:
            part = {"type": "text", "text": str(block)}

        # Preserve cache_control on all block types (Anthropic prompt caching)
        if hasattr(block, "cache_control") and block.cache_control:
            part["cache_control"] = block.cache_control
        parts.append(part)
    return parts or [{"type": "text", "text": ""}]
```

#### Update `_count_message_tokens()` to handle `list[ContentBlock]` content

```python
def _count_message_tokens(self, msg: Message) -> int:
    """Count tokens for a single message including overhead."""
    content_tokens = self._count_content_tokens(msg.content)
    role_tokens = self._count_text_tokens(msg.role)
    return content_tokens + role_tokens + _MESSAGE_OVERHEAD_TOKENS

def _count_content_tokens(self, content: str | list[object]) -> int:
    """Count tokens for message content (text string or ContentBlock list)."""
    if isinstance(content, str):
        return self._count_text_tokens(content)

    from app.ai.protocols import AudioBlock, ImageBlock, TextBlock

    total = 0
    for block in content:
        if isinstance(block, TextBlock):
            total += self._count_text_tokens(block.text)
        elif isinstance(block, ImageBlock):
            # OpenAI: ~85 tokens base + 170 per 512x512 tile (detail=auto)
            # Conservative estimate: 765 tokens per image (mid-size)
            total += _IMAGE_TOKEN_ESTIMATE
        elif isinstance(block, AudioBlock):
            if block.transcript:
                total += self._count_text_tokens(block.transcript)
            else:
                # Estimate: ~1 token per 10 bytes of base64 audio data
                total += max(100, len(block.data) // 40)
        else:
            # StructuredOutputBlock or unknown — minimal overhead
            total += 50
    return total
```

### Step 5: Input Validation (`app/ai/adapters/`)

Add image data validation before serialization in both adapters. Create a shared validation function:

```python
# In each adapter's _serialize_content or _build_messages_payload:

_MAX_IMAGE_SIZE_BYTES = 20 * 1024 * 1024  # 20MB base64 (~15MB raw)
_VALID_IMAGE_TYPES = frozenset({"image/png", "image/jpeg", "image/gif", "image/webp"})

def _validate_image(block: ImageBlock) -> None:
    """Validate image block data before sending to provider."""
    if block.media_type not in _VALID_IMAGE_TYPES:
        raise AIExecutionError(
            f"Unsupported image type: {block.media_type}. "
            f"Supported: {', '.join(sorted(_VALID_IMAGE_TYPES))}"
        )
    if len(block.data) > _MAX_IMAGE_SIZE_BYTES:
        raise AIExecutionError(
            f"Image data exceeds maximum size ({_MAX_IMAGE_SIZE_BYTES // (1024*1024)}MB)"
        )
```

Call `_validate_image()` inside `_serialize_content()` before building the image payload for both adapters. Place the validation function at module level in each adapter file (not shared — each adapter is self-contained).
