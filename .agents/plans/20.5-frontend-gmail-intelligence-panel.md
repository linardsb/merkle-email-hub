**1b. Add service method** to `app/qa_engine/service.py`:

```python
async def check_bimi_readiness(self, data: BIMICheckRequest) -> BIMICheckResponse:
    """Check BIMI readiness for a domain."""
    settings = get_settings()
    if not settings.qa_bimi.enabled:
        raise ForbiddenError("BIMI readiness check is not enabled")

    from app.qa_engine.bimi import BIMIReadinessChecker

    checker = BIMIReadinessChecker()
    status = await checker.check_domain(data.domain)

    return BIMICheckResponse(
        domain=status.domain,
        ready=status.ready,
        dmarc_ready=status.dmarc_ready,
        dmarc_policy=status.dmarc_policy,
        dmarc_record=status.dmarc_record,
        dmarc_info=DMARCInfoSchema(
            policy=status.dmarc_info.policy,
            subdomain_policy=status.dmarc_info.subdomain_policy,
            pct=status.dmarc_info.pct,
        ) if status.dmarc_info else None,
        bimi_record_exists=status.bimi_record_exists,
        bimi_record=status.bimi_record,
        bimi_svg_url=status.bimi_svg_url,
        bimi_authority_url=status.bimi_authority_url,
        svg_valid=status.svg_valid,
        svg_validation=SVGValidationSchema(
            valid=status.svg_validation.valid,
            issues=status.svg_validation.issues,
        ) if status.svg_validation else None,
        cmc_status=status.cmc_status,
        generated_record=status.generated_record,
        issues=status.issues,
    )
```

```python
@router.post(
    "/bimi-check",
    response_model=BIMICheckResponse,
    status_code=200,
    summary="Check BIMI readiness for a domain",

@limiter.limit("5/minute")
async def bimi_check(
    request: Request,
    data: BIMICheckRequest,
    service: QAEngineService = Depends(get_service),
    _user: User = Depends(get_current_user),
) -> BIMICheckResponse:
    """Check DMARC, BIMI DNS record, SVG validation, and CMC status."""
    _ = request
    return await service.check_bimi_readiness(data)
```
