### Verification Metadata Snapshot Test Pattern

```python
def test_mock_vlm_loop_produces_metadata(self, case_id: str) -> None:
    case_dir = _DEBUG_DIR / case_id
    structure = load_structure_from_json(case_dir / "structure.json")
    tokens = load_tokens_from_json(case_dir / "tokens.json")
    converter = DesignConverterService()

    # Mock VLM to return decreasing corrections across iterations
    mock_results = [
        VerificationResult(iteration=0, fidelity_score=0.80, ...corrections=3),
        VerificationResult(iteration=1, fidelity_score=0.95, ...corrections=1),
        VerificationResult(iteration=2, fidelity_score=0.99, converged=True),
    ]

    with (
        _mock_vlm_enabled(),
        patch("app.design_sync.visual_verify.run_verification_loop", ...) as mock_loop,
    ):
        mock_loop.return_value = _make_loop_result(iterations_count=3)
        result = converter.convert(structure, tokens)

    assert result.verification_iterations == 3
    assert result.verification_final_fidelity >= 0.95
```
