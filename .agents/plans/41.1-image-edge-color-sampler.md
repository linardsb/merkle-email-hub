## Implementation Steps

### Step 1: Create `app/design_sync/image_sampler.py`

```python
"""Image edge color sampling for background color continuity."""
from __future__ import annotations

import io
from pathlib import Path
from typing import Literal

from PIL import Image
import numpy as np

from app.core.logging import get_logger

logger = get_logger(__name__)


def sample_edge_color(
    image_path: Path,
    edge: Literal["top", "bottom"],
    *,
    strip_height: int = 4,
    tolerance: int = 10,
    min_uniformity: float = 0.80,
) -> str | None:
    """Sample the dominant color of an image edge strip.

    Reads a pixel strip from the top or bottom of the image, clusters
    pixels by RGB proximity (±tolerance per channel), and returns the
    hex color if the largest cluster covers ≥min_uniformity of pixels.

    Returns None for photographic/gradient edges or on any error.
    """
    path = Path(image_path)
    if not path.is_file():
        logger.warning("design_sync.edge_sample_missing_file", path=str(path))
        return None

    try:
        img = Image.open(path).convert("RGB")
    except Exception as exc:
        logger.warning("design_sync.edge_sample_open_failed", path=str(path), error=str(exc))
        return None

    w, h = img.size
    if h < strip_height or w == 0:
        return None

    # Extract strip
    if edge == "top":
        strip = img.crop((0, 0, w, strip_height))
    else:
        strip = img.crop((0, h - strip_height, w, h))

    pixels = np.array(strip).reshape(-1, 3)  # (N, 3) uint8
    return _dominant_color(pixels, tolerance=tolerance, min_uniformity=min_uniformity)


def _dominant_color(
    pixels: np.ndarray,
    *,
    tolerance: int,
    min_uniformity: float,
) -> str | None:
    """Find dominant color in pixel array via greedy clustering.

    Iterates pixels, assigns each to the first cluster whose centroid
    is within ±tolerance per RGB channel, or starts a new cluster.
    Returns hex of largest cluster if it meets min_uniformity threshold.
    """
    if len(pixels) == 0:
        return None

    # Greedy single-pass clustering
    clusters: list[tuple[np.ndarray, int]] = []  # (sum_rgb, count)

    for px in pixels:
        assigned = False
        for i, (rgb_sum, count) in enumerate(clusters):
            centroid = rgb_sum / count
            if np.all(np.abs(px.astype(np.int16) - centroid.astype(np.int16)) <= tolerance):
                clusters[i] = (rgb_sum + px.astype(np.int64), count + 1)
                assigned = True
                break
        if not assigned:
            clusters.append((px.astype(np.int64).copy(), 1))

    if not clusters:
        return None

    # Find largest cluster
    best_sum, best_count = max(clusters, key=lambda c: c[1])
    uniformity = best_count / len(pixels)

    if uniformity < min_uniformity:
        return None

    centroid = (best_sum / best_count).astype(np.uint8)
    r, g, b = int(centroid[0]), int(centroid[1]), int(centroid[2])
    hex_color = f"#{r:02X}{g:02X}{b:02X}"

    logger.info(
        "design_sync.edge_color_sampled",
        color=hex_color,
        uniformity=round(uniformity, 3),
        total_pixels=len(pixels),
    )
    return hex_color
```

**Key design decisions:**
- Greedy single-pass clustering (no scipy dependency) — sufficient for edge strips where solid colors dominate
- `np.int16` for diff to avoid uint8 underflow
- `np.int64` for accumulator to avoid overflow on large strips
- Configurable params as kwargs with sane defaults — no feature flag needed at utility level
- Returns uppercase hex (`#0252B5`) matching Figma export convention

```python
"""Tests for image edge color sampling."""
from __future__ import annotations

import io
from pathlib import Path

import numpy as np
import pytest
from PIL import Image

from app.design_sync.image_sampler import sample_edge_color, _dominant_color


# ---------------------------------------------------------------------------
# Helpers
# ---------------------------------------------------------------------------

def _make_solid_png(w: int, h: int, rgb: tuple[int, int, int]) -> bytes:
    img = Image.new("RGB", (w, h), rgb)
    buf = io.BytesIO()
    img.save(buf, format="PNG")
    return buf.getvalue()


def _make_split_png(
    w: int, h: int,
    top_rgb: tuple[int, int, int],
    bottom_rgb: tuple[int, int, int],
    split_y: int | None = None,
) -> bytes:
    """Image with top half one color, bottom half another."""
    split = split_y or h // 2
    img = Image.new("RGB", (w, h), top_rgb)
    for y in range(split, h):
        for x in range(w):
            img.putpixel((x, y), bottom_rgb)
    buf = io.BytesIO()
    img.save(buf, format="PNG")
    return buf.getvalue()


def _make_noisy_png(w: int, h: int, seed: int = 42) -> bytes:
    """Random pixel noise — no dominant color."""
    rng = np.random.default_rng(seed)
    arr = rng.integers(0, 256, size=(h, w, 3), dtype=np.uint8)
    img = Image.fromarray(arr, "RGB")
    buf = io.BytesIO()
    img.save(buf, format="PNG")
    return buf.getvalue()


# ---------------------------------------------------------------------------
# Tests: sample_edge_color
# ---------------------------------------------------------------------------

class TestSampleEdgeColor:
    def test_solid_bottom_returns_hex(self, tmp_path: Path) -> None:
        p = tmp_path / "solid.png"
        p.write_bytes(_make_solid_png(100, 50, (2, 82, 181)))
        assert sample_edge_color(p, "bottom") == "#0252B5"

    def test_solid_top_returns_hex(self, tmp_path: Path) -> None:
        p = tmp_path / "solid.png"
        p.write_bytes(_make_solid_png(100, 50, (255, 128, 0)))
        assert sample_edge_color(p, "top") == "#FF8000"

    def test_split_image_top_edge(self, tmp_path: Path) -> None:
        p = tmp_path / "split.png"
        p.write_bytes(_make_split_png(100, 100, (255, 0, 0), (0, 0, 255)))
        assert sample_edge_color(p, "top") == "#FF0000"

    def test_split_image_bottom_edge(self, tmp_path: Path) -> None:
        p = tmp_path / "split.png"
        p.write_bytes(_make_split_png(100, 100, (255, 0, 0), (0, 0, 255)))
        assert sample_edge_color(p, "bottom") == "#0000FF"

    def test_noisy_returns_none(self, tmp_path: Path) -> None:
        p = tmp_path / "noisy.png"
        p.write_bytes(_make_noisy_png(200, 200))
        assert sample_edge_color(p, "bottom") is None

    def test_missing_file_returns_none(self, tmp_path: Path) -> None:
        assert sample_edge_color(tmp_path / "missing.png", "top") is None

    def test_tiny_image_below_strip_height(self, tmp_path: Path) -> None:
        p = tmp_path / "tiny.png"
        p.write_bytes(_make_solid_png(10, 2, (0, 0, 0)))
        # Default strip_height=4, image height=2 → None
        assert sample_edge_color(p, "top") is None

    def test_custom_strip_height(self, tmp_path: Path) -> None:
        p = tmp_path / "small.png"
        p.write_bytes(_make_solid_png(10, 2, (0, 0, 0)))
        assert sample_edge_color(p, "top", strip_height=2) == "#000000"

    def test_tolerance_clusters_near_colors(self, tmp_path: Path) -> None:
        """Pixels within ±10 tolerance should cluster together."""
        img = Image.new("RGB", (100, 10), (100, 100, 100))
        # Add slight variation in bottom strip
        for x in range(100):
            for y in range(6, 10):
                img.putpixel((x, y), (105, 95, 100))  # within ±10
        buf = io.BytesIO()
        img.save(buf, format="PNG")
        p = tmp_path / "near.png"
        p.write_bytes(buf.getvalue())
        result = sample_edge_color(p, "bottom")
        assert result is not None  # Should cluster

    def test_beyond_tolerance_returns_none(self, tmp_path: Path) -> None:
        """Alternating distant colors should not cluster."""
        img = Image.new("RGB", (100, 10), (0, 0, 0))
        for x in range(100):
            for y in range(6, 10):
                color = (255, 0, 0) if x % 2 == 0 else (0, 0, 255)
                img.putpixel((x, y), color)
        buf = io.BytesIO()
        img.save(buf, format="PNG")
        p = tmp_path / "alt.png"
        p.write_bytes(buf.getvalue())
        assert sample_edge_color(p, "bottom") is None

    def test_corrupt_file_returns_none(self, tmp_path: Path) -> None:
        p = tmp_path / "corrupt.png"
        p.write_bytes(b"not an image")
        assert sample_edge_color(p, "bottom") is None

    def test_1px_wide_image(self, tmp_path: Path) -> None:
        p = tmp_path / "narrow.png"
        p.write_bytes(_make_solid_png(1, 10, (128, 64, 32)))
        assert sample_edge_color(p, "top") == "#804020"


# ---------------------------------------------------------------------------
# Tests: _dominant_color (unit)
# ---------------------------------------------------------------------------

class TestDominantColor:
    def test_empty_array(self) -> None:
        assert _dominant_color(np.array([]).reshape(0, 3), tolerance=10, min_uniformity=0.8) is None

    def test_single_pixel(self) -> None:
        px = np.array([[42, 100, 200]], dtype=np.uint8)
        assert _dominant_color(px, tolerance=10, min_uniformity=0.8) == "#2A64C8"

    def test_uniform_pixels(self) -> None:
        px = np.full((500, 3), [10, 20, 30], dtype=np.uint8)
        assert _dominant_color(px, tolerance=10, min_uniformity=0.8) == "#0A141E"

    def test_below_threshold(self) -> None:
        # 50/50 split — largest cluster = 50%, below 80%
        a = np.full((50, 3), [255, 0, 0], dtype=np.uint8)
        b = np.full((50, 3), [0, 0, 255], dtype=np.uint8)
        px = np.vstack([a, b])
        assert _dominant_color(px, tolerance=10, min_uniformity=0.8) is None
```

## Preflight Warnings

- Greedy clustering is O(N·K) where K = number of clusters. For a 4px × 1200px strip (4800 pixels), this is fast. If strip_height or image width grows large, consider numpy vectorized approach.
- `_dominant_color` iterates Python-level loop over pixels — acceptable for edge strips but not for full images.
- No feature flag needed for 41.1 (pure utility). Feature flag `DESIGN_SYNC__EDGE_COLOR_ENABLED` deferred to 41.2 pipeline integration.

## Security Checklist

- No endpoints added (pure utility, no HTTP surface)
- File path input: `Path.is_file()` check prevents traversal to non-existent paths
- No user input reaches this code directly — called from internal converter pipeline
- Pillow: no untrusted image decompression bombs — images come from Figma export pipeline (trusted source). Consider `Image.MAX_IMAGE_PIXELS` if paranoid.
