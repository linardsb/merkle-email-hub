# Plan: 35.9 Figma Webhooks & Live Preview Sync

## Context

Current flow: designer edits Figma → manually clicks "Sync" → waits for API → views tokens. This breaks the design-to-code feedback loop. Figma webhooks (`FILE_UPDATE`) fire within seconds of save. Combined with Phase 33.8 `_compute_token_diff()` and Phase 24 WebSocket infrastructure (`app/streaming/websocket/`), this enables near-real-time preview updates.

### Step 5: Webhook Handler (`app/design_sync/webhook.py`)

New module — core webhook logic:

```python
"""Figma webhook signature verification, event handling, and debounced sync."""

import asyncio
import hashlib
import hmac
from datetime import datetime, timezone

from app.core.config import get_settings
from app.core.logging import get_logger
from app.core.redis import get_redis, redis_available
from app.design_sync.exceptions import WebhookSignatureError
from app.design_sync.schemas import FigmaWebhookPayload

logger = get_logger(__name__)

def verify_signature(payload: bytes, signature: str, passcode: str) -> None:
    """Verify Figma webhook HMAC-SHA256 signature. Raises WebhookSignatureError."""
    expected = hmac.new(
        passcode.encode(), payload, hashlib.sha256
    ).hexdigest()
    if not hmac.compare_digest(expected, signature):
        raise WebhookSignatureError("Invalid webhook signature")

async def enqueue_debounced_sync(file_key: str, connection_id: int) -> None:
    """Set Redis debounce key. Background task triggers sync after TTL expires."""
    settings = get_settings()
    ttl = settings.design_sync.webhook_debounce_seconds
    redis = await get_redis()
    key = f"figma_webhook:{file_key}"
    value = f"{connection_id}:{datetime.now(timezone.utc).isoformat()}"
    await redis.setex(key, ttl, value)
    logger.info("design_sync.webhook_debounced", file_key=file_key, ttl=ttl)

async def process_debounced_sync(file_key: str) -> int | None:
    """Check if debounce expired and return connection_id, or None if still pending."""
    redis = await get_redis()
    key = f"figma_webhook:{file_key}"
    value = await redis.get(key)
    if value is not None:
        return None  # Still within debounce window
    # Key expired — caller should trigger sync
    # Return connection_id from the last enqueue (stored before expiry)
    # This is handled by the background worker pattern
    return None
```

**Debounce strategy**: On webhook, set `figma_webhook:{file_key}` with 5s TTL + schedule an `asyncio.create_task` that sleeps `ttl + 0.5s`, then checks if the key expired (no new webhooks). If expired, trigger `sync_connection()`.

```python
async def register_figma_webhook(
    self, connection_id: int, *, team_id: str, user: User
) -> str:
    """Register Figma webhook for a connection. Returns webhook_id."""
    conn = await self._repo.get_connection(connection_id)
    if conn is None:
        raise ConnectionNotFoundError(connection_id)
    if conn.provider != "figma":
        raise UnsupportedProviderError("Webhooks only supported for Figma")

    settings = get_settings()
    access_token = decrypt_token(conn.encrypted_token, settings.design_sync.encryption_key or settings.auth.jwt_secret_key)
    figma = FigmaDesignSyncService(access_token)

    endpoint = f"{settings.app.base_url}/api/v1/design-sync/webhooks/figma"
    webhook_id = await figma.register_webhook(
        team_id=team_id,
        endpoint=endpoint,
        passcode=settings.design_sync.figma_webhook_passcode,
    )
    await self._repo.update_webhook_id(conn, webhook_id)
    logger.info("design_sync.webhook_registered", connection_id=connection_id, webhook_id=webhook_id)
    return webhook_id

async def unregister_figma_webhook(self, connection_id: int, *, user: User) -> None:
    """Remove Figma webhook for a connection."""
    conn = await self._repo.get_connection(connection_id)
    if conn is None:
        raise ConnectionNotFoundError(connection_id)
    if not conn.webhook_id:
        return

    settings = get_settings()
    access_token = decrypt_token(conn.encrypted_token, settings.design_sync.encryption_key or settings.auth.jwt_secret_key)
    figma = FigmaDesignSyncService(access_token)
    await figma.delete_webhook(conn.webhook_id)
    await self._repo.update_webhook_id(conn, None)
    logger.info("design_sync.webhook_unregistered", connection_id=connection_id)

async def handle_webhook_sync(self, connection_id: int) -> DesignSyncUpdateMessage | None:
    """Run sync triggered by webhook, compute diff, return WS message if changes found."""
    conn = await self._repo.get_connection(connection_id)
    if conn is None:
        return None
    # Sync (reuses existing pipeline)
    await self.sync_connection(connection_id, user=None, _webhook=True)
    # Compute diff
    diff_entries = await self.get_token_diff(connection_id)
    if not diff_entries.entries:
        return None  # No changes
    return DesignSyncUpdateMessage(
        connection_id=connection_id,
        diff_summary=_format_diff_summary(diff_entries.entries),
        total_changes=len(diff_entries.entries),
        timestamp=datetime.now(timezone.utc),
    )
```

```python
@router.post("/webhooks/figma", status_code=200, include_in_schema=False)
async def handle_figma_webhook(request: Request, db: AsyncSession = Depends(get_db)) -> dict:
    """Receive Figma FILE_UPDATE webhook. Must respond < 5s."""
    settings = get_settings()
    if not settings.design_sync.figma_webhook_enabled:
        return {"status": "disabled"}

    # 1. Verify signature
    body = await request.body()
    signature = request.headers.get("X-Figma-Signature", "")
    verify_signature(body, signature, settings.design_sync.figma_webhook_passcode)

    # 2. Parse payload
    payload = FigmaWebhookPayload.model_validate_json(body)
    if payload.event_type != "FILE_UPDATE":
        return {"status": "ignored"}

    # 3. Look up connection
    service = DesignSyncService(db)
    conn = await service._repo.get_connection_by_file_ref(payload.file_key)
    if conn is None:
        logger.info("design_sync.webhook_no_connection", file_key=payload.file_key)
        return {"status": "ok"}

    # 4. Enqueue debounced sync (fire-and-forget)
    await enqueue_debounced_sync(payload.file_key, conn.id)
    asyncio.create_task(_debounced_sync_worker(conn.id, payload.file_key, db))

    return {"status": "ok"}

@router.post("/connections/{connection_id}/webhook", status_code=201)
async def register_webhook(
    connection_id: int,
    team_id: str = Body(..., embed=True),
    user: User = Depends(require_admin),
    db: AsyncSession = Depends(get_db),
) -> dict:
    """Register Figma webhook for live sync. Admin only."""
    service = DesignSyncService(db)
    webhook_id = await service.register_figma_webhook(connection_id, team_id=team_id, user=user)
    return {"webhook_id": webhook_id}

@router.delete("/connections/{connection_id}/webhook", status_code=204)
async def unregister_webhook(
    connection_id: int,
    user: User = Depends(require_admin),
    db: AsyncSession = Depends(get_db),
) -> None:
    """Remove Figma webhook. Admin only."""
    service = DesignSyncService(db)
    await service.unregister_figma_webhook(connection_id, user=user)
```

**Debounced sync worker** (in `webhook.py` or `routes.py`):
```python
async def _debounced_sync_worker(connection_id: int, file_key: str, db: AsyncSession) -> None:
    """Wait for debounce, then run sync + broadcast."""
    settings = get_settings()
    await asyncio.sleep(settings.design_sync.webhook_debounce_seconds + 0.5)
    redis = await get_redis()
    if await redis.get(f"figma_webhook:{file_key}") is not None:
        return  # New event arrived, another worker will handle it

    service = DesignSyncService(db)
    msg = await service.handle_webhook_sync(connection_id)
    if msg is None:
        return

    # Broadcast via WebSocket
    conn = await service._repo.get_connection(connection_id)
    if conn and conn.project_id:
        room_id = f"project:{conn.project_id}"
        manager = get_collab_manager()
        await manager.broadcast_json(room_id, msg.model_dump(mode="json"))
        logger.info("design_sync.webhook_broadcast", connection_id=connection_id, room=room_id)
```

### Step 8: Frontend Hook

**`cms/apps/web/src/hooks/use-design-sync-live.ts`**:

```typescript
import { useCallback, useEffect, useRef } from "react";
import { toast } from "sonner";
import { useWebSocket } from "./use-websocket";

interface DesignSyncUpdate {
  type: "design_sync_update";
  connection_id: number;
  diff_summary: string;
  total_changes: number;
  preview_url: string | null;
  timestamp: string;
}

export function useDesignSyncLive(connectionId: number | null) {
  const debounceRef = useRef<ReturnType<typeof setTimeout>>();
  const { lastMessage } = useWebSocket();

  const handleUpdate = useCallback(
    (update: DesignSyncUpdate) => {
      if (update.connection_id !== connectionId) return;
      // Debounce UI updates (300ms)
      clearTimeout(debounceRef.current);
      debounceRef.current = setTimeout(() => {
        toast.info(`Design updated — ${update.diff_summary}`);
      }, 300);
    },
    [connectionId],
  );

  useEffect(() => {
    if (!connectionId || !lastMessage) return;
    try {
      const data = JSON.parse(lastMessage.data) as DesignSyncUpdate;
      if (data.type === "design_sync_update") {
        handleUpdate(data);
      }
    } catch {
      // Not a design sync message
    }
  }, [connectionId, lastMessage, handleUpdate]);

  useEffect(() => () => clearTimeout(debounceRef.current), []);
}
```

## Security Checklist

| Concern | Mitigation |
|---------|-----------|
| Webhook authenticity | HMAC-SHA256 via `hmac.compare_digest()` (timing-safe) |
| Passcode storage | `DESIGN_SYNC__FIGMA_WEBHOOK_PASSCODE` env var only, never in DB |
| Rate limiting | Webhook endpoint: 60/min per IP via existing `limiter` |
| Auth for registration | `require_admin` dependency — viewer/developer cannot register webhooks |
| WS message scope | `broadcast_json` to `project:{id}` room — only authenticated project members |
| Input validation | `FigmaWebhookPayload` Pydantic model validates structure |
| Redis key injection | `file_key` from validated payload; key prefix `figma_webhook:` scoped |
| BOLA | `sync_connection()` checks user access via `project_id` |

## Preflight Warnings

- `app/design_sync/routes.py` has 24 existing functions — adding 3 more is fine but keep the webhook endpoint at the top for visibility.
