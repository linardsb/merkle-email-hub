# Plan: 35.4 MJML Email Section Templates

## Context
The MJML generation in 35.3 builds MJML programmatically via `mjml_generator.py`. For well-known patterns (hero with background image, 2-column product grid, CTA with VML button), pre-built Jinja2 MJML templates produce higher-quality output by encoding email-specific best practices. This is the industry standard approach (Emailify, Stripo).

### Existing MJML Generator Pattern (35.3)

`generate_mjml(layout, tokens, container_width=600) -> str` builds complete `<mjml>` doc. Calls `_mjml_section()` per section, which delegates to `_mjml_columns()` → `_mjml_column_content()` → `_mjml_text()` / `_mjml_image()` / `_mjml_button()`. Section comment markers (`<!-- section:node_id:type -->`) injected in MJML, then `inject_section_markers()` replaces them with `<div data-section-type data-node-id>` wrappers in compiled HTML.

`_convert_mjml_from_layout()` in converter_service.py calls `generate_mjml()` → `compile_mjml()` → `inject_section_markers()`.

### ComponentMatcher Pattern

`match_section(section, idx, container_width=600) -> ComponentMatch` maps `EmailSectionType` → component slug. Column layouts override section type. `match_all()` iterates sections.

### Jinja2 in Codebase

Jinja2 is already a dependency (used in routes/templates). `autoescape=True` is standard.

### Step 1: Create `MjmlTemplateContext` and `MjmlTemplateEngine`

**File:** `app/design_sync/mjml_template_engine.py`

```python
from __future__ import annotations

from dataclasses import dataclass, field
from pathlib import Path

from jinja2 import Environment, FileSystemLoader, select_autoescape

from app.core.logging import get_logger
from app.design_sync.figma.layout_analyzer import (
    ColumnLayout,
    EmailSection,
    EmailSectionType,
)
from app.design_sync.protocol import ExtractedColor, ExtractedTypography

logger = get_logger(__name__)

_TEMPLATE_DIR = Path(__file__).parent / "mjml_templates"

# Section type → template filename mapping
_SECTION_TEMPLATE_MAP: dict[EmailSectionType, str] = {
    EmailSectionType.HERO: "hero.mjml.j2",
    EmailSectionType.HEADER: "header.mjml.j2",
    EmailSectionType.CTA: "cta.mjml.j2",
    EmailSectionType.FOOTER: "footer.mjml.j2",
    EmailSectionType.SOCIAL: "footer.mjml.j2",  # Social reuses footer template
    EmailSectionType.SPACER: "spacer.mjml.j2",
    EmailSectionType.DIVIDER: "spacer.mjml.j2",  # Divider uses spacer with border
    EmailSectionType.NAV: "header.mjml.j2",  # Nav reuses header template
}

# Column layout → content template mapping
_COLUMN_TEMPLATE_MAP: dict[ColumnLayout, str] = {
    ColumnLayout.SINGLE: "content_single.mjml.j2",
    ColumnLayout.TWO_COLUMN: "content_two_col.mjml.j2",
    ColumnLayout.THREE_COLUMN: "content_three_col.mjml.j2",
    ColumnLayout.MULTI_COLUMN: "content_multi_col.mjml.j2",
}


@dataclass(frozen=True)
class MjmlPalette:
    """Resolved brand palette for template injection."""
    primary: str = "#333333"
    secondary: str = "#666666"
    background: str = "#ffffff"
    text: str = "#000000"
    accent: str = "#0066cc"
    muted: str = "#999999"


@dataclass(frozen=True)
class MjmlTypography:
    """Resolved typography for template injection."""
    heading_font: str = "Arial, Helvetica, sans-serif"
    body_font: str = "Arial, Helvetica, sans-serif"
    heading_size: str = "24px"
    body_size: str = "16px"
    heading_line_height: str = "1.3"
    body_line_height: str = "1.5"
    heading_weight: str = "700"
    body_weight: str = "400"


@dataclass(frozen=True)
class MjmlTemplateContext:
    """Context passed to every MJML Jinja2 template."""
    palette: MjmlPalette = field(default_factory=MjmlPalette)
    typography: MjmlTypography = field(default_factory=MjmlTypography)
    dark_colors: tuple[ExtractedColor, ...] = ()
    container_width: int = 600
    section_padding: str = "20px 0"


class MjmlTemplateEngine:
    """Jinja2-based MJML section template renderer."""

    def __init__(self, template_dir: Path | None = None) -> None:
        resolved = template_dir or _TEMPLATE_DIR
        self._env = Environment(
            loader=FileSystemLoader(str(resolved)),
            autoescape=select_autoescape(default_for_string=True, default=True),
            trim_blocks=True,
            lstrip_blocks=True,
        )
        self._template_dir = resolved

    def resolve_template_name(self, section: EmailSection) -> str:
        """Map section to its MJML template filename."""
        # CONTENT + IMAGE_FULL handled by column layout
        if section.section_type == EmailSectionType.CONTENT:
            return _COLUMN_TEMPLATE_MAP.get(
                section.column_layout, "content_single.mjml.j2"
            )
        # Full-width image
        if (
            section.section_type in (EmailSectionType.CONTENT, EmailSectionType.UNKNOWN)
            and section.images
            and not section.texts
            and not section.buttons
        ):
            return "image_full.mjml.j2"
        return _SECTION_TEMPLATE_MAP.get(
            section.section_type, "content_single.mjml.j2"
        )

    def render_section(
        self, section: EmailSection, ctx: MjmlTemplateContext,
    ) -> str:
        """Render one EmailSection to MJML markup (no <mjml>/<mj-body> wrapper)."""
        template_name = self.resolve_template_name(section)
        template = self._env.get_template(template_name)
        return template.render(
            section=section,
            palette=ctx.palette,
            typo=ctx.typography,
            dark_colors=ctx.dark_colors,
            container_width=ctx.container_width,
            section_padding=ctx.section_padding,
        )

    def render_email(
        self,
        sections: list[EmailSection],
        ctx: MjmlTemplateContext,
        *,
        preheader: str = "",
    ) -> str:
        """Render a complete MJML document from sections."""
        parts: list[str] = []
        for section in sections:
            parts.append(self.render_section(section, ctx))

        dark_css = _build_dark_mode_css(ctx.dark_colors)
        body_mjml = "\n".join(parts)

        return _MJML_DOC_TEMPLATE.format(
            container_width=ctx.container_width,
            body_font=ctx.typography.body_font,
            body_size=ctx.typography.body_size,
            body_line_height=ctx.typography.body_line_height,
            body_color=ctx.palette.text,
            preheader=preheader,
            dark_css=dark_css,
            body=body_mjml,
        )


def build_template_context(
    tokens: "ExtractedTokens",
    *,
    container_width: int = 600,
) -> MjmlTemplateContext:
    """Build MjmlTemplateContext from ExtractedTokens."""
    from app.design_sync.converter import convert_colors_to_palette, convert_typography

    palette_dict = convert_colors_to_palette(tokens.colors)
    typo_dict = convert_typography(tokens.typography)

    palette = MjmlPalette(
        primary=palette_dict.get("primary", "#333333"),
        secondary=palette_dict.get("secondary", "#666666"),
        background=palette_dict.get("background", "#ffffff"),
        text=palette_dict.get("text", "#000000"),
        accent=palette_dict.get("accent", "#0066cc"),
        muted=palette_dict.get("muted", "#999999"),
    )
    typography = MjmlTypography(
        heading_font=typo_dict.get("heading_font", "Arial, Helvetica, sans-serif"),
        body_font=typo_dict.get("body_font", "Arial, Helvetica, sans-serif"),
        heading_size=typo_dict.get("heading_size", "24px"),
        body_size=typo_dict.get("base_size", "16px"),
        heading_line_height=typo_dict.get("heading_line_height", "1.3"),
        body_line_height=typo_dict.get("body_line_height", "1.5"),
        heading_weight=typo_dict.get("heading_weight", "700"),
        body_weight=typo_dict.get("body_weight", "400"),
    )
    dark = tuple(tokens.dark_colors) if tokens.dark_colors else ()

    return MjmlTemplateContext(
        palette=palette,
        typography=typography,
        dark_colors=dark,
        container_width=container_width,
    )


def _build_dark_mode_css(dark_colors: tuple[ExtractedColor, ...]) -> str:
    """Generate @media (prefers-color-scheme: dark) CSS block."""
    if not dark_colors:
        return ""
    rules: list[str] = []
    for dc in dark_colors:
        name = dc.name.lower().replace(" ", "-")
        rules.append(f"  .dark-{name} {{ color: {dc.hex} !important; }}")
    return (
        "@media (prefers-color-scheme: dark) {\n"
        + "\n".join(rules)
        + "\n}"
    )


_MJML_DOC_TEMPLATE = """\
<mjml>
  <mj-head>
    <mj-attributes>
      <mj-all font-family="{body_font}" />
      <mj-text font-size="{body_size}" line-height="{body_line_height}" color="{body_color}" />
      <mj-body width="{container_width}px" />
    </mj-attributes>
    <mj-preview>{preheader}</mj-preview>
    <mj-style>
      {dark_css}
    </mj-style>
  </mj-head>
  <mj-body width="{container_width}px" background-color="#ffffff">
    {body}
  </mj-body>
</mjml>
"""
```

## Preflight Warnings

- `mjml_generator.py` already handles all 10 section types programmatically — template engine is an *alternative* path, not a replacement. Both must coexist.
- `autoescape=True` in Jinja2 will escape `&`, `<`, `>` in text content — verify this doesn't double-escape content that's already HTML-escaped in `TextBlock.content`.
- `ColumnGroup` dataclass may not have `width` attribute — verify before using in `content_multi_col.mjml.j2`.
- MJML `<mj-hero>` is deprecated in MJML 4.x — use `<mj-section>` with `background-url` instead.
- `inject_section_markers()` depends on `<!-- section:X:Y -->` comment format — templates must match exactly.
