# Plan: 11.2a Shared QA Rule Engine — YAML-Driven Check Definitions

## Context

Each QA check currently hardcodes validation logic in Python methods. Adding or tuning rules requires code changes. Task 11.2a introduces a shared rule engine that loads check definitions from YAML files, enabling:
- Adding rules by editing YAML (no Python changes)
- Sharing check type evaluators across all 10+ QA checks
- Exposing rules to agents via RAG knowledge base (`make seed-knowledge`)
- Per-project rule enable/disable by rule ID (future)

The two research docs (`docs/email-accessibility-wcag-aa.md` — 250+ WCAG rules, `docs/html-email-components.md` — 280+ component rules) become machine-parseable YAML powering both QA validation and agent knowledge.

**Scope:** Build the engine + `email_structure.yaml` + refactor `html_validation.py`. The accessibility YAML + rewrite is 11.3's job (it will consume the same engine).

```python
"""Shared rule engine — loads YAML rule definitions and evaluates against lxml DOM.

Used by html_validation (email structure rules), accessibility (WCAG AA rules),
and future checks (dark_mode, fallback, spam, etc.).
Rule YAML files also serve as RAG knowledge base entries for agent context.
"""

from __future__ import annotations

import re
from dataclasses import dataclass, field
from pathlib import Path

import yaml
from lxml.html import HtmlElement

from app.core.logging import get_logger
from app.qa_engine.check_config import QACheckConfig

logger = get_logger(__name__)


@dataclass(frozen=True)
class Rule:
    """A single validation rule loaded from YAML."""

    id: str                          # Unique rule identifier (e.g., "doc-doctype")
    group: str                       # Logical group (e.g., "skeleton", "layout")
    name: str                        # Human-readable name
    check_type: str                  # Evaluator type (attr_present, element_absent, etc.)
    message: str                     # Issue message when rule fails
    wcag: str = ""                   # WCAG criterion (empty for non-a11y rules)
    level: str = ""                  # WCAG level: A, AA, AAA (empty for non-a11y)
    selector: str = ""               # XPath or CSS-like selector for elements
    attr: str = ""                   # Single attribute to check
    attrs: tuple[str, ...] = ()      # Multiple attributes (all must be present)
    value: str = ""                  # Expected attribute value
    values: tuple[str, ...] = ()     # Multiple acceptable values (any match = pass)
    patterns: tuple[str, ...] = ()   # Regex patterns to match
    min_count: int = 0               # Minimum element count (element_count)
    max_count: int = -1              # Maximum element count (-1 = no limit)
    custom_fn: str = ""              # Named Python function for custom type
    deduction_key: str = ""          # Key in defaults.yaml params dict
    default_deduction: float = 0.10  # Fallback deduction if key not in config
    per_element: bool = False        # Deduct per matching element vs once
    cap_key: str = ""                # Key for max_issues_reported in config
    default_cap: int = 10            # Default cap if cap_key not in config


@dataclass
class RuleResult:
    """Result of evaluating a single rule."""

    rule_id: str
    passed: bool
    issues: list[str] = field(default_factory=list)
    deduction: float = 0.0
```

#### 1b: YAML loader

```python
_rules_cache: dict[str, list[Rule]] = {}


def load_rules(yaml_path: Path) -> list[Rule]:
    """Load rules from a YAML file. Cached after first successful load."""
    cache_key = str(yaml_path)
    if cache_key in _rules_cache:
        return _rules_cache[cache_key]

    if not yaml_path.exists():
        logger.warning("rule_engine.yaml_missing", path=cache_key)
        return []

    try:
        with yaml_path.open() as f:
            data: dict[str, Any] = yaml.safe_load(f) or {}
    except yaml.YAMLError:
        logger.error("rule_engine.yaml_invalid", path=cache_key)
        return []

    rules: list[Rule] = []
    for entry in data.get("rules", []):
        try:
            rules.append(Rule(
                id=entry["id"],
                group=entry["group"],
                name=entry["name"],
                check_type=entry["check"],
                message=entry["message"],
                wcag=entry.get("wcag", ""),
                level=entry.get("level", ""),
                selector=entry.get("selector", ""),
                attr=entry.get("attr", ""),
                attrs=tuple(entry.get("attrs", ())),
                value=entry.get("value", ""),
                values=tuple(entry.get("values", ())),
                patterns=tuple(entry.get("patterns", ())),
                min_count=entry.get("min_count", 0),
                max_count=entry.get("max_count", -1),
                custom_fn=entry.get("custom_fn", ""),
                deduction_key=entry.get("deduction_key", ""),
                default_deduction=entry.get("default_deduction", 0.10),
                per_element=entry.get("per_element", False),
                cap_key=entry.get("cap_key", ""),
                default_cap=entry.get("default_cap", 10),
            ))
        except (KeyError, TypeError) as exc:
            logger.warning("rule_engine.rule_skipped", path=cache_key, error=str(exc))

    _rules_cache[cache_key] = rules
    logger.info("rule_engine.rules_loaded", path=cache_key, count=len(rules))
    return rules
```

```python
class CheckFn(Protocol):
    """Signature for check type evaluator functions."""

    def __call__(
        self,
        doc: HtmlElement,
        raw_html: str,
```

#### 1e: Check type evaluators (~15 types)

Each is a pure function. Implement all of:

1. **`attr_present`** — Element(s) matching selector must have the specified attr(s).
   - If `attrs` (tuple): ALL must be present.
   - If `attr` (single): that one must be present.
   - If `per_element`: deduct per non-compliant element (capped). Otherwise deduct once.

2. **`attr_value`** — Attribute must equal expected value.
   - `value` for single expected value. `values` for multiple acceptable values (any match = pass).
   - Same per_element/cap logic.

3. **`attr_empty`** — Attribute must be present and be empty string (`alt=""`).
   - For tracking pixels, decorative images.

4. **`attr_pattern`** — Attribute value must match regex pattern(s).
   - All patterns must match (AND logic).

5. **`attr_absent`** — Attribute must NOT be present on matching elements.
   - For removing `summary` from layout tables.

6. **`element_present`** — At least one element matching selector must exist.
   - Pass if count >= 1. If `min_count` specified, use that.

7. **`element_absent`** — No elements matching selector should exist.
   - Fail if any found. `max_count=0` implied.

8. **`element_count`** — Count of matching elements must be within `min_count`..`max_count`.
   - `max_count=-1` means no upper limit.

9. **`parent_has`** — Each matching element's parent must have specified attr/value.
   - For `<li>` inside `<ul>`/`<ol>`, `<td>` inside `<tr>`, etc.
   - `values` field lists acceptable parent tag names.

10. **`children_match`** — First child of matching element must meet criteria.
    - For `<details>` first child is `<summary>`.
    - `value` = expected first child tag name.

11. **`text_content`** — Element text content must match (or not match) pattern.
    - `patterns` with regex. If pattern starts with `!` (negation prefix), content must NOT match.

12. **`sibling_check`** — Detect consecutive sibling elements sharing same attribute value.
    - For redundant adjacent links with same `href`.

13. **`style_contains`** — CSS in `<style>` blocks or inline `style` must match pattern.
    - Search `<style>` element text and/or inline `style` attributes.

14. **`raw_html_pattern`** — Regex match against raw HTML string.
    - For DOCTYPE detection, MSO conditionals, nested `<a>` tags.
    - Case-insensitive by default. `patterns` field.
    - If ALL patterns match → **pass** (presence check). To check absence, use rule logic.
    - Convention: if message says "Missing X", the rule fails when pattern does NOT match.

15. **`custom`** — Delegates to a named Python function via `_CUSTOM_CHECKS[rule.custom_fn]`.
    - Returns `(issues, total_deduction)` directly.
    - Wraps into `RuleResult`.

**Selector conversion:** lxml doesn't support CSS selectors natively. Use `doc.cssselect(selector)` via `lxml.cssselect` or convert simple selectors to XPath:
- `"html"` → `doc.findall(".//html")` or just use `doc` (it IS the html element)
- `"head meta[charset]"` → `doc.xpath(".//head//meta[@charset]")`
- `"img"` → `doc.iter("img")`
- `"body > table"` → `doc.xpath(".//body/table")`

Since lxml has `cssselect` module, prefer `doc.cssselect(selector)` for complex selectors. For simple tag-name selectors, use `doc.iter(tag)`.

Add `lxml[cssselect]` dependency note (check if `cssselect` is already available — lxml ships with it as optional).

Actually, use a simpler approach: `doc.xpath()` with manually converted paths for the initial set of rules. This avoids the cssselect dependency. The selectors needed for email_structure rules are simple enough:

```python
def _select(doc: HtmlElement, selector: str) -> list[HtmlElement]:
    """Convert simple CSS-like selectors to lxml element lists."""
    # Simple tag: "img", "table", "a"
    if re.match(r"^[a-z]+$", selector):
        return list(doc.iter(selector))
    # Tag with attribute: "meta[charset]", "link[rel='stylesheet']"
    m = re.match(r"^([a-z]+)\[([a-z-]+)(?:=['\"]?([^'\"]*)['\"]?)?\]$", selector)
    if m:
        tag, attr, val = m.group(1), m.group(2), m.group(3)
        elements = list(doc.iter(tag))
        if val is not None:
            return [e for e in elements if (e.get(attr) or "").lower() == val.lower()]
        return [e for e in elements if e.get(attr) is not None]
    # Scoped: "head meta[charset]" -> find meta[charset] under head
    parts = selector.split()
    if len(parts) == 2:
        parent_tag = parts[0]
        parents = list(doc.iter(parent_tag))
        child_selector = parts[1]
        results: list[HtmlElement] = []
        for p in parents:
            results.extend(_select(p, child_selector))
        return results
    # Fallback: try xpath
    try:
        return doc.xpath(f".//{selector.replace(' ', '//')}")
    except Exception:
        return []
```

For the initial 20 rules, this covers all needed selectors. Future rule files can extend the selector parser as needed.

#### 1f: RuleEngine class

```python
class RuleEngine:
    """Evaluates a set of rules against an HTML document."""

    def __init__(self, rules: list[Rule]) -> None:
        self.rules = rules

    def evaluate(
        self,
        doc: HtmlElement,
        raw_html: str,
        config: QACheckConfig | None = None,
    ) -> tuple[list[str], float]:
        """Run all rules, return (all_issues, total_deduction)."""
        all_issues: list[str] = []
        total_deduction = 0.0

        for rule in self.rules:
            check_fn = _CHECK_TYPES.get(rule.check_type)
            if check_fn is None:
                logger.warning(
                    "rule_engine.unknown_check_type",
                    rule_id=rule.id,
                    check_type=rule.check_type,
                )
                continue

            result = check_fn(doc, raw_html, rule, config)
            if not result.passed:
                all_issues.extend(result.issues)
                total_deduction += result.deduction

        return all_issues, total_deduction
```

**Note on `raw_html_pattern` for template/base:** The doctype check uses `raw_html_pattern` to assert a pattern IS present (fail if missing). But template/base checks need to fail if a pattern IS found. Rather than adding inversion logic to `raw_html_pattern`, use `custom` functions for these — they're simple 3-line functions. Change `enhance-no-template` and `enhance-no-base` to `check: "custom"` with `custom_fn: "template_element"` and `custom_fn: "base_tag"`.

For the doctype rule (`doc-doctype`): `raw_html_pattern` should fail when the pattern is NOT found (absence = failure). This is the natural direction for most presence checks.

**Decision:** `raw_html_pattern` convention = **fail when pattern NOT found** (checks for required patterns). For "must not exist" checks, use `element_absent` or `custom`.

**Important design decision:** Custom check functions need the deduction value and cap from the Rule, but the CustomCheckFn protocol only receives `(doc, raw_html, config)`. Two options:

**Option A (chosen):** Custom functions calculate their own deductions using config params directly. They know their deduction keys (hardcoded within each function, matching the YAML rule's `deduction_key`). This is the pattern already used by `html_validation.py` — each method calls `_param(config, "deduction_foo")`.

**Option B:** Extend CustomCheckFn to also receive the Rule. More flexible but adds coupling.

Go with **Option A** — it matches the existing code exactly and keeps the function signatures simple. The RuleEngine's `_check_custom` evaluator calls the custom function, gets back `(issues, deduction)`, and wraps it in a RuleResult.

```python
"""HTML validation check — DOM-parsed structural validation for email HTML.

Uses the shared rule engine with rules loaded from rules/email_structure.yaml.
Complex checks delegated to custom functions in custom_checks.py.
"""

from __future__ import annotations

from pathlib import Path

from lxml import html as lxml_html

from app.qa_engine.check_config import QACheckConfig
from app.qa_engine.schemas import QACheckResult

# Import custom checks to ensure they are registered
import app.qa_engine.custom_checks  # noqa: F401
from app.qa_engine.rule_engine import RuleEngine, load_rules

_RULES_PATH = Path(__file__).parent.parent / "rules" / "email_structure.yaml"


class HtmlValidationCheck:
    """Validates HTML structure and syntax using YAML-driven rule engine.

    Loads rules from rules/email_structure.yaml covering:
    A: Document Skeleton (5 rules)
    B: Tag Integrity (3 rules)
    C: Content Integrity (2 rules)
    D: Email-Specific Structure (4 rules)
    E: Progressive Enhancement (6 rules)
    """

    name = "html_validation"

    def __init__(self) -> None:
        self._rules = load_rules(_RULES_PATH)
        self._engine = RuleEngine(self._rules)

    async def run(self, html: str, config: QACheckConfig | None = None) -> QACheckResult:
        """Run all structural validation checks against the provided HTML."""
        if not html or not html.strip():
            return QACheckResult(
                check_name=self.name,
                passed=False,
                score=0.0,
                details="Empty HTML document",
                severity="error",
            )

        try:
            doc = lxml_html.document_fromstring(html)
        except Exception:
            return QACheckResult(
                check_name=self.name,
                passed=False,
                score=0.0,
                details="HTML could not be parsed",
                severity="error",
            )

        issues, total_deduction = self._engine.evaluate(doc, html, config)

        score = max(0.0, round(1.0 - total_deduction, 2))
        passed = len(issues) == 0
        return QACheckResult(
            check_name=self.name,
            passed=passed,
            score=score,
            details="; ".join(issues) if issues else None,
            severity="error" if not passed else "info",
        )
```
