# Plan: 9.6 Graph-Informed Blueprint Route Selection

## Context

The blueprint engine currently has a **static** campaign graph: scaffolder → qa_gate → maizzle_build → export, with a recovery loop through fixers. The only audience-aware logic is `is_node_relevant()` in `route_advisor.py`, which skips `outlook_fixer` (no Word engine clients) and `dark_mode` (no dark mode required). These are simple boolean filters.

Phase 9.6 extends this to **dynamic** route selection: the route advisor analyses the template content + project subgraph (9.5) to recommend node sequence changes — adding, removing, or reordering nodes based on graph-derived audience data. The advisory runs **before** execution and produces a `RoutingPlan` that the engine follows.

### Key Design Decisions

1. **Advisory, not mandatory** — The route advisor proposes changes; the engine applies them. A `force_full_pipeline` option lets developers override.
2. **Pre-execution only** — The plan is computed once before the main loop. During execution, `is_node_relevant()` still gates individual nodes (backward compatible).
3. **No new API endpoints** — The routing plan is internal. Results surface via `skipped_nodes` + a new `routing_decisions` field on `BlueprintRunResponse`.
4. **Deterministic** — No LLM calls. Rules are pure functions over audience data + HTML content analysis.

from __future__ import annotations

from dataclasses import dataclass
from enum import Enum


class RoutingAction(str, Enum):
    """What the route advisor recommends for a node."""
    SKIP = "skip"           # Node unnecessary for this audience/content
    INCLUDE = "include"     # Node should run (default)
    PRIORITISE = "prioritise"  # Node especially important — run early if possible


@dataclass(frozen=True)
class RoutingDecision:
    """A single routing recommendation for one node."""
    node_name: str
    action: RoutingAction
    reason: str  # Human-readable justification for audit trail


@dataclass(frozen=True)
class RoutingPlan:
    """Pre-execution routing plan for a blueprint run."""
    decisions: tuple[RoutingDecision, ...]
    skip_nodes: frozenset[str]       # Convenience: node names with action=SKIP
    prioritise_nodes: frozenset[str]  # Convenience: node names with action=PRIORITISE
    force_full: bool = False          # Developer override — ignore skip decisions
```

import re

# --- Content detection helpers (pure, no I/O) ---

_AMP_PATTERNS = re.compile(
    r"<amp-|amp4email|⚡4email", re.IGNORECASE
)

_PERSONALISATION_PATTERNS = re.compile(
    r"\{\{|{%|%%\[|AMPscript|ContentBlockByName|Lookup\(", re.IGNORECASE
)

_ACCESSIBILITY_KEYWORDS = re.compile(
    r"wcag|accessibility|a11y|screen.?reader|alt.?text|contrast", re.IGNORECASE
)


def _has_amp_content(html: str, brief: str) -> bool:
    """Check if template uses AMP for Email components."""
    return bool(_AMP_PATTERNS.search(html) or _AMP_PATTERNS.search(brief))


def _has_personalisation_syntax(html: str, brief: str) -> bool:
    """Check if template contains Liquid/AMPscript/dynamic content markers."""
    return bool(_PERSONALISATION_PATTERNS.search(html) or _PERSONALISATION_PATTERNS.search(brief))


def _mentions_accessibility(brief: str) -> bool:
    """Check if brief explicitly requests accessibility work."""
    return bool(_ACCESSIBILITY_KEYWORDS.search(brief))
```

### Step 3: Implement `build_routing_plan()` in `route_advisor.py`

This is the core function. It combines audience data + content analysis into a routing plan:

```python
def build_routing_plan(
    node_names: list[str],
    audience_profile: AudienceProfile | None,
    html: str,
    brief: str,
    force_full: bool = False,
) -> RoutingPlan:
    """Build a pre-execution routing plan based on audience + content analysis.

    Rules (applied in order):
    1. Audience-based: nodes without matching audience are SKIP
    2. Content-based: nodes whose domain isn't present in HTML/brief are SKIP
    3. Priority: nodes whose domain IS heavily present get PRIORITISE
    4. Deterministic nodes (qa_gate, recovery_router, etc.) are never skipped

    Args:
        node_names: All node names in the blueprint definition.
        audience_profile: Target audience constraints (may be None).
        html: Initial HTML content (may be empty for generation briefs).
        brief: Campaign brief text.
        force_full: If True, all SKIP decisions become INCLUDE.

    Returns:
        RoutingPlan with per-node decisions and convenience sets.
    """
    decisions: list[RoutingDecision] = []

    # Deterministic nodes are never skipped
    deterministic = {"qa_gate", "recovery_router", "maizzle_build", "export"}

    # Pre-compute content signals once
    has_amp = _has_amp_content(html, brief)
    has_personalisation = _has_personalisation_syntax(html, brief)
    mentions_a11y = _mentions_accessibility(brief)

    for name in node_names:
        if name in deterministic:
            continue  # No decision needed — always run

        # Rule 1: Audience-based skip (existing is_node_relevant logic)
        if not is_node_relevant(name, audience_profile):
            decisions.append(RoutingDecision(
                node_name=name,
                action=RoutingAction.SKIP,
                reason=_audience_skip_reason(name, audience_profile),
            ))
            continue

        # Rule 2: Content-based skip for fixers (not scaffolder — it generates)
        decision = _content_based_decision(
            name, audience_profile, html, brief,
            has_amp=has_amp,
            has_personalisation=has_personalisation,
            mentions_a11y=mentions_a11y,
        )
        if decision is not None:
            decisions.append(decision)

    skip_nodes = frozenset(
        d.node_name for d in decisions if d.action == RoutingAction.SKIP
    )
    prioritise_nodes = frozenset(
        d.node_name for d in decisions if d.action == RoutingAction.PRIORITISE
    )

    logger.info(
        "route_advisor.plan_built",
        total_nodes=len(node_names),
        skip=len(skip_nodes),
        prioritise=len(prioritise_nodes),
        force_full=force_full,
    )

    return RoutingPlan(
        decisions=tuple(decisions),
        skip_nodes=skip_nodes,
        prioritise_nodes=prioritise_nodes,
        force_full=force_full,
    )


def _audience_skip_reason(name: str, profile: AudienceProfile | None) -> str:
    """Generate human-readable reason for audience-based skip."""
    if name == "outlook_fixer":
        return "No Microsoft Word-engine clients in target audience"
    if name == "dark_mode":
        return "No personas require dark mode support"
    return f"Node '{name}' not relevant for target audience"


def _content_based_decision(
    name: str,
    audience_profile: AudienceProfile | None,
    html: str,
    brief: str,
    *,
    has_amp: bool,
    has_personalisation: bool,
    mentions_a11y: bool,
) -> RoutingDecision | None:
    """Content-based routing decision for a node. Returns None if no decision."""
    # Personalisation node: skip if no dynamic content markers
    if name == "personalisation" and not has_personalisation:
        return RoutingDecision(
            node_name=name,
            action=RoutingAction.SKIP,
            reason="No Liquid/AMPscript/dynamic content in template or brief",
        )

    # Accessibility node: prioritise if brief mentions it explicitly
    if name == "accessibility" and mentions_a11y:
        return RoutingDecision(
            node_name=name,
            action=RoutingAction.PRIORITISE,
            reason="Brief explicitly requests accessibility work",
        )

    # Code reviewer: prioritise if HTML is large (>50KB — nearing Gmail clip threshold)
    if name == "code_reviewer" and len(html.encode("utf-8")) > 50_000:
        return RoutingDecision(
            node_name=name,
            action=RoutingAction.PRIORITISE,
            reason="HTML exceeds 50KB — code review can optimize before Gmail clip threshold",
        )

    # Dark mode: prioritise if audience has mixed engines (Word + WebKit/Blink)
    if name == "dark_mode" and audience_profile is not None:
        engines = {c.engine for c in audience_profile.clients}
        if len(engines) > 1 and ClientEngine.WORD in engines:
            return RoutingDecision(
                node_name=name,
                action=RoutingAction.PRIORITISE,
                reason="Mixed rendering engines including Word — dark mode requires careful handling",
            )

    return None
```

def _skip_summary(node_name: str, plan: RoutingPlan) -> str:
    """Get the skip reason from routing plan for the progress log."""
    for d in plan.decisions:
        if d.node_name == node_name and d.action.value == "skip":
            return f"Skipped: {d.reason}"
    return "Skipped: not relevant for target audience"
