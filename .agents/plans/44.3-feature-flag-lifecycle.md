## Implementation Steps

### Step 1: Create `feature-flags.yaml`

Root-level YAML manifest. Schema:

```yaml
# Feature Flag Manifest — all flags must be registered here
# CI warns on flags >90 days without removal_date, fails on >180 days
flags:
  - name: COLLAB_WS__ENABLED           # env var name exactly as in .env.example
    description: Real-time WebSocket collaboration
    owner: platform-team
    created: "2026-01-15"
    removal_date: null                  # null = permanent, needs justification
    status: beta                        # alpha | beta | ga | deprecated
    permanent_reason: "Requires per-deployment opt-in"  # required when removal_date is null
```

**All 33 .env.example flags + additional config.py-only flags must be registered.** Group by config class prefix (AI, KNOWLEDGE, QA, BLUEPRINT, etc.). Use `config.py` inline comments for env var names. For flags without env var in .env.example (e.g., `DESIGN_SYNC__CONVERTER_ENABLED`, `BLUEPRINT__RECOVERY_LEDGER_ENABLED`), still register them — the audit script checks config.py too.

**Source of truth for flag names:** The env var comment in `config.py` (e.g., `# AI__ADAPTIVE_ROUTING_ENABLED`). For flags without a comment, construct from class prefix + field name (e.g., `KnowledgeConfig.auto_tag_enabled` → `KNOWLEDGE__AUTO_TAG_ENABLED`).

### Step 2: Create `scripts/flag-audit.py`

Follow `validate-overlays.py` pattern (the closest existing analogue).

**Core logic:**

```python
#!/usr/bin/env python3
"""Feature flag lifecycle audit (Phase 44.3)."""

from __future__ import annotations

import argparse
import re
import sys
from dataclasses import dataclass
from datetime import date, timedelta
from pathlib import Path

import yaml

_PROJECT_ROOT = Path(__file__).resolve().parents[1]

WARN_THRESHOLD_DAYS = 90
ERROR_THRESHOLD_DAYS = 180

@dataclass
class Finding:
    flag: str
    code: str  # UNREGISTERED | PAST_REMOVAL_DATE | NO_REMOVAL_PLAN_90D | NO_REMOVAL_PLAN_180D | ORPHANED_MANIFEST
    severity: str  # error | warn
    detail: str

def parse_env_flags(env_path: Path) -> set[str]:
    """Extract _ENABLED flag names from .env.example."""
    # Match both commented (# FOO=) and uncommented (FOO=) lines
    pattern = re.compile(r"^#?\s*([A-Z_]+_ENABLED)\s*=")
    flags: set[str] = set()
    for line in env_path.read_text().splitlines():
        m = pattern.match(line)
        if m:
            flags.add(m.group(1))
    return flags

def parse_config_flags(config_path: Path) -> set[str]:
    """Extract env var names from config.py inline comments."""
    # Match: field_name: bool = ...  # ENV_VAR_NAME
    pattern = re.compile(r"#\s*([A-Z_]+_ENABLED)")
    flags: set[str] = set()
    for line in config_path.read_text().splitlines():
        m = pattern.search(line)
        if m:
            flags.add(m.group(1))
    return flags

def load_manifest(manifest_path: Path) -> list[dict]:
    """Load feature-flags.yaml and return flag entries."""
    data = yaml.safe_load(manifest_path.read_text())
    return data.get("flags", [])

def audit(manifest_path: Path, env_path: Path, config_path: Path, today: date) -> list[Finding]:
    """Run all audit checks. Returns findings sorted by severity."""
    findings: list[Finding] = []
    manifest_entries = load_manifest(manifest_path)
    manifest_names = {e["name"] for e in manifest_entries}

    # Source flags from both .env.example and config.py comments
    source_flags = parse_env_flags(env_path) | parse_config_flags(config_path)

    # 1. Unregistered: in source but not manifest
    for flag in sorted(source_flags - manifest_names):
        findings.append(Finding(flag, "UNREGISTERED", "error",
            f"Flag in source but not in {manifest_path.name}"))

    # 2. Orphaned: in manifest but not in source (flag was removed from code)
    for flag in sorted(manifest_names - source_flags):
        findings.append(Finding(flag, "ORPHANED_MANIFEST", "warn",
            f"Flag in {manifest_path.name} but not found in source"))

    # 3. Lifecycle checks on manifest entries
    for entry in manifest_entries:
        name = entry["name"]
        created = date.fromisoformat(entry["created"])
        removal = entry.get("removal_date")
        age_days = (today - created).days
        status = entry.get("status", "unknown")

        if removal:
            removal_date = date.fromisoformat(removal)
            if removal_date < today and status != "deprecated":
                findings.append(Finding(name, "PAST_REMOVAL_DATE", "error",
                    f"Removal date {removal} has passed ({(today - removal_date).days}d ago)"))
        else:
            # No removal date — check if permanent_reason is provided
            reason = entry.get("permanent_reason")
            if not reason and age_days > ERROR_THRESHOLD_DAYS:
                findings.append(Finding(name, "NO_REMOVAL_PLAN_180D", "error",
                    f"Flag is {age_days}d old with no removal_date or permanent_reason"))
            elif not reason and age_days > WARN_THRESHOLD_DAYS:
                findings.append(Finding(name, "NO_REMOVAL_PLAN_90D", "warn",
                    f"Flag is {age_days}d old with no removal_date or permanent_reason"))

    return sorted(findings, key=lambda f: (0 if f.severity == "error" else 1, f.flag))
```

**Main function:**

```python
def main() -> int:
    parser = argparse.ArgumentParser(description="Audit feature flag lifecycle")
    parser.add_argument("--manifest", type=Path,
        default=_PROJECT_ROOT / "feature-flags.yaml")
    parser.add_argument("--env-example", type=Path,
        default=_PROJECT_ROOT / ".env.example")
    parser.add_argument("--config", type=Path,
        default=_PROJECT_ROOT / "app" / "core" / "config.py")
    parser.add_argument("--warn-only", action="store_true",
        help="Exit 0 even on errors (for initial rollout)")
    args = parser.parse_args()

    today = date.today()
    findings = audit(args.manifest, args.env_example, args.config, today)

    errors = [f for f in findings if f.severity == "error"]
    warnings = [f for f in findings if f.severity == "warn"]

    if warnings:
        print(f"⚠ {len(warnings)} warning(s):", file=sys.stderr)
        for f in warnings:
            print(f"  WARN  {f.code:30s} {f.flag}: {f.detail}", file=sys.stderr)

    if errors:
        print(f"✗ {len(errors)} error(s):", file=sys.stderr)
        for f in errors:
            print(f"  ERROR {f.code:30s} {f.flag}: {f.detail}", file=sys.stderr)

    total = len(findings)
    if total == 0:
        print(f"✓ Flag audit passed ({len(load_manifest(args.manifest))} flags registered)")
        return 0

    if errors and not args.warn_only:
        return 1
    return 0

if __name__ == "__main__":
    raise SystemExit(main())
```

**Key design decisions:**
- Checks BOTH `.env.example` AND `config.py` comments — catches flags that exist in code but not in .env.example
- `permanent_reason` field — flags intentionally kept forever must document why
- `--warn-only` flag — allows gradual rollout (add to `check` as warn-only first)
- `ORPHANED_MANIFEST` — catches flags removed from code but still in manifest
- No runtime imports — purely static analysis of files
