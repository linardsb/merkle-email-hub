# Plan: Workspace Agent Context Panel

## Context
The workspace needs a "Context" tab in the bottom panel (alongside Chat and Runs) that surfaces the active intelligence layers powering agent decisions. This gives developers visibility into what data agents see: audience profile, failure patterns, SKILL.md versions, and component context. Without this, the intelligence pipeline is a black box.

## Design

### Location
New **"Context"** tab in `BottomPanel` (third tab after Chat and Runs). Follows the existing tab bar pattern with `role="tablist"` and `aria-selected`.

### Layout
The context panel uses a 4-section accordion layout, each section collapsible:

```

│ 🎯 Audience Profile                              ▼     │
│ ┌─────────┬──────────┬──────────┬──────────┐           │
```

### Backend: Agent Skills Endpoint

A lightweight read-only endpoint that returns per-agent skill metadata. Located in `app/ai/routes.py` (existing AI routes file).

**Endpoint:** `GET /api/v1/agents/skills`

**Response:**
```json
{
  "agents": [
    {
      "name": "scaffolder",
      "skill_file": "SKILL.md",
      "l3_files": ["layout-patterns.md", "maizzle-specifics.md", "responsive-techniques.md", "table-architecture.md"],
      "has_failure_warnings": true
    },
    ...
  ]
}
```

**Implementation:** Scan `app/ai/agents/{agent}/` directories for SKILL.md and `skills/*.md` files. Check `traces/analysis.json` mtime for failure warnings flag.

### Step 1: Backend — Agent Skills Endpoint

**File: `app/ai/agents/skills_routes.py`**

```python
"""Read-only endpoint exposing per-agent skill metadata."""
from __future__ import annotations

import json
from pathlib import Path

from fastapi import APIRouter, Depends

from app.auth.dependencies import require_auth
from app.core.logging import get_logger

logger = get_logger(__name__)
router = APIRouter(prefix="/agents", tags=["agents"])

AGENT_NAMES = [
    "scaffolder", "dark_mode", "content", "outlook_fixer",
    "accessibility", "personalisation", "code_reviewer",
    "knowledge", "innovation",
]

AGENTS_DIR = Path(__file__).resolve().parent  # app/ai/agents/
TRACES_DIR = Path(__file__).resolve().parents[2] / "traces"


def _get_agent_skill_info(agent_name: str) -> dict:
    agent_dir = AGENTS_DIR / agent_name
    skill_file = agent_dir / "SKILL.md"
    skills_dir = agent_dir / "skills"

    l3_files: list[str] = []
    if skills_dir.is_dir():
        l3_files = sorted(f.name for f in skills_dir.glob("*.md"))

    # Check if failure warnings would fire for this agent
    has_failure_warnings = False
    analysis_path = TRACES_DIR / "analysis.json"
    if analysis_path.exists():
        try:
            data = json.loads(analysis_path.read_text())
            agent_data = data.get("per_agent", {}).get(agent_name, {})
            criteria = agent_data.get("per_criterion", {})
            has_failure_warnings = any(
                c.get("pass_rate", 1.0) < 0.85 for c in criteria.values()
            )
        except (json.JSONDecodeError, AttributeError):
            pass

    return {
        "name": agent_name,
        "skill_file": "SKILL.md" if skill_file.exists() else None,
        "l3_files": l3_files,
        "has_failure_warnings": has_failure_warnings,
    }


@router.get("/skills", dependencies=[Depends(require_auth)])
async def list_agent_skills() -> dict:
    """Return skill metadata for all agents."""
    agents = [_get_agent_skill_info(name) for name in AGENT_NAMES]
    return {"agents": agents}
```

### Step 4: Component Detection Utility

**File: `cms/apps/web/src/lib/detect-components.ts`**

```typescript
/**
 * Extract component slugs from editor HTML.
 * Matches `<component src="components/slug">` and `<component src="slug">` patterns.
 */
export function detectComponentRefs(html: string): string[] {
  const pattern = /<component\s+[^>]*src=["'](?:components\/)?([^"']+)["'][^>]*>/gi;
  const slugs = new Set<string>();
  let match: RegExpExecArray | null;
  while ((match = pattern.exec(html)) !== null) {
    slugs.add(match[1]);
  }
  return Array.from(slugs);
}
```

```tsx
// ── Collapsible Section ──
function Section({
  icon: Icon,
  title,
  badge,
  defaultOpen = true,
  children,
}: {
  icon: React.ComponentType<{ className?: string }>;
  title: string;
  badge?: string;
  defaultOpen?: boolean;
  children: React.ReactNode;
}) {
  const [open, setOpen] = useState(defaultOpen);

  return (

// ── Audience Section ──
function AudienceSection({ projectId }: { projectId: number }) {
  const t = useTranslations("workspace.contextPanel");
  const { data: brief, isLoading } = useCompatibilityBrief(projectId);

  if (isLoading) {
    return <p className="text-xs text-muted-foreground">{t("loading")}</p>;
  }

  if (!brief || !brief.clients || brief.clients.length === 0) {
    return <p className="text-xs text-muted-foreground">{t("audienceEmpty")}</p>;
  }

  return (
    <div className="space-y-2">
      {/* Client badges */}
      <div className="flex flex-wrap gap-1.5">
        {brief.clients.map((client: { client_id: string; name: string }) => (
          <span
            key={client.client_id}
            className="rounded-full bg-badge-default-bg px-2 py-0.5 text-[10px] font-medium text-badge-default-text"
          >
            {client.name}
          </span>
        ))}
      </div>
      {/* Summary pills */}
      <div className="flex flex-wrap gap-3 text-xs text-muted-foreground">
        {brief.dark_mode_required && (
          <span className="flex items-center gap-1">
            <Moon className="h-3 w-3" /> {t("darkModeRequired")}
          </span>
        )}
        {brief.mobile_viewports && brief.mobile_viewports.length > 0 && (
          <span className="flex items-center gap-1">
            <Smartphone className="h-3 w-3" />
            {t("mobileViewports", { count: brief.mobile_viewports.length })}
          </span>
        )}
        {brief.constraints && brief.constraints.length > 0 && (
          <span className="flex items-center gap-1">
            <Shield className="h-3 w-3" />
            {t("cssConstraints", { count: brief.constraints.length })}
          </span>
        )}
      </div>
    </div>
  );
}

// ── Failure Patterns Section ──
function FailurePatternsSection({ projectId }: { projectId: number }) {
  const t = useTranslations("workspace.contextPanel");
  const { data: stats, isLoading } = useFailurePatternStats(projectId);
  const { data: patterns } = useFailurePatterns({
    projectId,
    pageSize: 5,
  });

  if (isLoading) {
    return <p className="text-xs text-muted-foreground">{t("loading")}</p>;
  }

  if (!stats || stats.total_patterns === 0) {
    return <p className="text-xs text-muted-foreground">{t("noFailurePatterns")}</p>;
  }

  return (
    <div className="space-y-2">
      {/* Top patterns list */}
      {patterns?.items.slice(0, 5).map((p) => (
        <div key={p.id} className="flex items-start gap-2 text-xs">
          <AlertTriangle className="mt-0.5 h-3 w-3 shrink-0 text-status-warning" />
          <div>
            <span className="font-medium text-foreground">{p.agent_name}</span>
            <span className="text-muted-foreground"> → {p.qa_check}</span>
            {p.frequency > 1 && (
              <span className="text-muted-foreground"> (×{p.frequency})</span>
            )}
            {p.client_ids.length > 0 && (
              <span className="text-muted-foreground"> — {p.client_ids.join(", ")}</span>
            )}
          </div>
        </div>
      ))}
      {/* Summary */}
      <div className="flex gap-3 text-[10px] text-muted-foreground">
        {stats.top_agent && <span>{t("topAgent", { agent: stats.top_agent })}</span>}
        {stats.top_check && <span>{t("topCheck", { check: stats.top_check })}</span>}
      </div>
    </div>
  );
}

// ── Agent Skills Section ──
function AgentSkillsSection() {
  const t = useTranslations("workspace.contextPanel");
  const { data, isLoading } = useAgentSkills();

  if (isLoading) {
    return <p className="text-xs text-muted-foreground">{t("loading")}</p>;
  }

  if (!data || data.agents.length === 0) {
    return <p className="text-xs text-muted-foreground">{t("noSkillData")}</p>;
  }

  return (
    <div className="space-y-1">
      {data.agents.map((agent) => (
        <div key={agent.name} className="flex items-center gap-2 text-xs">
          <span className="w-28 truncate font-medium text-foreground">
            {agent.name}
          </span>
          {agent.skill_file ? (
            <span className="text-muted-foreground">
              SKILL.md
              {agent.l3_files.length > 0 && ` + ${agent.l3_files.length} L3`}
            </span>
          ) : (
            <span className="text-muted-foreground italic">{t("noSkill")}</span>
          )}
          {agent.has_failure_warnings && (
            <AlertTriangle className="h-3 w-3 text-status-warning" title={t("failureWarningsActive")} />
          )}
        </div>
      ))}
    </div>
  );
}

// ── Component Context Section ──
function ComponentContextSection({ editorContent }: { editorContent: string }) {
  const t = useTranslations("workspace.contextPanel");
  const refs = useMemo(() => detectComponentRefs(editorContent), [editorContent]);

  if (refs.length === 0) {
    return <p className="text-xs text-muted-foreground">{t("noComponentsDetected")}</p>;
  }

  return (
    <div className="flex flex-wrap gap-1.5">
      {refs.map((slug) => (
        <span
          key={slug}
          className="rounded bg-badge-default-bg px-2 py-0.5 text-[10px] font-medium text-badge-default-text"
        >
          {slug}
        </span>
      ))}
    </div>
  );
}

// ── Main Panel ──
export function AgentContextPanel({ projectId, editorContent }: AgentContextPanelProps) {

  const { data: stats } = useFailurePatternStats(projectId);

  return (
    <ScrollArea className="h-full">
      <div className="divide-y divide-border">
        <Section
          icon={Users}
          title={t("audienceProfile")}
          defaultOpen={true}
        >
          <AudienceSection projectId={projectId} />
        </Section>

        <Section
          icon={AlertTriangle}
          title={t("failurePatterns")}
          badge={stats?.total_patterns ? String(stats.total_patterns) : undefined}
          defaultOpen={true}
        >
          <FailurePatternsSection projectId={projectId} />
        </Section>

        <Section
          icon={BookOpen}
          title={t("agentSkills")}
          defaultOpen={false}
        >
          <AgentSkillsSection />
        </Section>

        <Section
          icon={Puzzle}
          title={t("componentContext")}
          defaultOpen={true}
        >
          <ComponentContextSection editorContent={editorContent} />
        </Section>
      </div>
    </ScrollArea>
  );
}
```
