```python
from __future__ import annotations
from app.design_sync.protocol import DesignNode, DesignNodeType, DesignFileStructure

def make_design_node(
    id: str = "node1",
    name: str = "Frame",
    type: DesignNodeType = DesignNodeType.FRAME,
    children: list[DesignNode] | None = None,
    **overrides: object,
) -> DesignNode:
    """Build a DesignNode with sensible defaults, override any field."""
    defaults = dict(
        id=id, name=name, type=type, children=children or [],
        width=600.0, height=400.0, x=0.0, y=0.0,
        visible=True, opacity=1.0,
    )
    defaults.update(overrides)
    return DesignNode(**defaults)  # type: ignore[arg-type]

def make_file_structure(*frames: DesignNode, file_name: str = "test") -> DesignFileStructure:
    page = make_design_node(id="page1", name="Page", type=DesignNodeType.PAGE, children=list(frames))
    return DesignFileStructure(file_name=file_name, pages=[page])
```

```python
"""Property-based tests for design_sync pipeline."""
from __future__ import annotations
import hypothesis.strategies as st
from hypothesis import given, settings, assume
from app.design_sync.protocol import DesignNode, DesignNodeType, DesignFileStructure
from app.design_sync.figma.tree_normalizer import normalize_tree
from app.design_sync.converter import sanitize_web_tags_for_email
import re

# --- Strategies ---

node_types = st.sampled_from(list(DesignNodeType))

@st.composite
def design_nodes(draw, max_depth=3):
    """Generate arbitrary DesignNode trees."""
    type_ = draw(node_types)
    depth = draw(st.integers(min_value=0, max_value=max_depth))
    children = []
    if depth > 0 and type_ in (DesignNodeType.FRAME, DesignNodeType.GROUP, DesignNodeType.PAGE):
        children = draw(st.lists(design_nodes(max_depth=depth - 1), max_size=5))
    return DesignNode(
        id=draw(st.text(min_size=1, max_size=10, alphabet=st.characters(whitelist_categories=("L", "N")))),
        name=draw(st.text(min_size=1, max_size=30)),
        type=type_,
        children=children,
        width=draw(st.floats(0.0, 2000.0, allow_nan=False)),
        height=draw(st.floats(0.0, 2000.0, allow_nan=False)),
        x=draw(st.floats(-500.0, 2000.0, allow_nan=False)),
        y=draw(st.floats(-500.0, 5000.0, allow_nan=False)),
        opacity=draw(st.floats(0.0, 1.0, allow_nan=False)),
        visible=draw(st.booleans()),
        text_content=draw(st.text(max_size=200)) if type_ == DesignNodeType.TEXT else None,
    )

# --- Property Tests ---

class TestOpacityPreservation:
    """Opacity=0.0 must survive the full pipeline."""

    @given(opacity=st.floats(0.0, 1.0, allow_nan=False))
    def test_opacity_roundtrip(self, opacity: float) -> None:
        node = DesignNode(id="n1", name="F", type=DesignNodeType.FRAME,
                          children=[], opacity=opacity, visible=True)
        assert node.opacity == opacity

    @given(opacity=st.floats(0.0, 1.0, allow_nan=False))
    def test_opacity_zero_not_dropped(self, opacity: float) -> None:
        node = DesignNode(id="n1", name="F", type=DesignNodeType.FRAME,
                          children=[], opacity=opacity, visible=True, width=100, height=50)
        page = DesignNode(id="p", name="Page", type=DesignNodeType.PAGE, children=[node])
        struct = DesignFileStructure(file_name="t", pages=[page])
        result, _stats = normalize_tree(struct)
        if opacity > 0.0:  # visible nodes preserved
            assert len(result.pages[0].children) == 1
            assert result.pages[0].children[0].opacity == opacity

class TestNormalizeNeverCrashes:
    """normalize_tree must never raise on valid DesignNode trees."""

    @given(tree=design_nodes(max_depth=3))
    @settings(max_examples=200)
    def test_normalize_no_crash(self, tree: DesignNode) -> None:
        page = DesignNode(id="p", name="Page", type=DesignNodeType.PAGE, children=[tree])
        struct = DesignFileStructure(file_name="t", pages=[page])
        result, stats = normalize_tree(struct)
        assert result is not None
        assert stats is not None

class TestSanitizeWebTags:
    """sanitize_web_tags_for_email produces balanced, email-safe HTML."""

    @given(st.text(max_size=500))
    def test_never_crashes(self, html: str) -> None:
        result = sanitize_web_tags_for_email(html)
        assert isinstance(result, str)

    @given(st.text(min_size=1, max_size=200))
    def test_no_layout_divs(self, content: str) -> None:
        """Output should not contain div with layout CSS."""

        result = sanitize_web_tags_for_email(html)
```

```python
"""Email HTML validity assertion and tests."""
from __future__ import annotations
import re, pathlib
from html.parser import HTMLParser
from dataclasses import dataclass, field
import pytest

GOLDEN_DIR = pathlib.Path(__file__).resolve().parents[2] / "ai" / "templates" / "library"

@dataclass
class EmailHtmlViolation:
    rule: str
    message: str
    element: str = ""

class EmailHtmlValidator(HTMLParser):
    """Validate HTML for email client compatibility."""

    LAYOUT_CSS = re.compile(r'(width|max-width|min-width|flex|float|display\s*:\s*flex|display\s*:\s*grid|position\s*:\s*(absolute|relative|fixed))', re.I)
    VOID_ELEMENTS = frozenset({"area","base","br","col","embed","hr","img","input","link","meta","param","source","track","wbr"})

    def __init__(self) -> None:
        super().__init__()
        self.violations: list[EmailHtmlViolation] = []
        self._in_mso_comment = False
        self._tag_stack: list[str] = []

    def handle_starttag(self, tag: str, attrs: list[tuple[str, str | None]]) -> None:
        tag_l = tag.lower()
        attr_dict = {k.lower(): (v or "") for k, v in attrs}

        if tag_l not in self.VOID_ELEMENTS:
            self._tag_stack.append(tag_l)

        # G1: tables must have role="presentation"
        if tag_l == "table" and attr_dict.get("role") != "presentation":
            self.violations.append(EmailHtmlViolation(
                "G1", 'Missing role="presentation"', f"<table id/class={attr_dict.get('class','')}>"))

        # G-REF-2: images must have display:block and meaningful alt
        if tag_l == "img":
            style = attr_dict.get("style", "")
            alt = attr_dict.get("alt", "")
            if "display:block" not in style.replace(" ", "").lower() and "display: block" not in style.lower():
                self.violations.append(EmailHtmlViolation(
                    "G-REF-2a", "Image missing display:block", f"<img alt={alt!r}>"))
            if not alt or alt in ("mj-image", "image", "img", "photo", "picture"):
                self.violations.append(EmailHtmlViolation(
                    "G-REF-2b", f"Image has non-meaningful alt={alt!r}", f"<img src=...>"))

        # No <div> with layout CSS (except in MSO comments)
        if tag_l == "div" and not self._in_mso_comment:
            style = attr_dict.get("style", "")
            if self.LAYOUT_CSS.search(style):
                # Allow div.column pattern (G-REF-1)
                classes = attr_dict.get("class", "")
                if "column" not in classes:
                    self.violations.append(EmailHtmlViolation(
                        "NO-DIV-LAYOUT", f"<div> with layout CSS: {style[:60]}", f"<div class={classes!r}>"))

    def handle_comment(self, data: str) -> None:
        if "[if mso]" in data or "[if gte mso" in data:
            self._in_mso_comment = True
        if "endif" in data:
            self._in_mso_comment = False

    def handle_endtag(self, tag: str) -> None:
        tag_l = tag.lower()
        if tag_l not in self.VOID_ELEMENTS and self._tag_stack and self._tag_stack[-1] == tag_l:
            self._tag_stack.pop()


def assert_valid_email_html(html: str, *, ignore_rules: set[str] | None = None) -> None:
    """Assert that HTML conforms to email best practices.

    Rules:
    - G1: Every <table> has role="presentation"
    - G-REF-2a: Every <img> has display:block
    - G-REF-2b: Every <img> has meaningful alt text
    - NO-DIV-LAYOUT: No <div> with layout CSS (width/flex/float)
    """
    ignore = ignore_rules or set()
    validator = EmailHtmlValidator()
    validator.feed(html)
    violations = [v for v in validator.violations if v.rule not in ignore]
    if violations:
        msg = "\n".join(f"  [{v.rule}] {v.message} — {v.element}" for v in violations)
        raise AssertionError(f"Email HTML violations:\n{msg}")


# --- Tests ---

class TestAssertValidEmailHtml:
    def test_valid_html_passes(self) -> None:
        html = '''<table role="presentation"><tr><td>
            <img src="x.png" alt="Product photo" style="display:block">
            <p style="margin:0">Hello</p>
        </td></tr></table>'''
        assert_valid_email_html(html)

    def test_missing_role_presentation(self) -> None:
        html = '<table><tr><td>Content</td></tr></table>'
        with pytest.raises(AssertionError, match="G1"):
            assert_valid_email_html(html)

    def test_img_missing_display_block(self) -> None:
        html = '<table role="presentation"><tr><td><img src="x.png" alt="Logo"></td></tr></table>'
        with pytest.raises(AssertionError, match="G-REF-2a"):
            assert_valid_email_html(html)

    def test_img_bad_alt_mj_image(self) -> None:
        html = '<table role="presentation"><tr><td><img src="x.png" alt="mj-image" style="display:block"></td></tr></table>'
        with pytest.raises(AssertionError, match="G-REF-2b"):
            assert_valid_email_html(html)

    def test_div_with_layout_css(self) -> None:
        html = '<table role="presentation"><tr><td><div style="width:300px;float:left">X</div></td></tr></table>'
        with pytest.raises(AssertionError, match="NO-DIV-LAYOUT"):
            assert_valid_email_html(html)

    def test_div_column_allowed(self) -> None:
        html = '<table role="presentation"><tr><td><div class="column" style="width:300px">X</div></td></tr></table>'
        assert_valid_email_html(html)  # column pattern OK

    def test_mso_comment_div_allowed(self) -> None:
        html = '''<table role="presentation"><tr><td>
            <!--[if mso]><div style="width:600px"><![endif]-->
            <p style="margin:0">Content</p>
            <!--[if mso]></div><![endif]-->
        </td></tr></table>'''
        assert_valid_email_html(html)

    def test_ignore_rules(self) -> None:
        html = '<table><tr><td>Content</td></tr></table>'
        assert_valid_email_html(html, ignore_rules={"G1"})


class TestGoldenTemplateValidity:
    """Validate golden templates against email HTML rules."""

    @pytest.mark.parametrize("template", sorted(GOLDEN_DIR.glob("*.html")), ids=lambda p: p.stem)
    def test_golden_template(self, template: pathlib.Path) -> None:
        html = template.read_text()
        # Golden templates may have some known exceptions
        assert_valid_email_html(html, ignore_rules={"G-REF-2b"})  # some templates use placeholder alt
```
