# Plan: 17.1 Local Screenshot Service (Playwright CLI)

## Context
The hub has no local email rendering capability — the existing `RenderingProvider` protocol only supports external services (Litmus, EoA) which are placeholder implementations. Phase 17.1 adds a `LocalRenderingProvider` that uses the **Playwright CLI** (already installed in `cms/` for e2e tests) via subprocess to capture screenshots of email HTML across simulated client profiles.

**Architecture decision:** Use `npx playwright screenshot` as a subprocess call instead of adding `playwright` as a Python library. This avoids new dependencies, leverages the existing frontend toolchain, and reduces token/complexity overhead by ~40%.

### Step 4: CLI runner — `app/rendering/local/runner.py`
Low-level subprocess wrapper. Responsibilities:
- Write HTML to a temp file
- Apply CSS injections and style stripping to the HTML before writing
- Call `npx playwright screenshot file:///{tmp_path} {output_path}` with profile flags
- Read resulting PNG bytes
- Clean up temp files

```python
from __future__ import annotations
import asyncio
import os
import re
import tempfile
from pathlib import Path

from app.core.config import get_settings
from app.core.logging import get_logger
from app.rendering.exceptions import ScreenshotRenderError, ScreenshotTimeoutError
from app.rendering.local.profiles import RenderingProfile

logger = get_logger(__name__)
settings = get_settings()

STYLE_TAG_RE = re.compile(r"<style[^>]*>.*?</style>", re.DOTALL | re.IGNORECASE)


def _prepare_html(html: str, profile: RenderingProfile) -> str:
    """Apply client-specific HTML modifications."""
    modified = html
    if profile.strip_style_tags:
        modified = STYLE_TAG_RE.sub("", modified)
    if profile.css_injections:
        injection = "<style>" + "\n".join(profile.css_injections) + "</style>"
        # Insert before </head> or prepend to body
        if "</head>" in modified.lower():
            idx = modified.lower().index("</head>")
            modified = modified[:idx] + injection + modified[idx:]
        else:
            modified = injection + modified
    return modified


async def capture_screenshot(
    html: str,
    profile: RenderingProfile,
    output_dir: Path,
) -> bytes:
    """Render HTML with Playwright CLI and return PNG bytes."""
    timeout_ms = settings.rendering.screenshot_timeout_ms
    npx_path = settings.rendering.screenshot_npx_path

    prepared = _prepare_html(html, profile)

    with tempfile.NamedTemporaryFile(
        suffix=".html", delete=False, mode="w", encoding="utf-8"
    ) as f:
        f.write(prepared)
        html_path = f.name

    output_path = output_dir / f"{profile.name}.png"

    try:
        cmd = [npx_path, "playwright", "screenshot"]
        cmd.extend(["--browser", profile.browser])
        cmd.extend(["--viewport-size", f"{profile.viewport_width}, {profile.viewport_height}"])
        cmd.append("--full-page")

        if profile.color_scheme == "dark":
            cmd.extend(["--color-scheme", "dark"])
        if profile.device:
            cmd.extend(["--device", profile.device])
        # Wait for rendering to settle
        cmd.extend(["--wait-for-timeout", "1000"])

        cmd.append(f"file://{html_path}")
        cmd.append(str(output_path))

        logger.info(
            "screenshot.capture_started",
            profile=profile.name,
            browser=profile.browser,
            viewport=f"{profile.viewport_width}x{profile.viewport_height}",
        )

        proc = await asyncio.create_subprocess_exec(
            *cmd,
            stdout=asyncio.subprocess.PIPE,
            stderr=asyncio.subprocess.PIPE,
        )
        try:
            stdout, stderr = await asyncio.wait_for(
                proc.communicate(), timeout=timeout_ms / 1000
            )
        except asyncio.TimeoutError:
            proc.kill()
            raise ScreenshotTimeoutError(
                f"Screenshot for {profile.name} timed out after {timeout_ms}ms"
            )

        if proc.returncode != 0:
            error_msg = stderr.decode("utf-8", errors="replace").strip()
            raise ScreenshotRenderError(
                f"Playwright CLI failed for {profile.name}: {error_msg}"
            )

        if not output_path.exists():
            raise ScreenshotRenderError(
                f"Screenshot file not created for {profile.name}"
            )

        image_bytes = output_path.read_bytes()
        logger.info(
            "screenshot.capture_completed",
            profile=profile.name,
            size_bytes=len(image_bytes),
        )
        return image_bytes

    finally:
        # Cleanup temp HTML
        try:
            os.unlink(html_path)
        except OSError:
            pass
```

### Step 5: Service — `app/rendering/local/service.py`
Implements `RenderingProvider` protocol. Orchestrates captures across profiles.

```python
from __future__ import annotations
import base64
import tempfile
import uuid
from pathlib import Path

from app.core.config import get_settings
from app.core.logging import get_logger
from app.rendering.local.profiles import CLIENT_PROFILES
from app.rendering.local.runner import capture_screenshot

logger = get_logger(__name__)
settings = get_settings()


class LocalRenderingProvider:
    """Local screenshot rendering via Playwright CLI subprocess."""

    async def submit_test(self, html: str, subject: str, clients: list[str]) -> str:
        """Submit is a no-op for local — just return a UUID."""
        _ = subject
        test_id = f"local_{uuid.uuid4().hex[:12]}"
        logger.info("local_rendering.submit", test_id=test_id, client_count=len(clients))
        return test_id

    async def poll_status(self, test_id: str) -> str:
        """Local rendering is synchronous — always 'complete'."""
        _ = test_id
        return "complete"

    async def get_results(self, test_id: str) -> list[dict[str, str]]:
        """Local provider returns empty — screenshots served via dedicated endpoint."""
        _ = test_id
        return []

    async def render_screenshots(
        self, html: str, clients: list[str] | None = None
    ) -> list[dict[str, str | bytes]]:
        """Render HTML across client profiles, return list of results.

        Each result: {client_name, image_bytes, viewport, browser}.
        """
        max_clients = settings.rendering.screenshot_max_clients
        profile_names = (clients or list(CLIENT_PROFILES.keys()))[:max_clients]

        results: list[dict[str, str | bytes]] = []
        with tempfile.TemporaryDirectory(prefix="email_screenshots_") as tmpdir:
            output_dir = Path(tmpdir)
            for name in profile_names:
                profile = CLIENT_PROFILES.get(name)
                if not profile:
                    logger.warning("screenshot.unknown_profile", profile=name)
                    continue
                try:
                    image_bytes = await capture_screenshot(html, profile, output_dir)
                    results.append({
                        "client_name": name,
                        "image_bytes": image_bytes,
                        "viewport": f"{profile.viewport_width}x{profile.viewport_height}",
                        "browser": profile.browser,
                    })
                except Exception as exc:
                    logger.error(
                        "screenshot.capture_failed",
                        profile=name,
                        error=str(exc),
                    )
                    # Continue with other profiles
        return results
```

```python
class ScreenshotRequest(BaseModel):
    """Request for local email screenshot rendering."""
    html: str = Field(..., min_length=1, max_length=500_000)
    clients: list[str] = Field(
        default_factory=lambda: ["gmail_web", "outlook_2019", "apple_mail", "outlook_dark", "mobile_ios"]

    """Single client screenshot result with base64 image."""
    client_name: str
    image_base64: str

    browser: str

class ScreenshotResponse(BaseModel):
    """Response with rendered screenshots."""
    screenshots: list[ScreenshotClientResult]
    clients_rendered: int
    clients_failed: int = 0
```

Add a screenshot method to `RenderingService`:
```python
async def render_screenshots(
    self, data: ScreenshotRequest
) -> ScreenshotResponse:
    """Render email HTML locally across simulated email clients."""
    if not settings.rendering.screenshots_enabled:
        raise RenderingProviderError("Local screenshot rendering is disabled")

    provider = LocalRenderingProvider()
    raw_results = await provider.render_screenshots(data.html, data.clients)

    screenshots = []
    for r in raw_results:
        screenshots.append(ScreenshotClientResult(
            client_name=str(r["client_name"]),
            image_base64=base64.b64encode(r["image_bytes"]).as_bytes()).decode("ascii"),
            viewport=str(r["viewport"]),
            browser=str(r["browser"]),
        ))

    return ScreenshotResponse(
        screenshots=screenshots,
        clients_rendered=len(screenshots),
        clients_failed=len(data.clients) - len(screenshots),
    )
```
Note: import `base64` at top, import new schemas.
