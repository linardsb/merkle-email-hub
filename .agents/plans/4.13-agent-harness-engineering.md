# Plan: 4.13 Agent Harness Engineering (Phase 2)

## Context

Task 4.13 defines 6 harness components that transform agents from single-pass generators into iterative, self-correcting production workers. Phase 1 (blueprint engine) is already complete — the engine provides QA-gated recovery routing with bounded self-correction.

**What's already implemented** (no work needed):
- Blueprint engine with QA gate → recovery router → fixer loop (bounded to 2 iterations)
- Progressive disclosure of L3 skill files via `detect_relevant_skills()` in all 9 agents
- Confidence scoring (0.0–1.0) with threshold-based routing to `needs_review`
- Handoff history tracking and memory persistence
- Model routing via `TaskTier` ("lightweight"/"standard"/"complex")
- 10-point QA gate system in `app/qa_engine/checks/`

**What's missing** (this plan implements):
1. **TAOR Execution Loop** — Agent-level iterative loop with think/act/observe/repeat
2. **PreCompletionChecklist** — Verification middleware before declaring success
3. **Linter-Gated Guardrails** — Deterministic HTML validation on agent output
4. **Progress Log** — Per-run structured log of agent decisions and actions
5. **Model-Agnostic Escalation** — Confidence-triggered model tier upgrade
6. **Brand Constraint Enforcement** — Hard "must-not" rules applied deterministically

**Design philosophy**: Build these as composable middleware in `BaseAgentService`, NOT as engine-level changes. The engine already handles inter-node routing. The harness wraps individual agent execution to make each agent internally iterative and self-verifying.

### Step 2: Guardrails (`app/ai/agents/harness/guardrails.py`)

Deterministic validation — no LLM reasoning involved.

```python
"""Deterministic guardrails for agent output validation.

These run BEFORE the output is returned, catching issues that
don't require LLM judgement — just parsing and pattern matching.
"""

from __future__ import annotations

import re

from app.ai.agents.harness.schemas import GuardrailResult, GuardrailVerdict
from app.core.logging import get_logger

logger = get_logger(__name__)


def validate_html_structure(html: str) -> GuardrailResult:
    """Check that HTML has basic structural integrity.

    Validates: DOCTYPE present, <html> tag, <head>/<body> structure,
    no unclosed critical tags. Does NOT validate full W3C compliance.
    """
    issues: list[str] = []

    if not html.strip():
        return GuardrailResult(
            name="html_structure",
            verdict=GuardrailVerdict.FAIL,
            message="Empty HTML output",
        )

    html_lower = html.lower()

    # DOCTYPE check (loose — agents may omit for fragments)
    if "<html" in html_lower and "<!doctype" not in html_lower:
        issues.append("Missing DOCTYPE declaration")

    # Paired tag check for critical structural tags
    critical_pairs = [("html", "</html>"), ("head", "</head>"), ("body", "</body>")]
    for open_tag, close_tag in critical_pairs:
        if f"<{open_tag}" in html_lower and close_tag not in html_lower:
            issues.append(f"Unclosed <{open_tag}> tag")

    # Check for broken MSO conditionals (common agent failure)
    mso_opens = len(re.findall(r"<!--\[if\s", html))
    mso_closes = len(re.findall(r"<!\[endif\]-->", html))
    if mso_opens != mso_closes:
        issues.append(
            f"Mismatched MSO conditionals: {mso_opens} opens vs {mso_closes} closes"
        )

    if issues:
        return GuardrailResult(
            name="html_structure",
            verdict=GuardrailVerdict.FAIL,
            message="; ".join(issues),
        )
    return GuardrailResult(name="html_structure", verdict=GuardrailVerdict.PASS)


def validate_no_truncation(html: str) -> GuardrailResult:
    """Detect if LLM output was likely truncated mid-tag.

    Common failure: LLM hits max_tokens and output ends mid-HTML.
    """
    stripped = html.rstrip()
    if not stripped:
        return GuardrailResult(
            name="truncation_check",
            verdict=GuardrailVerdict.PASS,
        )

    # Check for output ending mid-tag
    truncation_signals = [
        stripped.endswith("<"),
        stripped.endswith("</"),
        bool(re.search(r"<\w+[^>]*$", stripped)),  # unclosed tag at end
        stripped.endswith("<!--"),
    ]

    if any(truncation_signals):
        return GuardrailResult(
            name="truncation_check",
            verdict=GuardrailVerdict.FAIL,
            message="Output appears truncated mid-tag",
        )
    return GuardrailResult(name="truncation_check", verdict=GuardrailVerdict.PASS)


def validate_brand_constraints(
    html: str,
    must_nots: list[str] | None = None,
) -> GuardrailResult:
    """Enforce hard brand constraints (Must-Nots).

    Brand constraints are deterministic rules, NOT LLM-evaluated:
    - "Never use colour #FF0000" → regex match
    - "Never modify brand header" → pattern match
    - Custom per-project rules from brand_compliance config

    Args:
        html: The agent output HTML.
        must_nots: List of forbidden patterns (regex strings).
    """
    if not must_nots:
        return GuardrailResult(
            name="brand_constraints",
            verdict=GuardrailVerdict.PASS,
        )

    violations: list[str] = []
    for pattern in must_nots:
        try:
            if re.search(pattern, html, re.IGNORECASE):
                violations.append(f"Brand constraint violated: {pattern}")
        except re.error:
            logger.warning(
                "harness.brand_constraint_invalid_regex",
                pattern=pattern,
            )

    if violations:
        return GuardrailResult(
            name="brand_constraints",
            verdict=GuardrailVerdict.FAIL,
            message="; ".join(violations),
        )
    return GuardrailResult(name="brand_constraints", verdict=GuardrailVerdict.PASS)


def run_all_guardrails(
    html: str,
    *,
    brand_must_nots: list[str] | None = None,
) -> list[GuardrailResult]:
    """Run all deterministic guardrail checks on agent output.

    Returns list of results. Caller decides whether to retry or reject.
    """
    return [
        validate_html_structure(html),
        validate_no_truncation(html),
        validate_brand_constraints(html, brand_must_nots),
    ]
```

### Step 3: Pre-Completion Checklist (`app/ai/agents/harness/pre_completion.py`)

Intercepts "done" signals and forces a final verification pass.

```python
"""PreCompletionChecklist — final verification before declaring success.

Prevents the "Ralph Wiggum Loop" where agents declare victory without
testing. Runs lightweight deterministic checks against the original brief.
"""

from __future__ import annotations

from app.ai.agents.harness.guardrails import run_all_guardrails
from app.ai.agents.harness.schemas import (
    GuardrailResult,
    GuardrailVerdict,
    PreCompletionResult,
)
from app.core.logging import get_logger

logger = get_logger(__name__)


def check_output_not_empty(html: str) -> GuardrailResult:
    """Verify agent actually produced HTML output."""
    if not html or len(html.strip()) < 50:
        return GuardrailResult(
            name="output_not_empty",
            verdict=GuardrailVerdict.FAIL,
            message=f"Output too short ({len(html.strip())} chars, minimum 50)",
        )
    return GuardrailResult(name="output_not_empty", verdict=GuardrailVerdict.PASS)


def check_brief_alignment(html: str, brief: str) -> GuardrailResult:
    """Basic check that output relates to the brief.

    Deterministic heuristic: if brief mentions specific sections/components
    (e.g., "hero", "CTA", "footer"), check they appear in output.
    NOT an LLM call — just keyword matching.
    """
    if not brief:
        return GuardrailResult(name="brief_alignment", verdict=GuardrailVerdict.PASS)

    brief_lower = brief.lower()
    html_lower = html.lower()

    # Extract expected sections from brief
    expected_sections: list[str] = []
    section_keywords = {
        "hero": ["hero", "banner", "header image"],
        "cta": ["cta", "call to action", "button"],
        "footer": ["footer", "unsubscribe"],
        "header": ["header", "navigation", "nav"],
        "preheader": ["preheader", "preview text"],
    }

    for section, keywords in section_keywords.items():
        if any(kw in brief_lower for kw in keywords):
            expected_sections.append(section)

    if not expected_sections:
        return GuardrailResult(name="brief_alignment", verdict=GuardrailVerdict.PASS)

    missing: list[str] = []
    for section in expected_sections:
        # Check for section presence in HTML (class names, comments, or content)
        section_patterns = section_keywords[section]
        found = any(p in html_lower for p in section_patterns)
        if not found:
            missing.append(section)

    if missing:
        return GuardrailResult(
            name="brief_alignment",
            verdict=GuardrailVerdict.WARN,
            message=f"Brief mentions {', '.join(missing)} but not found in output",
        )
    return GuardrailResult(name="brief_alignment", verdict=GuardrailVerdict.PASS)


def run_pre_completion_checklist(
    html: str,
    brief: str,
    *,
    brand_must_nots: list[str] | None = None,
) -> PreCompletionResult:
    """Run the full pre-completion checklist.

    Combines guardrail checks + brief alignment + output validation.
    All checks are deterministic — no LLM calls.

    Returns:
        PreCompletionResult with pass/fail and individual check results.
    """
    checks: list[GuardrailResult] = []

    # 1. Output existence check
    checks.append(check_output_not_empty(html))

    # 2. All guardrail checks (HTML structure, truncation, brand)
    checks.extend(run_all_guardrails(html, brand_must_nots=brand_must_nots))

    # 3. Brief alignment check
    checks.append(check_brief_alignment(html, brief))

    # Determine overall pass/fail (FAIL = hard reject, WARN = pass with notes)
    failures = tuple(
        c.message for c in checks if c.verdict == GuardrailVerdict.FAIL
    )
    passed = len(failures) == 0

    if not passed:
        logger.info(
            "harness.pre_completion_failed",
            failure_count=len(failures),
            failures=list(failures),
        )

    return PreCompletionResult(
        passed=passed,
        checks=tuple(checks),
        failures=failures,
    )
```

### Step 4: Model Escalation (`app/ai/agents/harness/escalation.py`)

```python
"""Model-agnostic escalation — confidence-triggered tier upgrade.

When an agent's output confidence drops below threshold, the harness
escalates to a more capable model tier for the retry. Uses the
existing provider registry — no new LLM integration needed.
"""

from __future__ import annotations

from app.ai.agents.harness.schemas import EscalationReason
from app.ai.routing import TaskTier
from app.core.logging import get_logger

logger = get_logger(__name__)

# Tier upgrade path: lightweight → standard → complex
_TIER_UPGRADE: dict[TaskTier, TaskTier] = {
    "lightweight": "standard",
    "standard": "complex",
    "complex": "complex",  # already at max
}


def should_escalate(
    *,
    confidence: float | None,
    iteration: int,
    current_tier: TaskTier,
    confidence_threshold: float = 0.6,
) -> tuple[bool, EscalationReason | None]:
    """Determine if model tier should be escalated.

    Escalation triggers:
    1. Low confidence on first attempt → try stronger model
    2. Retry after failure (iteration > 0) and still at lightweight/standard

    Returns:
        (should_escalate, reason) tuple.
    """
    if current_tier == "complex":
        return False, None  # already at max

    # Low confidence trigger
    if confidence is not None and confidence < confidence_threshold:
        return True, EscalationReason.LOW_CONFIDENCE

    # Retry trigger — if we're retrying, escalate to increase success chances
    if iteration > 0 and current_tier != "complex":
        return True, EscalationReason.RETRY_FAILURE

    return False, None


def get_escalated_tier(current_tier: TaskTier) -> TaskTier:
    """Get the next tier up from the current one."""
    return _TIER_UPGRADE.get(current_tier, "complex")
```

### Step 5: Progress Log (`app/ai/agents/harness/progress_log.py`)

Structured logging for each harness execution.

```python
"""Progress log for harness-wrapped agent execution.

Records every phase of the TAOR loop with structured metadata,
enabling post-hoc analysis of agent behaviour and debugging.
"""

from __future__ import annotations

import time

from app.ai.agents.harness.schemas import HarnessLog, ProgressEntry, TAORPhase
from app.core.logging import get_logger

logger = get_logger(__name__)


class ProgressTracker:
    """Tracks progress through a harness execution.

    Usage:
        tracker = ProgressTracker("scaffolder", max_iterations=3)
        tracker.record_think(0, "Analyzing brief for layout complexity")
        tracker.record_act(0, "Generated 4200 chars HTML", tier="standard")
        tracker.record_observe(0, "QA: 8/10 passed, dark_mode failed", confidence=0.72)
        tracker.record_repeat(0, "Retrying with dark_mode fixes")
        log = tracker.to_log()
    """

    def __init__(self, agent_name: str, max_iterations: int = 3) -> None:
        self._agent_name = agent_name
        self._max_iterations = max_iterations
        self._entries: list[ProgressEntry] = []
        self._phase_start: float = 0.0
        self._escalated = False
        self._escalation_reason: str | None = None
        self._original_tier: str = ""
        self._final_tier: str = ""
        self._pre_completion_passed = True
        self._guardrail_failures: list[str] = []
        self._iteration_count = 0

    def _start_timer(self) -> None:
        self._phase_start = time.monotonic()

    def _elapsed_ms(self) -> float:
        return round((time.monotonic() - self._phase_start) * 1000, 1)

    def record_think(self, iteration: int, action: str) -> None:
        """Record the think phase (analyzing task, detecting skills)."""
        self._start_timer()
        self._entries.append(
            ProgressEntry(
                phase=TAORPhase.THINK,
                iteration=iteration,
                action=action,
                observation="",
            )
        )

    def record_act(
        self, iteration: int, action: str, *, tier: str = ""
    ) -> None:
        """Record the act phase (LLM call, HTML generation)."""
        self._entries.append(
            ProgressEntry(
                phase=TAORPhase.ACT,
                iteration=iteration,
                action=action,
                observation="",
                model_tier=tier,
                duration_ms=self._elapsed_ms(),
            )
        )
        self._start_timer()

    def record_observe(
        self,
        iteration: int,
        observation: str,
        *,
        confidence: float | None = None,
    ) -> None:
        """Record the observe phase (guardrails, pre-completion, QA)."""
        self._entries.append(
            ProgressEntry(
                phase=TAORPhase.OBSERVE,
                iteration=iteration,
                action="validate",
                observation=observation,
                confidence=confidence,
                duration_ms=self._elapsed_ms(),
            )
        )
        self._start_timer()

    def record_repeat(self, iteration: int, reason: str) -> None:
        """Record the repeat decision (why we're looping)."""
        self._entries.append(
            ProgressEntry(
                phase=TAORPhase.REPEAT,
                iteration=iteration,
                action="retry",
                observation=reason,
            )
        )
        self._iteration_count = iteration + 1

    def record_complete(self, iteration: int, summary: str) -> None:
        """Record successful completion."""
        self._entries.append(
            ProgressEntry(
                phase=TAORPhase.COMPLETE,
                iteration=iteration,
                action="done",
                observation=summary,
            )
        )
        self._iteration_count = iteration + 1

    def set_escalation(self, reason: str, original: str, final: str) -> None:
        """Record that model escalation occurred."""
        self._escalated = True
        self._escalation_reason = reason
        self._original_tier = original
        self._final_tier = final

    def set_pre_completion(self, passed: bool, failures: list[str]) -> None:
        """Record pre-completion checklist result."""
        self._pre_completion_passed = passed
        self._guardrail_failures = failures

    def to_log(self) -> HarnessLog:
        """Build the final harness log."""
        from app.ai.agents.harness.schemas import EscalationReason

        reason = None
        if self._escalation_reason:
            try:
                reason = EscalationReason(self._escalation_reason)
            except ValueError:
                pass

        return HarnessLog(
            agent_name=self._agent_name,
            iterations=self._iteration_count,
            max_iterations=self._max_iterations,
            escalated=self._escalated,
            escalation_reason=reason,
            original_tier=self._original_tier,
            final_tier=self._final_tier,
            pre_completion_passed=self._pre_completion_passed,
            guardrail_failures=self._guardrail_failures,
            entries=list(self._entries),
        )
```

### Step 6: TAOR Execution Loop (`app/ai/agents/harness/taor.py`)

The core iterative loop that wraps `BaseAgentService.process()`.

```python
"""TAOR execution loop — Think-Act-Observe-Repeat.

Wraps an agent's single-pass execution in an iterative loop that:
1. THINK: Detect skills, prepare context
2. ACT: Call LLM, generate output
3. OBSERVE: Run guardrails + pre-completion checklist
4. REPEAT: If validation fails, feed errors back and retry (up to max_iterations)
"""

from __future__ import annotations

from typing import Any

from app.ai.agents.harness.escalation import get_escalated_tier, should_escalate
from app.ai.agents.harness.guardrails import run_all_guardrails
from app.ai.agents.harness.pre_completion import run_pre_completion_checklist
from app.ai.agents.harness.progress_log import ProgressTracker
from app.ai.agents.harness.schemas import (
    GuardrailVerdict,
    HarnessConfig,
    HarnessLog,
)
from app.ai.routing import TaskTier
from app.ai.shared import extract_confidence
from app.core.logging import get_logger

logger = get_logger(__name__)


class TAORLoop:
    """Iterative execution wrapper for agent processing.

    Transforms a single-pass agent into an iterative, self-correcting
    processor. The loop runs guardrails and pre-completion checks after
    each attempt, feeding failures back for retry.

    Usage:
        loop = TAORLoop(config=HarnessConfig(max_iterations=3))
        result, log = await loop.execute(agent_service, request)
    """

    def __init__(self, config: HarnessConfig | None = None) -> None:
        self._config = config or HarnessConfig()

    async def execute(
        self,
        agent: Any,  # BaseAgentService subclass
        request: Any,
        *,
        brief: str = "",
        brand_must_nots: list[str] | None = None,
    ) -> tuple[Any, HarnessLog]:
        """Run the TAOR loop on an agent request.

        Args:
            agent: BaseAgentService instance.
            request: Agent-specific request object.
            brief: Original brief for pre-completion alignment check.
            brand_must_nots: Forbidden brand patterns for guardrails.

        Returns:
            (agent_response, harness_log) tuple.
        """
        tracker = ProgressTracker(agent.agent_name, self._config.max_iterations)
        current_tier: TaskTier = agent._get_model_tier(request)
        tracker._original_tier = current_tier  # noqa: SLF001
        tracker._final_tier = current_tier  # noqa: SLF001

        last_response: Any = None
        last_failures: list[str] = []

        for iteration in range(self._config.max_iterations):
            # ── THINK ──
            tracker.record_think(
                iteration,
                f"Iteration {iteration}: preparing with tier={current_tier}",
            )

            # Inject previous failures into request if retrying
            if iteration > 0 and last_failures:
                request = self._inject_harness_feedback(request, last_failures)

            # ── ACT ──
            tracker._start_timer()  # noqa: SLF001
            response = await agent.process(request)
            last_response = response

            html = self._extract_html(response)
            tracker.record_act(
                iteration,
                f"Generated {len(html)} chars",
                tier=current_tier,
            )

            # ── OBSERVE ──
            # Run guardrails
            guardrail_results = run_all_guardrails(
                html, brand_must_nots=brand_must_nots
            )
            guardrail_failures = [
                r.message for r in guardrail_results if r.verdict == GuardrailVerdict.FAIL
            ]

            # Run pre-completion checklist
            pre_completion = run_pre_completion_checklist(
                html, brief, brand_must_nots=brand_must_nots
            )

            confidence = extract_confidence(html)
            observation_parts: list[str] = []
            if guardrail_failures:
                observation_parts.append(f"Guardrails failed: {'; '.join(guardrail_failures)}")
            if not pre_completion.passed:
                observation_parts.append(
                    f"PreCompletion failed: {'; '.join(pre_completion.failures)}"
                )
            if confidence is not None:
                observation_parts.append(f"Confidence: {confidence:.2f}")

            tracker.record_observe(
                iteration,
                " | ".join(observation_parts) or "All checks passed",
                confidence=confidence,
            )

            all_failures = list(set(guardrail_failures) | set(pre_completion.failures))

            # ── DECIDE: REPEAT or COMPLETE ──
            if not all_failures:
                # All checks passed
                tracker.set_pre_completion(True, [])
                tracker.record_complete(iteration, "All guardrails and pre-completion passed")

                logger.info(
                    f"harness.{agent.agent_name}.completed",
                    iterations=iteration + 1,
                    confidence=confidence,
                    escalated=tracker._escalated,  # noqa: SLF001
                )
                return last_response, tracker.to_log()

            # Checks failed — decide whether to retry
            last_failures = all_failures

            if iteration < self._config.max_iterations - 1:
                # Check for model escalation
                if self._config.enable_escalation:
                    do_escalate, reason = should_escalate(
                        confidence=confidence,
                        iteration=iteration,
                        current_tier=current_tier,
                        confidence_threshold=self._config.confidence_escalation_threshold,
                    )
                    if do_escalate and reason is not None:
                        new_tier = get_escalated_tier(current_tier)
                        tracker.set_escalation(reason.value, current_tier, new_tier)
                        current_tier = new_tier
                        tracker._final_tier = current_tier  # noqa: SLF001

                        logger.info(
                            f"harness.{agent.agent_name}.escalated",
                            from_tier=tracker._original_tier,  # noqa: SLF001
                            to_tier=current_tier,
                            reason=reason.value,
                        )

                tracker.record_repeat(
                    iteration,
                    f"Retrying due to: {'; '.join(all_failures[:3])}",
                )
            else:
                # Max iterations reached — return last response with failures noted
                tracker.set_pre_completion(False, all_failures)
                tracker.record_complete(
                    iteration,
                    f"Max iterations reached with {len(all_failures)} unresolved failures",
                )

        logger.warning(
            f"harness.{agent.agent_name}.max_iterations_reached",
            iterations=self._config.max_iterations,
            unresolved=last_failures[:3],
        )
        return last_response, tracker.to_log()

    def _extract_html(self, response: Any) -> str:
        """Extract HTML from agent response (handles different response shapes)."""
        if hasattr(response, "html"):
            return response.html or ""
        if hasattr(response, "output"):
            return response.output or ""
        return ""

    def _inject_harness_feedback(self, request: Any, failures: list[str]) -> Any:
        """Inject harness feedback into the request for retry.

        Adds a harness_feedback field that BaseAgentService can read
        and include in the user message on retry.
        """
        # Use object.__setattr__ for frozen models, or direct set for mutable
        try:
            if hasattr(request, "model_copy"):
                # Pydantic model — create copy with extra field
                extra = getattr(request, "harness_feedback", [])
                return request.model_copy(
                    update={"harness_feedback": list(extra) + failures}
                )
            else:
                request.harness_feedback = failures  # type: ignore[attr-defined]
                return request
        except Exception:
            # If we can't inject, just return as-is
            return request
```
