# Plan: 12.4 Layout Analyzer & Brief Generator

## Context
Convert Figma document structure (`DesignFileStructure` from 12.1) into a Scaffolder-compatible campaign brief. Pure computation — no LLM calls, no database writes. The layout analyzer detects email sections, column layouts, text content, and image placeholders from the design tree. The brief generator transforms that analysis into structured markdown the Scaffolder can consume.

Requires two small protocol extensions: adding `x`/`y` position and `text_content` to `DesignNode`, plus updating the Figma provider to populate them.

## Design Decisions
1. **Column detection via x/y positions** — more robust than width-ratio inference; Figma's `absoluteBoundingBox` already provides x/y/width/height
2. **Text extraction via `text_content` field** — Figma returns `characters` on TEXT nodes; we surface it through the protocol
3. **Section detection** — name-based (case-insensitive pattern matching) + position-based fallback (topmost → header, bottommost → footer, large IMAGE child → hero)
4. **Preview endpoint** — lightweight GET route for the analyzed layout, so users can inspect/correct before brief generation
5. **Brief enrichment** — includes preheader detection, button styles, content hierarchy (H1/H2/body by font size), image dimensions, section spacing

In `_parse_node`, extract `x`/`y` from `absoluteBoundingBox` and `characters` from TEXT nodes:

```python
def _parse_node(self, node_data: dict[str, Any], current_depth: int, max_depth: int | None) -> DesignNode:
    raw_type = str(node_data.get("type", "UNKNOWN"))
    node_type = _FIGMA_NODE_TYPE_MAP.get(raw_type, DesignNodeType.OTHER)

    bbox = node_data.get("absoluteBoundingBox")
    width: float | None = None
    height: float | None = None
    x: float | None = None
    y: float | None = None
    if isinstance(bbox, dict):
        width = bbox.get("width")
        height = bbox.get("height")
        x = bbox.get("x")
        y = bbox.get("y")

    # Extract text content from TEXT nodes
    text_content: str | None = None
    if node_type == DesignNodeType.TEXT:
        raw_chars = node_data.get("characters")
        if isinstance(raw_chars, str) and raw_chars.strip():
            text_content = raw_chars.strip()

    children: list[DesignNode] = []
    if max_depth is None or current_depth < max_depth:
        for child_data in node_data.get("children", []):
            if isinstance(child_data, dict):
                children.append(self._parse_node(child_data, current_depth + 1, max_depth))

    return DesignNode(
        id=str(node_data.get("id", "")),
        name=str(node_data.get("name", "")),
        type=node_type,
        children=children,
        width=width,
        height=height,
        x=x,
        y=y,
        text_content=text_content,
    )
```

def _classify_by_name(name: str) -> EmailSectionType:
    """Match frame name against known email section patterns (case-insensitive)."""
    lower = name.lower().strip()
    for section_type, patterns in _SECTION_PATTERNS.items():
        for pattern in patterns:
            if pattern in lower:
                return section_type
    return EmailSectionType.UNKNOWN

    total: int,

### Step 6: Add service methods (`app/design_sync/service.py`)

Add two methods to `DesignSyncService`:

```python
async def analyze_layout(
    self,
    connection_id: int,
    user: User,
    *,
    selected_node_ids: list[str] | None = None,
    depth: int | None = None,
) -> LayoutAnalysisResponse:
    """Analyze layout of a design file and return detected sections.

    Fetches file structure (with depth=None for full tree, including
    text_content and x/y), runs LayoutAnalyzer, returns preview.
    """
    conn = await self._repo.get_connection(connection_id)
    if conn is None:
        raise ConnectionNotFoundError(f"Connection {connection_id} not found")
    if conn.project_id is not None:
        await self._verify_access(conn.project_id, user)

    provider = self._get_provider(conn.provider)
    access_token = decrypt_token(conn.encrypted_token)
    # Full depth to get all text content and positions
    structure = await provider.get_file_structure(conn.file_ref, access_token, depth=depth)

    # Filter to selected nodes if specified
    if selected_node_ids:
        structure = _filter_structure(structure, selected_node_ids)

    layout = analyze_layout(structure)
    return _layout_to_response(connection_id, layout)


async def generate_brief(
    self,
    connection_id: int,
    user: User,
    *,
    selected_node_ids: list[str] | None = None,
    include_tokens: bool = True,
) -> GenerateBriefResponse:
    """Generate a Scaffolder-compatible brief from design analysis.

    1. Fetch file structure (full depth)
    2. Optionally fetch design tokens
    3. Run LayoutAnalyzer
    4. Run BriefGenerator
    """
    conn = await self._repo.get_connection(connection_id)
    if conn is None:
        raise ConnectionNotFoundError(f"Connection {connection_id} not found")
    if conn.project_id is not None:
        await self._verify_access(conn.project_id, user)

    provider = self._get_provider(conn.provider)
    access_token = decrypt_token(conn.encrypted_token)
    structure = await provider.get_file_structure(conn.file_ref, access_token, depth=None)

    if selected_node_ids:
        structure = _filter_structure(structure, selected_node_ids)

    layout = analyze_layout(structure)

    tokens: ExtractedTokens | None = None
    if include_tokens:
        try:
            tokens = await provider.sync_tokens(conn.file_ref, access_token)
        except Exception:
            logger.warning("design_sync.brief_tokens_skipped", connection_id=connection_id)

    brief_text = generate_brief_text(
        layout,
        tokens=tokens,
        asset_url_prefix=f"/api/v1/design-sync/assets/{connection_id}",
        connection_id=connection_id,
    )

    sections_summary = ", ".join(s.section_type.value for s in layout.sections)

    return GenerateBriefResponse(
        connection_id=connection_id,
        brief=brief_text,
        sections_detected=len(layout.sections),
        layout_summary=sections_summary or "no sections detected",
    )
```

Helper function `_filter_structure` filters a `DesignFileStructure` to only include nodes whose IDs are in the selected set (keeping parent chain intact).
