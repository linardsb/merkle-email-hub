### Step 3: Create HTML sanitization module

```python
"""HTML sanitization for component uploads."""

import re

_SCRIPT_TAG_RE = re.compile(r"<script\b[^>]*>.*?</script>", re.DOTALL | re.IGNORECASE)
_EVENT_HANDLER_RE = re.compile(
    r"""\s+on\w+\s*=\s*(?:"[^"]*"|'[^']*'|[^\s>]*)""", re.IGNORECASE
)
_JS_PROTOCOL_RE = re.compile(r"""(href|src)\s*=\s*["']?\s*javascript:""", re.IGNORECASE)
_DANGEROUS_TAG_RE = re.compile(
    r"<(iframe|embed|object|form)\b[^>]*>.*?</\1>",
    re.DOTALL | re.IGNORECASE,
)
_DANGEROUS_SELF_CLOSING_RE = re.compile(
    r"<(iframe|embed|object)\b[^>]*/?>",
    re.IGNORECASE,
)
_DATA_URI_RE = re.compile(r"""(href|src)\s*=\s*["']?\s*data:""", re.IGNORECASE)


def sanitize_component_html(html: str) -> str:
    """Strip XSS vectors from component HTML.

    Removes script tags, event handlers, javascript: protocol,
    dangerous tags (iframe, embed, object, form), and data: URIs.
    Preserves MSO conditional comments needed for Outlook rendering.
    Preserves @media prefers-color-scheme rules for dark mode.

    Args:
        html: HTML string to sanitize.

    Returns:
        Sanitized HTML string.
    """
    result = html
    result = _SCRIPT_TAG_RE.sub("", result)
    result = _EVENT_HANDLER_RE.sub("", result)
    result = _JS_PROTOCOL_RE.sub(r'\1=""', result)
    result = _DANGEROUS_TAG_RE.sub("", result)
    result = _DANGEROUS_SELF_CLOSING_RE.sub("", result)
    result = _DATA_URI_RE.sub(r'\1=""', result)
    return result
```

async def create_component(self, data: ComponentCreate, user_id: int) -> ComponentResponse:
    logger.info("components.create_started", name=data.name)
    existing = await self.repository.get_by_slug(data.slug)
    if existing:
        raise ComponentAlreadyExistsError(f"Component with slug '{data.slug}' already exists")
    # Sanitize HTML before storage
    data.html_source = sanitize_component_html(data.html_source)
    if data.css_source:
        data.css_source = sanitize_component_html(data.css_source)
    component = await self.repository.create(data, user_id)
    resp = ComponentResponse.model_validate(component)
    resp.latest_version = 1
    return resp
```

async def create_version(
    self, component_id: int, data: VersionCreate, user_id: int
) -> VersionResponse:
    component = await self.repository.get(component_id)
    if not component:
        raise ComponentNotFoundError(f"Component {component_id} not found")
    # Sanitize HTML before storage
    data.html_source = sanitize_component_html(data.html_source)
    if data.css_source:
        data.css_source = sanitize_component_html(data.css_source)
    version = await self.repository.create_version(component_id, data, user_id)
    return VersionResponse.model_validate(version)

def upgrade() -> None:
    # Part A: Add compatibility column
    op.add_column("component_versions", sa.Column("compatibility", sa.JSON(), nullable=True))

    # Part B: Seed components
    # Use raw SQL inserts to avoid ORM dependency in migrations
    # Assumes user with id=1 exists (admin demo user from auth seeding)
    from app.components.data.seeds import COMPONENT_SEEDS

    components = sa.table(
        "components",
        sa.column("id", sa.Integer),
        sa.column("name", sa.String),
        sa.column("slug", sa.String),
        sa.column("description", sa.Text),
        sa.column("category", sa.String),
        sa.column("created_by_id", sa.Integer),
        sa.column("created_at", sa.DateTime),
        sa.column("updated_at", sa.DateTime),
    )

    versions = sa.table(
        "component_versions",
        sa.column("component_id", sa.Integer),
        sa.column("version_number", sa.Integer),
        sa.column("html_source", sa.Text),
        sa.column("css_source", sa.Text),
        sa.column("changelog", sa.Text),
        sa.column("compatibility", sa.JSON),
        sa.column("created_by_id", sa.Integer),
        sa.column("created_at", sa.DateTime),
        sa.column("updated_at", sa.DateTime),
    )

    now = datetime.now(UTC)
    for i, seed in enumerate(COMPONENT_SEEDS, start=1):
        op.execute(
            components.insert().values(
                name=seed["name"],
                slug=seed["slug"],
                description=seed["description"],
                category=seed["category"],
                created_by_id=1,
                created_at=now,
                updated_at=now,
            )
        )
        # Get the inserted component id using the slug
        conn = op.get_bind()
        result = conn.execute(
            sa.text("SELECT id FROM components WHERE slug = :slug"),
            {"slug": seed["slug"]},
        )
        component_id = result.scalar_one()
        op.execute(
            versions.insert().values(
                component_id=component_id,
                version_number=1,
                html_source=seed["html_source"],
                css_source=seed.get("css_source"),
                changelog="Initial seed version",
                compatibility=seed.get("compatibility"),
                created_by_id=1,
                created_at=now,
                updated_at=now,
            )
        )
```

def downgrade() -> None:
    # Remove seeded components by slug
    slugs = [
        "email-header", "email-footer", "cta-button", "hero-block",
        "product-card", "spacer", "social-icons", "image-block",

## Risks / Notes

- **Migration dependency on user id=1**: Seed migration assumes admin user exists. If running on empty DB, users migration must run first. This matches the existing personas pattern.
- **Seed data size**: 10 components with ~30-50 lines of HTML each. Migration file will be ~500-800 lines. This is expected for seed data.
- **No schema change to ComponentCreate**: The `compatibility` field is only on `VersionCreate` and `VersionResponse`, not on `ComponentCreate`. The initial version created via `ComponentCreate` won't have compatibility set (null). This is intentional — compatibility is version-level metadata that can be set when creating subsequent versions.
